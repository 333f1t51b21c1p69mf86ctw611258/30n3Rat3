select
	t1.ATTR_CONSTRAINT_TYPE_ID as attrConstraintTypeId,
	t1.NAME as name,
	t1.UPD_CNT as updCnt,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BSD_ATTR_CONSTRNT_TYPE_REF t1
	inner join BSD_ATTR_CONSTRNT_TYPE_VALUES t2 on t1.ATTR_CONSTRAINT_TYPE_ID = t2.ATTR_CONSTRAINT_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
