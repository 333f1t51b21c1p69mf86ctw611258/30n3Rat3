select
	t1.EXRATE_CLASS as exrateClass,
	t4.DISPLAY_VALUE as conversionDate,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	EXRATE_CLASS_REF t1
	inner join EXRATE_CLASS_VALUES t2 on t1.EXRATE_CLASS = t2.EXRATE_CLASS
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''CONVERSION_DATE'')
		and t3.value = t1.CONVERSION_DATE
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
