select
	t1.CONTAINERTYPE_ID as containertypeId,
	t1.BREAK_ON_ASSIGN as breakOnAssign,
	t1.BREAK_ON_DISC as breakOnDisc,
	t1.IS_DEFAULT as isDefault,
	t1.ALLOW_INDEPENDENT_TRANSITIONS as allowIndependentTransitions,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_CONTAINERTYPE_REF t1
	inner join INVS_CONTAINERTYPE_VALUES t2 on t1.CONTAINERTYPE_ID = t2.CONTAINERTYPE_ID
where
	t2.LANGUAGE_CODE = {2}
