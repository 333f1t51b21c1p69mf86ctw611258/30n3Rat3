select
	t1.BILL_DISP_METH as billDispMeth,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_DISP_METH_REF t1
	inner join BILL_DISP_METH_VALUES t2 on t1.BILL_DISP_METH = t2.BILL_DISP_METH
where
	t2.LANGUAGE_CODE = {2}
