select
	t1.ARCH_MODE_ID as archModeId,
	t1.IS_INTERNAL as isInternal,
	t1.IS_DEFAULT as isDefault,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARCH_MODE_REF t1
	inner join ARCH_MODE_VALUES t2 on t1.ARCH_MODE_ID = t2.ARCH_MODE_ID
where
	t2.LANGUAGE_CODE = {2}
