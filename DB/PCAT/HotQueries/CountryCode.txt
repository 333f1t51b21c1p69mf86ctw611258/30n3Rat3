select
	t1.COUNTRY_CODE as countryCode,
	t1.TAX_IDENTIFIER as taxIdentifier,
	t4.SERVICE_VERSION_ID as serviceVersionId,
	t4.ISO_CODE as currencyIsoCode,
	t4.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t5.DISPLAY_VALUE as rateCurrencyKey,
	t1.DEFAULT_CURRENCY_CODE as defaultCurrencyCode,
	t1.ISO_CODE as isoCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	COUNTRY_CODE_REF t1
	inner join COUNTRY_CODE_VALUES t2 on t1.COUNTRY_CODE = t2.COUNTRY_CODE
	inner join RATE_CURRENCY_KEY t3 on t1.DEFAULT_CURRENCY_CODE = t3.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t4 on t3.CURRENCY_CODE = t4.CURRENCY_CODE
		and t4.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t5 on t3.CURRENCY_CODE = t5.CURRENCY_CODE
		and t5.SERVICE_VERSION_ID = t4.SERVICE_VERSION_ID
		and t5.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
