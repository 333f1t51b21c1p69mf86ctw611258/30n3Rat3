select
	t3.DISPLAY_VALUE as invsSimCardConfigRef,
	t1.SIM_CARD_CONFIG_ID as simCardConfigId,
	t5.DISPLAY_VALUE as invsInterfaceRef,
	t1.INTERFACE_ID as interfaceId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal
from
	INVS_SIM_INTERFACE_ASSOC t1
	inner join INVS_SIM_CARD_CONFIG_REF t2 on t1.SIM_CARD_CONFIG_ID = t2.SIM_CARD_CONFIG_ID
	inner join INVS_SIM_CARD_CONFIG_VALUES t3 on t2.SIM_CARD_CONFIG_ID = t3.SIM_CARD_CONFIG_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_INTERFACE_REF t4 on t1.INTERFACE_ID = t4.INTERFACE_ID
	inner join INVS_INTERFACE_VALUES t5 on t4.INTERFACE_ID = t5.INTERFACE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
