select
	t1.MCCCODE_ID as mcccodeId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_MCCCODE_REF t1
	inner join INVS_MCCCODE_VALUES t2 on t1.MCCCODE_ID = t2.MCCCODE_ID
where
	t2.LANGUAGE_CODE = {2}
