select
	t1.NOTIF_IVR_LANGUAGE_ID as notifIvrLanguageId,
	t4.DISPLAY_VALUE as languageCodeRef,
	t1.BASE_LANGUAGE_CODE as baseLanguageCode,
	t1.LANGUAGE_LOCALE as languageLocale,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DESCRIPTION as description,
	t2.DISPLAY_VALUE as displayValue
from
	NOTIF_IVR_LANGUAGE_REF t1
	inner join NOTIF_IVR_LANGUAGE_VALUES t2 on t1.NOTIF_IVR_LANGUAGE_ID = t2.NOTIF_IVR_LANGUAGE_ID
	inner join LANGUAGE_CODE_REF t3 on t1.BASE_LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join LANGUAGE_CODE_VALUES t4 on t3.LANGUAGE_CODE = t4.INPUT_LANGUAGE
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
