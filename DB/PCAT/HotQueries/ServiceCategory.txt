select
	t1.SERVICE_CATEGORY_ID as serviceCategoryId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as externalIdTypeKey,
	t2.DISPLAY_EXTERNAL_ID_TYPE as displayExternalIdType,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t9.DISPLAY_VALUE as externalIdTypeKey,
	t2.PRIMARY_EXTERNAL_ID_TYPE as primaryExternalIdType,
	t12.DISPLAY_VALUE as externalIdTypeKey,
	t2.SECONDARY_EXTERNAL_ID_TYPE as secondaryExternalIdType,
	t15.DISPLAY_VALUE as externalIdTypeKey,
	t2.TERTIARY_EXTERNAL_ID_TYPE as tertiaryExternalIdType,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description,
        t2.AEND_ADDRESS_TYPE_ID as aendAddressTypeId,
        t2.BEND_ADDRESS_TYPE_ID as bendAddressTypeId
from
	SERVICE_CATEGORY_KEY t1
	inner join SERVICE_CATEGORY_REF t2 on t1.SERVICE_CATEGORY_ID = t2.SERVICE_CATEGORY_ID
	inner join SERVICE_CATEGORY_VALUES t3 on t2.SERVICE_CATEGORY_ID = t3.SERVICE_CATEGORY_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join EXTERNAL_ID_TYPE_KEY t4 on t2.DISPLAY_EXTERNAL_ID_TYPE = t4.EXTERNAL_ID_TYPE
	inner join EXTERNAL_ID_TYPE_REF t5 on t4.EXTERNAL_ID_TYPE = t5.EXTERNAL_ID_TYPE
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join EXTERNAL_ID_TYPE_VALUES t6 on t4.EXTERNAL_ID_TYPE = t6.EXTERNAL_ID_TYPE
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EXTERNAL_ID_TYPE_KEY t7 on t2.PRIMARY_EXTERNAL_ID_TYPE = t7.EXTERNAL_ID_TYPE
	left outer join EXTERNAL_ID_TYPE_REF t8 on t7.EXTERNAL_ID_TYPE = t8.EXTERNAL_ID_TYPE
		and t8.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	left outer join EXTERNAL_ID_TYPE_VALUES t9 on t7.EXTERNAL_ID_TYPE = t9.EXTERNAL_ID_TYPE
		and t9.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EXTERNAL_ID_TYPE_KEY t10 on t2.SECONDARY_EXTERNAL_ID_TYPE = t10.EXTERNAL_ID_TYPE
	left outer join EXTERNAL_ID_TYPE_REF t11 on t10.EXTERNAL_ID_TYPE = t11.EXTERNAL_ID_TYPE
		and t11.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	left outer join EXTERNAL_ID_TYPE_VALUES t12 on t10.EXTERNAL_ID_TYPE = t12.EXTERNAL_ID_TYPE
		and t12.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t12.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EXTERNAL_ID_TYPE_KEY t13 on t2.TERTIARY_EXTERNAL_ID_TYPE = t13.EXTERNAL_ID_TYPE
	left outer join EXTERNAL_ID_TYPE_REF t14 on t13.EXTERNAL_ID_TYPE = t14.EXTERNAL_ID_TYPE
		and t14.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	left outer join EXTERNAL_ID_TYPE_VALUES t15 on t13.EXTERNAL_ID_TYPE = t15.EXTERNAL_ID_TYPE
		and t15.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t15.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join ADDRESS_TYPE_REF t16 on t2.AEND_ADDRESS_TYPE_ID = t16.ADDRESS_TYPE_ID
	inner join ADDRESS_TYPE_VALUES t17 on t16.ADDRESS_TYPE_ID = t17.ADDRESS_TYPE_ID
		and t17.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ADDRESS_TYPE_REF t18 on t2.BEND_ADDRESS_TYPE_ID = t18.ADDRESS_TYPE_ID
	left outer join ADDRESS_TYPE_VALUES t19 on t18.ADDRESS_TYPE_ID = t19.ADDRESS_TYPE_ID
		and t19.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
