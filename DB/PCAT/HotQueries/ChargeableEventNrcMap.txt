select
	t1.CHARGEABLE_EVENT_NRC_MAP_ID as chargeableEventNrcMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as chargeableEventKey,
	t1.EVENT_ID as eventId,
        t12.DISPLAY_VALUE as channelId,
	t1.CHANNEL_ID as channelIdNum,
	t7.DISPLAY_VALUE as chargeableEventReasonKey,
	t1.REASON_ID as reasonId,
        t10.DISPLAY_VALUE as nrcTermKey,
	t1.NRC_TERM_ID as nrcTermId,
	t1.PROCEED_ON_ERROR as proceedOnError
from
	CHARGEABLE_EVENT_NRC_MAP t1
	inner join CHARGEABLE_EVENT_KEY t2 on t1.EVENT_ID = t2.EVENT_ID
	inner join CHARGEABLE_EVENT_REF t3 on t2.EVENT_ID = t3.EVENT_ID
		and t3.SERVICE_VERSION_ID = {1}
	inner join CHARGEABLE_EVENT_VALUES t4 on t2.EVENT_ID = t4.EVENT_ID
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join CHARGEABLE_EVENT_REASON_KEY t5 on t1.REASON_ID = t5.REASON_ID
	inner join CHARGEABLE_EVENT_REASON_REF t6 on t5.REASON_ID = t6.REASON_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join CHARGEABLE_EVENT_REASON_VALUES t7 on t5.REASON_ID = t7.REASON_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
        inner join NRC_TERM_KEY t8 on t1.NRC_TERM_ID = t8.NRC_TERM_ID
	inner join NRC_TERM_REF t9 on t8.NRC_TERM_ID = t9.NRC_TERM_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join NRC_TERM_VALUES t10 on t8.NRC_TERM_ID = t10.NRC_TERM_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t11 on t11.enumeration_key = lower(''CHARGEABLE_EVENT_CHANNEL_ID'')
		and t11.value = t1.CHANNEL_ID
	left outer join GENERIC_ENUMERATION_VALUES t12 on t12.enumeration_key = t11.enumeration_key
		and t12.value = t11.value
		and t12.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
