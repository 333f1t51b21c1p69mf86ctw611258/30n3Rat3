select
	t1.TAX_CODES_COMM_ID as taxCodesCommId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as chargeEltType,
	t1.CHARGE_ELT_ID as chargeEltId,
	t5.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as jurisdictionsKey,
	t1.JURISDICTION as jurisdiction,
	t8.DISPLAY_VALUE as taxCategory,
	t1.TAX_CODE as taxCode,
	t1.UTILITY_CODE as utilityCode,
	t1.RESALE_FLAG as resaleFlag,
	t10.DISPLAY_VALUE as taxUnitsIndicator,
	t1.TAX_UNIT_TYPE as taxUnitType,
        t13.DISPLAY_VALUE as unitsTypeKey
from
	TAX_CODES_COMM t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''CHARGE_ELT_TYPE'')
		and t2.value = t1.CHARGE_ELT_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
	left outer join JURISDICTIONS_KEY t4 on t1.JURISDICTION = t4.JURISDICTION
	left outer join JURISDICTIONS_REF t5 on t4.JURISDICTION = t5.JURISDICTION
		and t5.SERVICE_VERSION_ID = {1}
	left outer join JURISDICTIONS_VALUES t6 on t4.JURISDICTION = t6.JURISDICTION
		and t6.SERVICE_VERSION_ID = {1}
		and t6.LANGUAGE_CODE = {2}
	left outer join GUI_INDICATOR_REF t7 on t7.table_name = ''TAX_CODES_COMM''
		and t7.field_name = lower(''TAX_CATEGORY'')
		and t7.integer_value = t1.TAX_CATEGORY
	left outer join GUI_INDICATOR_VALUES t8 on t8.table_name = t7.table_name
		and t8.field_name = t7.field_name
		and t8.integer_value = t7.integer_value
		and t8.LANGUAGE_CODE = {2}
	left outer join GUI_INDICATOR_REF t9 on t9.table_name = ''TAX_CODES_COMM''
		and t9.field_name = lower(''TAX_UNITS_INDICATOR'')
		and t9.integer_value = t1.TAX_UNITS_INDICATOR
	left outer join GUI_INDICATOR_VALUES t10 on t10.table_name = t9.table_name
		and t10.field_name = t9.field_name
		and t10.integer_value = t9.integer_value
		and t10.LANGUAGE_CODE = {2}
        left outer join UNITS_TYPE_KEY t11 on t1.TAX_UNIT_TYPE = t11.UNIT_TYPE
        left outer join UNITS_TYPE_REF t12 on t1.TAX_UNIT_TYPE = t12.UNIT_TYPE
                and t12.SERVICE_VERSION_ID = t5.SERVICE_VERSION_ID
        left outer join UNITS_TYPE_VALUES t13 on t1.TAX_UNIT_TYPE = t13.UNIT_TYPE
                and t13.SERVICE_VERSION_ID = t5.SERVICE_VERSION_ID
                and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
