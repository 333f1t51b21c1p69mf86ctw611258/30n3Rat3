select
	t1.SERVICE_DATA_LOG_ID as serviceDataLogId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.ENTITY_ID1 as entityId1,
	t1.ENTITY_ID1_COLUMN as entityId1Column,
	t1.ENTITY_ID2 as entityId2,
	t1.ENTITY_ID2_COLUMN as entityId2Column,
	t1.TABLE_NAME as tableName,
	t1.COLUMN_NAME as columnName,
	t1.USER_ACTION as userAction,
	t1.USER_NAME as userName,
	t1.CHANGE_DT as changeDt,
	t1.OLD_VALUE as oldValue,
	t1.NEW_VALUE as newValue,
	t1.LANGUAGE_CODE as languageCode,
	t1.DISPLAY_VALUE as displayValue
from
	SERVICE_DATA_LOG t1
where
	t1.SERVICE_VERSION_ID = {1}
	and t1.RESELLER_VERSION_ID = {0}
