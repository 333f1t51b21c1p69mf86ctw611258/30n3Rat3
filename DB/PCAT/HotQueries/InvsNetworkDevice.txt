select
	t1.NETWORK_DEVICE_ID as networkDeviceId,
	t1.TIMEZONE as timezone,
	t4.DISPLAY_VALUE as invsRateCenterRef,
	t1.RATE_CENTER_ID as rateCenterId,
	t1.IS_CONTROLLER_EVENT_ID_ALPHA as isControllerEventIdAlpha,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_NETWORK_DEVICE_REF t1
	inner join INVS_NETWORK_DEVICE_VALUES t2 on t1.NETWORK_DEVICE_ID = t2.NETWORK_DEVICE_ID
	left outer join INVS_RATE_CENTER_REF t3 on t1.RATE_CENTER_ID = t3.RATE_CENTER_ID
	left outer join INVS_RATE_CENTER_VALUES t4 on t3.RATE_CENTER_ID = t4.RATE_CENTER_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
