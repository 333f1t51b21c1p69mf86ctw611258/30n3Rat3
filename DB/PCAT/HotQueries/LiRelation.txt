select
	t1.LI_RELATION_ID as liRelationId,
	t1.LOCATION_A as locationA,
	t1.LOCATION_B as locationB,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.LIA as lia,
	t1.LIB as lib,
	t1.HOME_TPPS_ID as homeTppsIdNum,
        t7.DISPLAY_VALUE as homeTppsId,
	t1.HOME_OPPS_ID as homeOppsIdNum,
        t4.DISPLAY_VALUE as homeOppsId,
	t1.ROAM_OPPS_ID as roamOppsId,
	t1.ROAM_TPPS_ID as roamTppsId,
	t1.NCF_OPPS_ID as ncfOppsId,
	t1.NCF_TPPS_ID as ncfTppsId,
	t1.USSD_CALLBACK_ID as ussdCallbackId,
	t1.LR_CHARACTERISTIC as lrCharacteristic,
	t1.GPRS as gprs,
	t1.GPRS_ROAM as gprsRoam
from
	LI_RELATION t1
        left outer join LI_RELATION_ID_KEY t2 on t1.HOME_OPPS_ID = t2.LI_RELATION_ID
        left outer join LI_RELATION_ID_REF t3 on t1.HOME_OPPS_ID = t3.LI_RELATION_ID
            and t1.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
        left outer join LI_RELATION_ID_VALUES t4 on t1.HOME_OPPS_ID = t4.LI_RELATION_ID
            and t1.SERVICE_VERSION_ID = t4.SERVICE_VERSION_ID
            and t4.LANGUAGE_CODE = {2}
        left outer join LI_RELATION_ID_KEY t5 on t1.HOME_TPPS_ID = t5.LI_RELATION_ID
        left outer join LI_RELATION_ID_REF t6 on t1.HOME_TPPS_ID = t6.LI_RELATION_ID
            and t1.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
        left outer join LI_RELATION_ID_VALUES t7 on t1.HOME_TPPS_ID = t7.LI_RELATION_ID
            and t1.SERVICE_VERSION_ID = t7.SERVICE_VERSION_ID
            and t4.LANGUAGE_CODE = t7.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
