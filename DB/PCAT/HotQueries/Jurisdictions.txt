select
	t1.JURISDICTION as jurisdiction,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as jurisdictionClassKey,
	t2.JURISDICTION_CLASS as jurisdictionClass,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	JURISDICTIONS_KEY t1
	inner join JURISDICTIONS_REF t2 on t1.JURISDICTION = t2.JURISDICTION
	inner join JURISDICTIONS_VALUES t3 on t2.JURISDICTION = t3.JURISDICTION
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	left outer join JURISDICTION_CLASS_KEY t4 on t2.JURISDICTION_CLASS = t4.JURISDICTION_CLASS
	left outer join JURISDICTION_CLASS_REF t5 on t4.JURISDICTION_CLASS = t5.JURISDICTION_CLASS
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	left outer join JURISDICTION_CLASS_VALUES t6 on t4.JURISDICTION_CLASS = t6.JURISDICTION_CLASS
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
