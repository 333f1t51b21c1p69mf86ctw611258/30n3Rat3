select
	t1.NOTIFICATION_QUEUE_ASSIGN_ID as notificationQueueAssignId,
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as externalInterfaceKey,
	t1.EXTERNAL_INTERFACE as externalInterface,
	t1.SDP as sdp,
	t1.PRIMARY_NA as primaryNa,
	t1.BACKUP_NA as backupNa,
	t1.SITE_ID as siteId
from
	NOTIFICATION_QUEUE_ASSIGN t1
	inner join EXTERNAL_INTERFACE_KEY t2 on t1.EXTERNAL_INTERFACE = t2.INTERFACE_ID
	inner join EXTERNAL_INTERFACE_REF t3 on t2.INTERFACE_ID = t3.INTERFACE_ID
		and t3.SERVICE_VERSION_ID = {1}
	inner join EXTERNAL_INTERFACE_VALUES t4 on t2.INTERFACE_ID = t4.INTERFACE_ID
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
