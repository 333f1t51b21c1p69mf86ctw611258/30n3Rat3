select
	t4.DISPLAY_VALUE as bundleKey,
	t1.BUNDLE_ID as bundleId,
	t7.DISPLAY_VALUE as rcTermKey,
	t1.RC_TERM_ID as rcTermId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.ASSOCIATION_TYPE as associationType,
	t1.OFFER_OVERRIDE as offerOverride,
	t9.DISPLAY_VALUE as billingFrequencyRef,
	t1.PERIOD_FREQUENCY as periodFrequency,
	t1.BILL_PERIOD as billPeriod,
	t1.FROM_DATE_OFFSET as fromDateOffset,
	t1.APPLY_DAY as applyDay,
	t1.ADVANCE_PERIODS as advancePeriods,
	t1.RECOVERY_DURATION as recoveryDuration,
	t11.DISPLAY_VALUE as balanceDistributionMethod,
	t1.IS_FOR_CONTRACT as isForContract,
	t1.CHARGE_ORDER as chargeOrder
from
	BUNDLE_RC_TERM_MAP t1
	inner join BUNDLE_KEY t2 on t1.BUNDLE_ID = t2.BUNDLE_ID
	inner join BUNDLE_REF t3 on t2.BUNDLE_ID = t3.BUNDLE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BUNDLE_VALUES t4 on t2.BUNDLE_ID = t4.BUNDLE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join RC_TERM_KEY t5 on t1.RC_TERM_ID = t5.RC_TERM_ID
	inner join RC_TERM_REF t6 on t5.RC_TERM_ID = t6.RC_TERM_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join RC_TERM_VALUES t7 on t5.RC_TERM_ID = t7.RC_TERM_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join BILLING_FREQUENCY_REF t8 on t1.PERIOD_FREQUENCY = t8.BILLING_FREQUENCY
	inner join BILLING_FREQUENCY_VALUES t9 on t8.BILLING_FREQUENCY = t9.BILLING_FREQUENCY
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t10 on t10.table_name = ''BUNDLE_RC_TERM_MAP''
		and t10.field_name = lower(''BALANCE_DISTRIBUTION_METHOD'')
		and t10.integer_value = t1.BALANCE_DISTRIBUTION_METHOD
	left outer join GUI_INDICATOR_VALUES t11 on t11.table_name = t10.table_name
		and t11.field_name = t10.field_name
		and t11.integer_value = t10.integer_value
		and t11.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
