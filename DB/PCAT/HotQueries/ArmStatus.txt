select
	t1.COLLECTION_STATUS as collectionStatus,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARM_STATUS_REF t1
	inner join ARM_STATUS_VALUES t2 on t1.COLLECTION_STATUS = t2.COLLECTION_STATUS
where
	t2.LANGUAGE_CODE = {2}
