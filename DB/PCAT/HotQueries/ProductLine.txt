select
	t1.PRODUCT_LINE_ID as productLineId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as productGroupKey,
	t2.PRODUCT_GROUP_ID as productGroupId,
	t2.FOREIGN_GROUP_CODE as foreignGroupCode,
	t2.USAGE_DATA_TABLE as usageDataTable,
	t2.USAGE_AUDIT_TABLE as usageAuditTable,
	t2.USAGE_FREE_TABLE as usageFreeTable,
	t2.USAGE_AUDIT_VIEW as usageAuditView,
	t2.USAGE_BAD_TABLE as usageBadTable,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	PRODUCT_LINE_KEY t1
	inner join PRODUCT_LINE_REF t2 on t1.PRODUCT_LINE_ID = t2.PRODUCT_LINE_ID
	inner join PRODUCT_LINE_VALUES t3 on t2.PRODUCT_LINE_ID = t3.PRODUCT_LINE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join PRODUCT_GROUP_KEY t4 on t2.PRODUCT_GROUP_ID = t4.PRODUCT_GROUP_ID
	inner join PRODUCT_GROUP_REF t5 on t4.PRODUCT_GROUP_ID = t5.PRODUCT_GROUP_ID
		and t5.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	inner join PRODUCT_GROUP_VALUES t6 on t4.PRODUCT_GROUP_ID = t6.PRODUCT_GROUP_ID
		and t6.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
