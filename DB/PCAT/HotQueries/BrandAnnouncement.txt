select
	t1.ANNOUNCEMENT_ID as announcementId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t5.DISPLAY_VALUE as notifIvrLanguageRef,
	t2.NOTIFICATION_LANGUAGE_ID as notificationLanguageId,
	t7.DISPLAY_VALUE as anncTypePromptRef,
	t2.ANNC_TYPE_ID as anncTypeId,
	t2.BRANDING_POINT as brandingPoint,
	t9.SERVICE_VERSION_ID as serviceVersionId,
	t10.DISPLAY_VALUE as brandKey,
	t2.BRANDING_ID as brandingId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	BRAND_ANNOUNCEMENT_KEY t1
	inner join BRAND_ANNOUNCEMENT_REF t2 on t1.ANNOUNCEMENT_ID = t2.ANNOUNCEMENT_ID
	inner join BRAND_ANNOUNCEMENT_VALUES t3 on t2.ANNOUNCEMENT_ID = t3.ANNOUNCEMENT_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join NOTIF_IVR_LANGUAGE_REF t4 on t2.NOTIFICATION_LANGUAGE_ID = t4.NOTIF_IVR_LANGUAGE_ID
	inner join NOTIF_IVR_LANGUAGE_VALUES t5 on t4.NOTIF_IVR_LANGUAGE_ID = t5.NOTIF_IVR_LANGUAGE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join ANNC_TYPE_PROMPT_REF t6 on t2.ANNC_TYPE_ID = t6.ANNC_TYPE_ID
	inner join ANNC_TYPE_PROMPT_VALUES t7 on t6.ANNC_TYPE_ID = t7.ANNC_TYPE_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join BRAND_KEY t8 on t2.BRANDING_ID = t8.BRANDING_ID
	inner join BRAND_REF t9 on t8.BRANDING_ID = t9.BRANDING_ID
		and t9.SERVICE_VERSION_ID = {1}
	inner join BRAND_VALUES t10 on t8.BRANDING_ID = t10.BRANDING_ID
		and t10.SERVICE_VERSION_ID = t9.SERVICE_VERSION_ID
		and t10.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
