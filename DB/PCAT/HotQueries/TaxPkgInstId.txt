select
	t1.TAX_PKG_INST_ID as taxPkgInstId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t5.DISPLAY_VALUE as taxPkgType,
	t2.PKG_ORDER as pkgOrder,
	t2.BASIC_DAYS as basicDays,
	t7.DISPLAY_VALUE as isBinned,
	t2.IS_DISTRIBUTED as isDistributed,
	t9.DISPLAY_VALUE as isFranchise,
	t2.IS_GUI_PRETAX as isGuiPretax,
	t2.EXCLUDE_CHARGE as excludeCharge,
	t11.DISPLAY_VALUE as roundingMethod,
	t13.DISPLAY_VALUE as taxType,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t15.DISPLAY_VALUE as isGraduated,
	t17.DISPLAY_VALUE as taxZeroCharge,
	t2.ROUNDING_FACTOR as roundingFactor,
	t2.BIN_ROUNDING as binRounding,
	t2.IS_LATE_FEE_EXEMPT as isLateFeeExempt,
	t19.DISPLAY_VALUE as isInclusive,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	TAX_PKG_INST_ID_KEY t1
	inner join TAX_PKG_INST_ID_REF t2 on t1.TAX_PKG_INST_ID = t2.TAX_PKG_INST_ID
	inner join TAX_PKG_INST_ID_VALUES t3 on t2.TAX_PKG_INST_ID = t3.TAX_PKG_INST_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''TAX_PKG_TYPE'')
		and t4.value = t2.TAX_PKG_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t6 on t6.enumeration_key = lower(''IS_BINNED'')
		and t6.value = t2.IS_BINNED
	left outer join GENERIC_ENUMERATION_VALUES t7 on t7.enumeration_key = t6.enumeration_key
		and t7.value = t6.value
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t8 on t8.enumeration_key = lower(''IS_FRANCHISE'')
		and t8.value = t2.IS_FRANCHISE
	left outer join GENERIC_ENUMERATION_VALUES t9 on t9.enumeration_key = t8.enumeration_key
		and t9.value = t8.value
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t10 on t10.enumeration_key = lower(''ROUNDING_METHOD'')
		and t10.value = t2.ROUNDING_METHOD
	left outer join GENERIC_ENUMERATION_VALUES t11 on t11.enumeration_key = t10.enumeration_key
		and t11.value = t10.value
		and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t12 on t12.enumeration_key = lower(''TAX_TYPE'')
		and t12.value = t2.TAX_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t13 on t13.enumeration_key = t12.enumeration_key
		and t13.value = t12.value
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t14 on t14.enumeration_key = lower(''IS_GRADUATED'')
		and t14.value = t2.IS_GRADUATED
	left outer join GENERIC_ENUMERATION_VALUES t15 on t15.enumeration_key = t14.enumeration_key
		and t15.value = t14.value
		and t15.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t16 on t16.enumeration_key = lower(''TAX_ZERO_CHARGE'')
		and t16.value = t2.TAX_ZERO_CHARGE
	left outer join GENERIC_ENUMERATION_VALUES t17 on t17.enumeration_key = t16.enumeration_key
		and t17.value = t16.value
		and t17.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t18 on t18.enumeration_key = lower(''IS_INCLUSIVE'')
		and t18.value = t2.IS_INCLUSIVE
	left outer join GENERIC_ENUMERATION_VALUES t19 on t19.enumeration_key = t18.enumeration_key
		and t19.value = t18.value
		and t19.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
