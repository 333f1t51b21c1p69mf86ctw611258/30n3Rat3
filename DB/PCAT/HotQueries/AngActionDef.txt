select
	t1.ANG_ACTION_DEF_ID as angActionDefId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.ANG_ACTION_ID as angActionId,
	t1.DESCRIPTION as description,
	t1.ACCOUNT_SESSION_REASON as accountSessionReason,
	t1.IS_ENABLED as isEnabled
from
	ANG_ACTION_DEF t1
where
	t1.RESELLER_VERSION_ID = {0}
