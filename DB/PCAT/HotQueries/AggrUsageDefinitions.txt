select
	t1.AGGR_USAGE_ID as aggrUsageId,
	t1.AGGR_USAGE_TYPE as aggrUsageType,
	t1.AGGR_USAGE_LEVEL as aggrUsageLevel,
	t1.AGGR_THRESHOLD_QUANTUM as aggrThresholdQuantum,
	t1.DEF_ORDER as defOrder,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.CREATE_DT as createDt,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	AGGR_USAGE_DEFINITIONS_REF t1
	inner join AGGR_USAGE_DEFINITIONS_VALUES t2 on t1.AGGR_USAGE_ID = t2.AGGR_USAGE_ID
where
	t2.LANGUAGE_CODE = {2}
