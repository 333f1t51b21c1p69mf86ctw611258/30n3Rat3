select
	t1.PROVIDER_ID as providerId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as providerClassKey,
	t2.PROVIDER_CLASS as providerClass,
	t2.MERCHANT_ACCOUNT_NO as merchantAccountNo,
	t2.MAIN_PROVIDER_FLAG as mainProviderFlag,
	t2.MERCHANT_ID as merchantId,
	t2.MERCHANT_CATEGORY_CODE as merchantCategoryCode,
	t2.SECURITY_CODE as securityCode,
	t2.MERCHANT_NAME as merchantName,
	t2.MERCHANT_CITY as merchantCity,
	t2.MERCHANT_STATE_PROV as merchantStateProv,
	t2.MERCHANT_ZIP as merchantZip,
	t2.MERCHANT_PHONE as merchantPhone,
	t2.AMEX_FLAG as amexFlag,
	t2.AMEX_DESCRIPTOR as amexDescriptor,
	t2.CLEARING_HOUSE_ID as clearingHouseId,
	t8.DISPLAY_VALUE as extSourceIdRef,
	t2.SOURCE_ID as sourceId,
	t2.HOME_PROVIDER_FLAG as homeProviderFlag,
	t2.BATCH_SEQ_NUM_OUT as batchSeqNumOut,
	t2.BATCH_SEQ_NUM_IN as batchSeqNumIn,
	t2.EXT_CONTACT_ID as extContactId,
	t2.COUNTRY_CODE as countryCode,
	t2.OPERATING_COMPANY_NUMBER as operatingCompanyNumber,
	t2.CUSTOMER_CARRIER_NAME_ABBR as customerCarrierNameAbbr,
	t2.CARRIER_ID_CODE as carrierIdCode,
	t2.ACCESS_CARRIER_NAME_ABBR as accessCarrierNameAbbr,
	t2.REVENUE_ACCOUNTING_OFFICE as revenueAccountingOffice,
	t2.PROVIDER_SERVICE_TYPE as providerServiceType,
	t2.SWITCHLESS_RESELLER as switchlessReseller,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t2.REGISTRATION as registration,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	SERVICE_PROVIDERS_KEY t1
	inner join SERVICE_PROVIDERS_REF t2 on t1.PROVIDER_ID = t2.PROVIDER_ID
	inner join SERVICE_PROVIDERS_VALUES t3 on t2.PROVIDER_ID = t3.PROVIDER_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join PROVIDER_CLASS_KEY t4 on t2.PROVIDER_CLASS = t4.PROVIDER_CLASS
	inner join PROVIDER_CLASS_REF t5 on t4.PROVIDER_CLASS = t5.PROVIDER_CLASS
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join PROVIDER_CLASS_VALUES t6 on t4.PROVIDER_CLASS = t6.PROVIDER_CLASS
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EXT_SOURCE_ID_REF t7 on t2.SOURCE_ID = t7.SOURCE_ID
	left outer join EXT_SOURCE_ID_VALUES t8 on t7.SOURCE_ID = t8.SOURCE_ID
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EXT_CONTACTS t9 on t2.EXT_CONTACT_ID = t9.EXT_CONTACT_ID
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
