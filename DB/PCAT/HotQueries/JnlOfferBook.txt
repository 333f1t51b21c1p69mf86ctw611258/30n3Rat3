select
	t1.OFFER_BOOK_ID as offerBookId,
	t1.G_L_CODE as gLCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	JNL_OFFER_BOOK_REF t1
	inner join JNL_OFFER_BOOK_VALUES t2 on t1.OFFER_BOOK_ID = t2.OFFER_BOOK_ID
where
	t2.LANGUAGE_CODE = {2}
