select
	t1.TEXT_ID as textId,
	t3.DISPLAY_VALUE as billTextGroupRef,
	t1.TEXT_GROUP_ID as textGroupId,
	t1.IS_INTERNAL as isInternal,
	t1.IS_HIDDEN as isHidden
from
	BILL_TEXT_GROUPS t1
	inner join BILL_TEXT_GROUP_REF t2 on t1.TEXT_GROUP_ID = t2.TEXT_GROUP_ID
	inner join BILL_TEXT_GROUP_VALUES t3 on t2.TEXT_GROUP_ID = t3.TEXT_GROUP_ID
		and t3.LANGUAGE_CODE = {2}
