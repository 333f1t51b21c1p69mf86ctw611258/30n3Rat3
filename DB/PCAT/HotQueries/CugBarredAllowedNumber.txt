select
	t1.CUG_NUMBER as cugNumber,
	t3.DISPLAY_VALUE as cugRef,
	t1.CUG_CODE as cugCode,
	t1.BARRED_ALLOWED_FLAG as barredAllowedFlag
from
	CUG_BARRED_ALLOWED_NUMBER t1
	inner join CUG_REF t2 on t1.CUG_CODE = t2.CUG_CODE
	inner join CUG_VALUES t3 on t2.CUG_CODE = t3.CUG_CODE
		and t3.LANGUAGE_CODE = {2}
