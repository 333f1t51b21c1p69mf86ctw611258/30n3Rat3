select
	t1.UNIT_CR_ID as unitCrId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.FOREIGN_CODE as foreignCode,
	t5.DISPLAY_VALUE as unitCrLevel,
	t7.DISPLAY_VALUE as unitCrDistrib,
	t10.DISPLAY_VALUE as usagePlanKey,
	t2.USAGE_PLAN_ID as usagePlanId,
	t13.DISPLAY_VALUE as productLineKey,
	t2.PRODUCT_LINE_ID as productLineId,
	t2.PROVIDER_ID as providerId,
	t2.FORMAT_CODE as formatCode,
	t2.DATE_ACTIVE as dateActive,
	t2.DATE_INACTIVE as dateInactive,
	t2.DATE_CREATED as dateCreated,
	t15.SERVICE_VERSION_ID as serviceVersionId,
	t16.DISPLAY_VALUE as ratableUnitClassKey,
	t2.RATABLE_UNIT_CLASS as ratableUnitClass,
	t18.DISPLAY_VALUE as unitCrDomain,
	t20.DISPLAY_VALUE as unitCrQuantum,
	t2.DOMAIN_ID as domainId,
	t2.MAXIMUM_ROLLOVER_UNITS as maximumRolloverUnits,
	t2.MAXIMUM_ROLLOVER_AMOUNT as maximumRolloverAmount,
	t2.ICB_ALLOW as icbAllow,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	UNIT_CR_ITEM_KEY t1
	inner join UNIT_CR_ITEM_REF t2 on t1.UNIT_CR_ID = t2.UNIT_CR_ID
	inner join UNIT_CR_ITEM_VALUES t3 on t2.UNIT_CR_ID = t3.UNIT_CR_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join GUI_INDICATOR_REF t4 on t4.table_name = ''UNIT_CR_ITEM_REF''
		and t4.field_name = lower(''UNIT_CR_LEVEL'')
		and t4.integer_value = t2.UNIT_CR_LEVEL
	left outer join GUI_INDICATOR_VALUES t5 on t5.table_name = t4.table_name
		and t5.field_name = t4.field_name
		and t5.integer_value = t4.integer_value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t6 on t6.table_name = ''UNIT_CR_ITEM_REF''
		and t6.field_name = lower(''UNIT_CR_DISTRIB'')
		and t6.integer_value = t2.UNIT_CR_DISTRIB
	left outer join GUI_INDICATOR_VALUES t7 on t7.table_name = t6.table_name
		and t7.field_name = t6.field_name
		and t7.integer_value = t6.integer_value
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join USAGE_PLAN_KEY t8 on t2.USAGE_PLAN_ID = t8.USAGE_PLAN_ID
	left outer join USAGE_PLAN_REF t9 on t8.USAGE_PLAN_ID = t9.USAGE_PLAN_ID
		and t9.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join USAGE_PLAN_VALUES t10 on t8.USAGE_PLAN_ID = t10.USAGE_PLAN_ID
		and t10.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join PRODUCT_LINE_KEY t11 on t2.PRODUCT_LINE_ID = t11.PRODUCT_LINE_ID
	left outer join PRODUCT_LINE_REF t12 on t11.PRODUCT_LINE_ID = t12.PRODUCT_LINE_ID
		and t12.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join PRODUCT_LINE_VALUES t13 on t11.PRODUCT_LINE_ID = t13.PRODUCT_LINE_ID
		and t13.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join RATABLE_UNIT_CLASS_KEY t14 on t2.RATABLE_UNIT_CLASS = t14.RATABLE_UNIT_CLASS
	inner join RATABLE_UNIT_CLASS_REF t15 on t14.RATABLE_UNIT_CLASS = t15.RATABLE_UNIT_CLASS
		and t15.SERVICE_VERSION_ID = {1}
	inner join RATABLE_UNIT_CLASS_VALUES t16 on t14.RATABLE_UNIT_CLASS = t16.RATABLE_UNIT_CLASS
		and t16.SERVICE_VERSION_ID = t15.SERVICE_VERSION_ID
		and t16.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t17 on t17.table_name = ''UNIT_CR_ITEM_REF''
		and t17.field_name = lower(''UNIT_CR_DOMAIN'')
		and t17.integer_value = t2.UNIT_CR_DOMAIN
	left outer join GUI_INDICATOR_VALUES t18 on t18.table_name = t17.table_name
		and t18.field_name = t17.field_name
		and t18.integer_value = t17.integer_value
		and t18.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t19 on t19.table_name = ''UNIT_CR_ITEM_REF''
		and t19.field_name = lower(''UNIT_CR_QUANTUM'')
		and t19.integer_value = t2.UNIT_CR_QUANTUM
	left outer join GUI_INDICATOR_VALUES t20 on t20.table_name = t19.table_name
		and t20.field_name = t19.field_name
		and t20.integer_value = t19.integer_value
		and t20.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
