select
	t1.POINT_REGION_ID as pointRegionId,
	t1.POINT_REGION as pointRegion,
	t6.DISPLAY_VALUE as pointRegionKey,
	t2.POINT_REGION as pointRegion,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	POINT_REGION_KEY t1
	inner join POINT_REGION_REF t2 on t1.POINT_REGION_ID = t2.POINT_REGION_ID
	inner join POINT_REGION_VALUES t3 on t2.POINT_REGION_ID = t3.POINT_REGION_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join POINT_REGION_KEY t4 on t2.POINT_REGION = t4.POINT_REGION
	inner join POINT_REGION_REF t5 on t4.POINT_REGION = t5.POINT_REGION
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join POINT_REGION_VALUES t6 on t4.POINT_REGION = t6.POINT_REGION
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
