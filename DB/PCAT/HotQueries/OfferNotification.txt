select
	t1.NOTIFICATION_ID as notificationId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as eventKey,
	t2.EVENT_ID as eventId,
	t8.SERVICE_VERSION_ID as serviceVersionId,
	t9.DISPLAY_VALUE as externalInterfaceKey,
	t2.EXT_INTERFACE_ID as extInterfaceId,
	t12.DISPLAY_VALUE as offerKey,
	t2.OFFER_ID as offerId,
	t14.DISPLAY_VALUE as notifIvrLanguageRef,
	t2.NOTIFICATION_LANGUAGE_ID as notificationLanguageId,
	t2.LATCH as latch,
	t2.DISPLAY_POINTER as displayPointer,
	t2.MESSAGE as message,
	t2.UNICODE_FLAG as unicodeFlag,
	t2.PRIORITY as priority,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	OFFER_NOTIFICATION_KEY t1
	inner join OFFER_NOTIFICATION_REF t2 on t1.NOTIFICATION_ID = t2.NOTIFICATION_ID
	inner join OFFER_NOTIFICATION_VALUES t3 on t2.NOTIFICATION_ID = t3.NOTIFICATION_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join EVENT_KEY t4 on t2.EVENT_ID = t4.EVENT_ID
	inner join EVENT_REF t5 on t4.EVENT_ID = t5.EVENT_ID
		and t5.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	inner join EVENT_VALUES t6 on t4.EVENT_ID = t6.EVENT_ID
		and t6.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join EXTERNAL_INTERFACE_KEY t7 on t2.EXT_INTERFACE_ID = t7.INTERFACE_ID
	inner join EXTERNAL_INTERFACE_REF t8 on t7.INTERFACE_ID = t8.INTERFACE_ID
		and t8.SERVICE_VERSION_ID = {1}
	inner join EXTERNAL_INTERFACE_VALUES t9 on t7.INTERFACE_ID = t9.INTERFACE_ID
		and t9.SERVICE_VERSION_ID = t8.SERVICE_VERSION_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join OFFER_KEY t10 on t2.OFFER_ID = t10.OFFER_ID
	inner join OFFER_REF t11 on t10.OFFER_ID = t11.OFFER_ID
		and t11.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	inner join OFFER_VALUES t12 on t10.OFFER_ID = t12.OFFER_ID
		and t12.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t12.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join NOTIF_IVR_LANGUAGE_REF t13 on t2.NOTIFICATION_LANGUAGE_ID = t13.NOTIF_IVR_LANGUAGE_ID
	inner join NOTIF_IVR_LANGUAGE_VALUES t14 on t13.NOTIF_IVR_LANGUAGE_ID = t14.NOTIF_IVR_LANGUAGE_ID
		and t14.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
