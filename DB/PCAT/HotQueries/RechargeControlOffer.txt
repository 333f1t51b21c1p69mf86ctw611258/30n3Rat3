select
	t4.DISPLAY_VALUE as rechargeControlKey,
	t1.SUPPLMTR_OFFER_KEY as supplmtrOfferKey,
	t1.DURATION as duration,
	t1.OFFER_INACTIVE_DT as offerInactiveDt,
	t1.OFFER_ACTIVE_DT as offerActiveDt,
	t1.ADD_DEL_FLAG as addDelFlag,
	t1.IS_EXPLICIT as isExplicit,
	t6.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId
from
	RECHARGE_CONTROL_OFFER t1
	inner join RECHARGE_CONTROL_KEY t2 on t1.SUPPLMTR_OFFER_KEY = t2.RCT_ID
	inner join RECHARGE_CONTROL_REF t3 on t2.RCT_ID = t3.RCT_ID
	inner join RECHARGE_CONTROL_VALUES t4 on t2.RCT_ID = t4.RCT_ID
		and t4.LANGUAGE_CODE = {2}
	inner join OFFER_KEY t5 on t1.OFFER_ID = t5.OFFER_ID
	inner join OFFER_REF t6 on t5.OFFER_ID = t6.OFFER_ID
		and t6.RESELLER_VERSION_ID = {0}
	inner join OFFER_VALUES t7 on t5.OFFER_ID = t7.OFFER_ID
		and t7.RESELLER_VERSION_ID = t6.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
