select
	t1.DISCONNECT_REASON as disconnectReason,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	DISCONNECT_REASON_REF t1
	inner join DISCONNECT_REASON_VALUES t2 on t1.DISCONNECT_REASON = t2.DISCONNECT_REASON
where
	t2.LANGUAGE_CODE = {2}
