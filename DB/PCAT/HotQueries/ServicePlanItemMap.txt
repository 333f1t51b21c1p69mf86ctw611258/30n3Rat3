select
	t1.SERV_PLAN_ITEM_MAP_ID as servPlanItemMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as servicePlanKey,
	t1.SERVICE_PLAN_ID as servicePlanId,
	t7.DISPLAY_VALUE as serviceItemKey,
	t1.SERVICE_ITEM_ID as serviceItemId,
	t9.SERVICE_VERSION_ID as serviceVersionId,
	t10.DISPLAY_VALUE as provisioningItemKey,
	t1.PROVISIONING_ITEM_ID as provisioningItemId
from
	SERVICE_PLAN_ITEM_MAP t1
	inner join SERVICE_PLAN_KEY t2 on t1.SERVICE_PLAN_ID = t2.SERVICE_PLAN_ID
	inner join SERVICE_PLAN_REF t3 on t2.SERVICE_PLAN_ID = t3.SERVICE_PLAN_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join SERVICE_PLAN_VALUES t4 on t2.SERVICE_PLAN_ID = t4.SERVICE_PLAN_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join SERVICE_ITEM_KEY t5 on t1.SERVICE_ITEM_ID = t5.SERVICE_ITEM_ID
	left outer join SERVICE_ITEM_REF t6 on t5.SERVICE_ITEM_ID = t6.SERVICE_ITEM_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join SERVICE_ITEM_VALUES t7 on t5.SERVICE_ITEM_ID = t7.SERVICE_ITEM_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join PROVISIONING_ITEM_KEY t8 on t1.PROVISIONING_ITEM_ID = t8.PROVISIONING_ITEM_ID
	left outer join PROVISIONING_ITEM_REF t9 on t8.PROVISIONING_ITEM_ID = t9.PROVISIONING_ITEM_ID
		and t9.SERVICE_VERSION_ID = {1}
	left outer join PROVISIONING_ITEM_VALUES t10 on t8.PROVISIONING_ITEM_ID = t10.PROVISIONING_ITEM_ID
		and t10.SERVICE_VERSION_ID = {1}
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
