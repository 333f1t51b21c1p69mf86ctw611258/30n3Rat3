select
	t1.ENUMERATION_KEY as enumerationKey,
	t1.VALUE as value,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.SORT_INDEX as sortIndex,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	GENERIC_ENUMERATION_REF t1
	inner join GENERIC_ENUMERATION_VALUES t2 on t1.ENUMERATION_KEY = t2.ENUMERATION_KEY
		and t1.VALUE = t2.VALUE
where
	t2.LANGUAGE_CODE = {2}
