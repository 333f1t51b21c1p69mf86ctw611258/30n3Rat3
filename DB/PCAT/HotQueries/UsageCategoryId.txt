select
	t1.USAGE_CATEGORY_ID as usageCategoryId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	USAGE_CATEGORY_ID_KEY t1
	inner join USAGE_CATEGORY_ID_REF t2 on t1.USAGE_CATEGORY_ID = t2.USAGE_CATEGORY_ID
	inner join USAGE_CATEGORY_ID_VALUES t3 on t2.USAGE_CATEGORY_ID = t3.USAGE_CATEGORY_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
