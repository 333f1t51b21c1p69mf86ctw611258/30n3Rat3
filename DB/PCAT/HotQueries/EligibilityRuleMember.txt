select
	t1.ELIGIBILITY_RULE_MEMBER_ID as eligibilityRuleMemberId,
	t4.DISPLAY_VALUE as eligibilityRuleKey,
	t1.ELIGIBILITY_RULE_ID as eligibilityRuleId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.ELIGIBILITY_RULE as eligibilityRule
from
	ELIGIBILITY_RULE_MEMBER t1
	inner join ELIGIBILITY_RULE_KEY t2 on t1.ELIGIBILITY_RULE_ID = t2.ELIGIBILITY_RULE_ID
	inner join ELIGIBILITY_RULE_REF t3 on t2.ELIGIBILITY_RULE_ID = t3.ELIGIBILITY_RULE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join ELIGIBILITY_RULE_VALUES t4 on t2.ELIGIBILITY_RULE_ID = t4.ELIGIBILITY_RULE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
