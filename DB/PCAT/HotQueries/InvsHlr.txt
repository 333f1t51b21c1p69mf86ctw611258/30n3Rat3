select
	t1.HLR_ID as hlrId,
	t4.DISPLAY_VALUE as invsLocationRef,
	t1.LOCATION_ID as locationId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_HLR_REF t1
	inner join INVS_HLR_VALUES t2 on t1.HLR_ID = t2.HLR_ID
	inner join INVS_LOCATION_REF t3 on t1.LOCATION_ID = t3.LOCATION_ID
	inner join INVS_LOCATION_VALUES t4 on t3.LOCATION_ID = t4.LOCATION_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
