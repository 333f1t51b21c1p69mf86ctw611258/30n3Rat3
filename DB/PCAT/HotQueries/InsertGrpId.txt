select
	t1.INSERT_GRP_ID as insertGrpId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.MAX_INSERTS as maxInserts,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INSERT_GRP_ID_REF t1
	inner join INSERT_GRP_ID_VALUES t2 on t1.INSERT_GRP_ID = t2.INSERT_GRP_ID
where
	t2.LANGUAGE_CODE = {2}
