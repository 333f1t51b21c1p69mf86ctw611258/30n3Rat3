select
	t1.LATE_FEE_GRP_ID as lateFeeGrpId,
	t3.DISPLAY_VALUE as vipCodeRef,
	t1.VIP_CODE as vipCode,
	t5.DISPLAY_VALUE as mktCodeRef,
	t1.MKT_CODE as mktCode,
	t1.FRANCHISE_CODE as franchiseCode,
	t1.COUNTY_CODE as countyCode,
	t1.CITY_CODE as cityCode,
	t7.DISPLAY_VALUE as countryCodeRef,
	t1.COUNTRY_CODE as countryCode,
	t1.STATE_CODE as stateCode,
	t9.SERVICE_VERSION_ID as serviceVersionId,
	t10.DISPLAY_VALUE as accountCategoryKey,
	t1.ACCOUNT_CATEGORY as accountCategory,
	t13.DISPLAY_VALUE as rateClassKey,
	t1.RATE_CLASS as rateClass
from
	LATE_FEE_ASSIGNMENT t1
	inner join VIP_CODE_REF t2 on t1.VIP_CODE = t2.VIP_CODE
	inner join VIP_CODE_VALUES t3 on t2.VIP_CODE = t3.VIP_CODE
		and t3.LANGUAGE_CODE = {2}
	left outer join MKT_CODE_REF t4 on t1.MKT_CODE = t4.MKT_CODE
	left outer join MKT_CODE_VALUES t5 on t4.MKT_CODE = t5.MKT_CODE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join COUNTRY_CODE_REF t6 on t1.COUNTRY_CODE = t6.COUNTRY_CODE
	left outer join COUNTRY_CODE_VALUES t7 on t6.COUNTRY_CODE = t7.COUNTRY_CODE
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join ACCOUNT_CATEGORY_KEY t8 on t1.ACCOUNT_CATEGORY = t8.ACCOUNT_CATEGORY
	inner join ACCOUNT_CATEGORY_REF t9 on t8.ACCOUNT_CATEGORY = t9.ACCOUNT_CATEGORY
		and t9.SERVICE_VERSION_ID = {1}
	inner join ACCOUNT_CATEGORY_VALUES t10 on t8.ACCOUNT_CATEGORY = t10.ACCOUNT_CATEGORY
		and t10.SERVICE_VERSION_ID = t9.SERVICE_VERSION_ID
		and t10.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join RATE_CLASS_KEY t11 on t1.RATE_CLASS = t11.RATE_CLASS
	left outer join RATE_CLASS_REF t12 on t11.RATE_CLASS = t12.RATE_CLASS
		and t12.SERVICE_VERSION_ID = t9.SERVICE_VERSION_ID
	left outer join RATE_CLASS_VALUES t13 on t11.RATE_CLASS = t13.RATE_CLASS
		and t13.SERVICE_VERSION_ID = t9.SERVICE_VERSION_ID
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
