select
	t1.ADJ_TRANS_CODE as adjTransCode,
	t1.TRANS_TARGET_TYPE as transTargetTypeId,
        t3.DISPLAY_VALUE as transTargetType,
	t1.TRANS_TARGET_ID as transTargetId,
	t1.ADJ_TRANS_CATEGORY as adjTransCategory,
	t1.TRANS_SIGN as transSign,
	t1.BILLING_CATEGORY as billingCategory,
	t1.BILLING_LEVEL as billingLevel,
	t1.DESCRIPTION_CODE as descriptionCode,
        t4.DESCRIPTION_TEXT as description,
	t1.IS_VIEWABLE as isViewable,
	t1.IS_MODIFIABLE as isModifiable,
	t1.IS_ADJUSTABLE as isAdjustable,
	t1.IS_JOURNALABLE as isJournalable,
	t1.IS_DISPLAYED_ONBILL as isDisplayedOnbill,
	t1.IS_DISCONNECT_CREDIT as isDisconnectCredit,
	t1.IS_REFINANCE as isRefinance,
	t1.ANNOTATION_TYPE as annotationType,
	t1.TAX_ON_INVOICE as taxOnInvoice,
	t1.IS_LATE_FEE_EXEMPT as isLateFeeExempt,
	t1.ALLOW_INTERIM_BILL as allowInterimBill,
	t1.IS_NEGATIVE_BILL_ADJ as isNegativeBillAdj
from
	ADJ_TRANS_DESCR t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''CHARGE_ELT_TYPE'')
		and t2.value = t1.TRANS_TARGET_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
        inner join DESCRIPTIONS t4 on t1.DESCRIPTION_CODE = t4.DESCRIPTION_CODE
                and t4.LANGUAGE_CODE = {2}
