select
	t1.MAJOR_PC_VERSION_NUM as majorPcVersionNum,
	t1.MINOR_PC_VERSION_NUM as minorPcVersionNum,
	t1.TARGET_GROUP_ID as targetGroupId,
	t1.STATUS as status
from
	PC_VERSION_TARGET_GROUP_MAP t1
	inner join PC_VERSION t2 on t1.MAJOR_PC_VERSION_NUM = t2.MAJOR_PC_VERSION_NUM
		and t1.MINOR_PC_VERSION_NUM = t2.MINOR_PC_VERSION_NUM
	inner join PC_VERSION t3 on t1.MAJOR_PC_VERSION_NUM = t3.MAJOR_PC_VERSION_NUM
		and t1.MINOR_PC_VERSION_NUM = t3.MINOR_PC_VERSION_NUM
	inner join DP_TARGET_GROUP_REF t4 on t1.TARGET_GROUP_ID = t4.TARGET_GROUP_ID
	inner join DP_TARGET_GROUP_VALUES t5 on t4.TARGET_GROUP_ID = t5.TARGET_GROUP_ID
		and t5.LANGUAGE_CODE = {2}
