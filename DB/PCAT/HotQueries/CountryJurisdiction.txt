select
	t3.DISPLAY_VALUE as countryCodeRef,
	t1.COUNTRY_CODE_ORIGIN as countryCodeOrigin,
	t5.DISPLAY_VALUE as countryCodeRefByCountryCodeTar,
	t1.COUNTRY_CODE_TARGET as countryCodeTarget,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t8.DISPLAY_VALUE as jurisdictionsKey,
	t1.JURISDICTION as jurisdiction
from
	COUNTRY_JURISDICTION t1
	inner join COUNTRY_CODE_REF t2 on t1.COUNTRY_CODE_ORIGIN = t2.COUNTRY_CODE
	inner join COUNTRY_CODE_VALUES t3 on t2.COUNTRY_CODE = t3.COUNTRY_CODE
		and t3.LANGUAGE_CODE = {2}
	inner join COUNTRY_CODE_REF t4 on t1.COUNTRY_CODE_TARGET = t4.COUNTRY_CODE
	inner join COUNTRY_CODE_VALUES t5 on t4.COUNTRY_CODE = t5.COUNTRY_CODE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join JURISDICTIONS_KEY t6 on t1.JURISDICTION = t6.JURISDICTION
	inner join JURISDICTIONS_REF t7 on t6.JURISDICTION = t7.JURISDICTION
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join JURISDICTIONS_VALUES t8 on t6.JURISDICTION = t8.JURISDICTION
		and t8.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
