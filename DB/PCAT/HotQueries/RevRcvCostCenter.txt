select
	t1.REV_RCV_COST_CTR as revRcvCostCtr,
	t1.G_L_CODE as gLCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	REV_RCV_COST_CENTER_REF t1
	inner join REV_RCV_COST_CENTER_VALUES t2 on t1.REV_RCV_COST_CTR = t2.REV_RCV_COST_CTR
where
	t2.LANGUAGE_CODE = {2}
