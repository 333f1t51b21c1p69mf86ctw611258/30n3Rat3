select
	t1.TEXT_ID as textId,
	t1.TAG as tag,
	t1.TEST_FLAG as testFlag,
	t1.IS_INTERNAL as isInternal,
	t1.IS_HIDDEN as isHidden,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_TEXT_REF t1
	inner join BILL_TEXT_VALUES t2 on t1.TEXT_ID = t2.TEXT_ID
		and t1.TEST_FLAG = t2.TEST_FLAG
where
	t2.LANGUAGE_CODE = {2}
