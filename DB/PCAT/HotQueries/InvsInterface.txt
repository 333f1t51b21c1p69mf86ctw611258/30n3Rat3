select
	t1.INTERFACE_ID as interfaceId,
	t1.SKIP_EXTERNAL as skipExternal,
	t1.EXECUTABLE as executable,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_INTERFACE_REF t1
	inner join INVS_INTERFACE_VALUES t2 on t1.INTERFACE_ID = t2.INTERFACE_ID
where
	t2.LANGUAGE_CODE = {2}
