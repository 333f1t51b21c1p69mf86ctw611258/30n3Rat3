select
	t1.COMP_STATUS as compStatus,
	t1.COMP_STATUS_TYPE as compStatusType,
	t1.GUIDE_FLAG as guideFlag,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CDR_COMP_STATUS_REF t1
	inner join CDR_COMP_STATUS_VALUES t2 on t1.COMP_STATUS = t2.COMP_STATUS
where
	t2.LANGUAGE_CODE = {2}
