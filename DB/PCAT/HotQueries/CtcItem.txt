select
	t1.ITEM_ID as itemId,
	t1.P_ITEM_ID as pItemId,
	t1.OFFER_ID as offerId,
	t4.DISPLAY_VALUE as ctcGenreRef,
	t1.PRIMARY_GENRE_ID as primaryGenreId,
	t6.DISPLAY_VALUE as ctcGenreRefBySecondaryGenreId,
	t1.SECONDARY_GENRE_ID as secondaryGenreId,
	t8.DISPLAY_VALUE as ctcGenreRefByTertiaryGenreId,
	t1.TERTIARY_GENRE_ID as tertiaryGenreId,
	t1.DURATION as duration,
	t1.NEW_FG as newFg,
	t1.CHOICE_FG as choiceFg
from
	CTC_ITEM t1
	inner join CTC_OFFER t2 on t1.OFFER_ID = t2.OFFER_ID
	inner join CTC_GENRE_REF t3 on t1.PRIMARY_GENRE_ID = t3.GENRE_ID
	inner join CTC_GENRE_VALUES t4 on t3.GENRE_ID = t4.GENRE_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join CTC_GENRE_REF t5 on t1.SECONDARY_GENRE_ID = t5.GENRE_ID
	left outer join CTC_GENRE_VALUES t6 on t5.GENRE_ID = t6.GENRE_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join CTC_GENRE_REF t7 on t1.TERTIARY_GENRE_ID = t7.GENRE_ID
	left outer join CTC_GENRE_VALUES t8 on t7.GENRE_ID = t8.GENRE_ID
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
