select
	t1.STOCK_PROPERTY_ID as stockPropertyId,
	t1.STOCK_PROPERTY_TAG as stockPropertyTag,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_STOCK_PROPERTY_REF t1
	inner join INVS_STOCK_PROPERTY_VALUES t2 on t1.STOCK_PROPERTY_ID = t2.STOCK_PROPERTY_ID
where
	t2.LANGUAGE_CODE = {2}
