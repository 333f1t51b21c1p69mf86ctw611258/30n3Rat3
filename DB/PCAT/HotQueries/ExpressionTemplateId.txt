select
	t1.EXPRESSION_TEMPLATE_ID as expressionTemplateId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.LOGICAL_OPERATOR as logicalOperator,
	t2.DOMAIN as domain,
	t5.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as segmentationKeyIdKey,
	t2.SEGMENTATION_KEY_ID as segmentationKeyId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t2.PRIORITY as priority,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	EXPRESSION_TEMPLATE_ID_KEY t1
	inner join EXPRESSION_TEMPLATE_ID_REF t2 on t1.EXPRESSION_TEMPLATE_ID = t2.EXPRESSION_TEMPLATE_ID
	inner join EXPRESSION_TEMPLATE_ID_VALUES t3 on t2.EXPRESSION_TEMPLATE_ID = t3.EXPRESSION_TEMPLATE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join SEGMENTATION_KEY_ID_KEY t4 on t2.SEGMENTATION_KEY_ID = t4.SEGMENTATION_KEY_ID
	left outer join SEGMENTATION_KEY_ID_REF t5 on t4.SEGMENTATION_KEY_ID = t5.SEGMENTATION_KEY_ID
		and t5.SERVICE_VERSION_ID = {1}
	left outer join SEGMENTATION_KEY_ID_VALUES t6 on t4.SEGMENTATION_KEY_ID = t6.SEGMENTATION_KEY_ID
		and t6.SERVICE_VERSION_ID = {1}
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
