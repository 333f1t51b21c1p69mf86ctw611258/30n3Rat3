select
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as serviceProvidersKey,
	t1.PROVIDER_ID as providerId,
	t1.PROVIDER_NAME as providerName,
	t1.PROVIDER_ADDRESS1 as providerAddress1,
	t1.PROVIDER_ADDRESS2 as providerAddress2,
	t1.PROVIDER_ADDRESS3 as providerAddress3,
	t1.PROVIDER_CITY as providerCity,
	t1.PROVIDER_STATE as providerState,
	t1.PROVIDER_ZIP as providerZip,
	t6.DISPLAY_VALUE as countryCodeRef,
	t1.PROVIDER_COUNTRY_CODE as providerCountryCode
from
	PROVIDER_ADDRESS t1
	inner join SERVICE_PROVIDERS_KEY t2 on t1.PROVIDER_ID = t2.PROVIDER_ID
	inner join SERVICE_PROVIDERS_REF t3 on t2.PROVIDER_ID = t3.PROVIDER_ID
		and t3.SERVICE_VERSION_ID = {1}
	inner join SERVICE_PROVIDERS_VALUES t4 on t2.PROVIDER_ID = t4.PROVIDER_ID
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join COUNTRY_CODE_REF t5 on t1.PROVIDER_COUNTRY_CODE = t5.COUNTRY_CODE
	inner join COUNTRY_CODE_VALUES t6 on t5.COUNTRY_CODE = t6.COUNTRY_CODE
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
