select
	t1.EQUIPMENT_CONDITION_ID as equipmentConditionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_EQUIPMENT_COND_REF t1
	inner join INVS_EQUIPMENT_COND_VALUES t2 on t1.EQUIPMENT_CONDITION_ID = t2.EQUIPMENT_CONDITION_ID
where
	t2.LANGUAGE_CODE = {2}
