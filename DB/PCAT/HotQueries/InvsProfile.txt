select
	t1.PROFILE_ID as profileId,
	t1.IS_SERVICE as isService,
	t1.IS_WAREHOUSE as isWarehouse,
	t1.ACTION_ID as actionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_PROFILE_REF t1
	inner join INVS_PROFILE_VALUES t2 on t1.PROFILE_ID = t2.PROFILE_ID
where
	t2.LANGUAGE_CODE = {2}
