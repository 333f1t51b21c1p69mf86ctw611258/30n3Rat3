select
	t1.ACTIVITY_CHRSTICS_ID as activityChrsticsId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	ACTIVITY_CHRSTICS_KEY t1
	inner join ACTIVITY_CHRSTICS_REF t2 on t1.ACTIVITY_CHRSTICS_ID = t2.ACTIVITY_CHRSTICS_ID
	inner join ACTIVITY_CHRSTICS_VALUES t3 on t2.ACTIVITY_CHRSTICS_ID = t3.ACTIVITY_CHRSTICS_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
