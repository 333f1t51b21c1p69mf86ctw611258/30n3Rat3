select
	t1.TT_SOURCE_ID as ttSourceId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	TT_SOURCE_REF t1
	inner join TT_SOURCE_VALUES t2 on t1.TT_SOURCE_ID = t2.TT_SOURCE_ID
where
	t2.LANGUAGE_CODE = {2}
