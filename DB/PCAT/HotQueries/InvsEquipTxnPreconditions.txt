select
	t1.PRECONDITION_ID as preconditionId,
	t3.DISPLAY_VALUE as invsEquipmentTxnRef,
	t1.EQUIPMENT_TRANSACTION_ID as equipmentTransactionId,
	t5.DISPLAY_VALUE as invsFieldMasterRef,
	t1.FIELD_ID as fieldId,
	t1.PRECONDITION_VALUE as preconditionValue
from
	INVS_EQUIP_TXN_PRECONDITIONS t1
	inner join INVS_EQUIPMENT_TXN_REF t2 on t1.EQUIPMENT_TRANSACTION_ID = t2.EQUIPMENT_TRANSACTION_ID
	inner join INVS_EQUIPMENT_TXN_VALUES t3 on t2.EQUIPMENT_TRANSACTION_ID = t3.EQUIPMENT_TRANSACTION_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_FIELD_MASTER_REF t4 on t1.FIELD_ID = t4.FIELD_ID
	inner join INVS_FIELD_MASTER_VALUES t5 on t4.FIELD_ID = t5.FIELD_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
