select
	t1.SUBSYSTEM_ID as subsystemId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as applicationIdKey,
	t2.APPLICATION_ID as applicationId,
	t2.IS_ENABLED as isEnabled,
	t2.CREATED_DATE as createdDate,
	t2.UPGRADED_DATE as upgradedDate,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	SUBSYSTEM_KEY t1
	inner join SUBSYSTEM_REF t2 on t1.SUBSYSTEM_ID = t2.SUBSYSTEM_ID
	inner join SUBSYSTEM_VALUES t3 on t2.SUBSYSTEM_ID = t3.SUBSYSTEM_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join APPLICATION_ID_KEY t4 on t2.APPLICATION_ID = t4.APPLICATION_ID
	inner join APPLICATION_ID_REF t5 on t4.APPLICATION_ID = t5.APPLICATION_ID
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join APPLICATION_ID_VALUES t6 on t4.APPLICATION_ID = t6.APPLICATION_ID
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
