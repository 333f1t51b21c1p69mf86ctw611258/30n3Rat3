select
	t1.REL_CONSTRAINT_COND_ID as relConstraintCondId,
	t1.REL_CONSTRAINT_ID as relConstraintId,
	t1.COND_OBJECT_ID as condObjectId,
	t1.COND_ATTRIBUTE_ID as condAttributeId,
	t6.DISPLAY_VALUE as bsdAttrConstrntTypeRef,
	t1.COND_ATTR_CONSTRAINT_TYPE_ID as condAttrConstraintTypeId,
	t1.COND_EXPRESSION as condExpression,
	t1.UPD_CNT as updCnt,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt
from
	BSD_REL_CONSTRNT_CONDITIONS t1
	inner join BSD_RELATIONAL_CONSTRAINTS t2 on t1.REL_CONSTRAINT_ID = t2.REL_CONSTRAINT_ID
	inner join BSD_OBJECT_ATTRIBUTES t3 on t1.COND_OBJECT_ID = t3.OBJECT_ID
		and t1.COND_ATTRIBUTE_ID = t3.ATTRIBUTE_ID
	inner join BSD_OBJECT_ATTRIBUTES t4 on t1.COND_OBJECT_ID = t4.OBJECT_ID
		and t1.COND_ATTRIBUTE_ID = t4.ATTRIBUTE_ID
	inner join BSD_ATTR_CONSTRNT_TYPE_REF t5 on t1.COND_ATTR_CONSTRAINT_TYPE_ID = t5.ATTR_CONSTRAINT_TYPE_ID
	inner join BSD_ATTR_CONSTRNT_TYPE_VALUES t6 on t5.ATTR_CONSTRAINT_TYPE_ID = t6.ATTR_CONSTRAINT_TYPE_ID
		and t6.LANGUAGE_CODE = {2}
