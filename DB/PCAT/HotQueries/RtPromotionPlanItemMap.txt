select
	t1.RT_PROMO_PI_MAP_ID as rtPromoPiMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as rtPromotionPlanKey,
	t1.RT_PROMOTION_PLAN_ID as rtPromotionPlanId,
	t7.DISPLAY_VALUE as rtBonusItemKey,
	t1.BONUS_ITEM_ID as bonusItemId,
	t10.DISPLAY_VALUE as rtDiscountItemKey,
	t1.DISCOUNT_ITEM_ID as discountItemId
from
	RT_PROMOTION_PLAN_ITEM_MAP t1
	inner join RT_PROMOTION_PLAN_KEY t2 on t1.RT_PROMOTION_PLAN_ID = t2.RT_PROMOTION_PLAN_ID
	inner join RT_PROMOTION_PLAN_REF t3 on t2.RT_PROMOTION_PLAN_ID = t3.RT_PROMOTION_PLAN_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join RT_PROMOTION_PLAN_VALUES t4 on t2.RT_PROMOTION_PLAN_ID = t4.RT_PROMOTION_PLAN_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join RT_BONUS_ITEM_KEY t5 on t1.BONUS_ITEM_ID = t5.BONUS_ITEM_ID
	left outer join RT_BONUS_ITEM_REF t6 on t5.BONUS_ITEM_ID = t6.BONUS_ITEM_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join RT_BONUS_ITEM_VALUES t7 on t5.BONUS_ITEM_ID = t7.BONUS_ITEM_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join RT_DISCOUNT_ITEM_KEY t8 on t1.DISCOUNT_ITEM_ID = t8.DISCOUNT_ITEM_ID
	left outer join RT_DISCOUNT_ITEM_REF t9 on t8.DISCOUNT_ITEM_ID = t9.DISCOUNT_ITEM_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join RT_DISCOUNT_ITEM_VALUES t10 on t8.DISCOUNT_ITEM_ID = t10.DISCOUNT_ITEM_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
