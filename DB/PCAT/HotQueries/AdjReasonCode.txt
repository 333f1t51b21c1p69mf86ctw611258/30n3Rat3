select
	t1.ADJ_REASON_CODE as adjReasonCode,
	t1.JOURNAL_CODE as journalCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ADJ_REASON_CODE_REF t1
	inner join ADJ_REASON_CODE_VALUES t2 on t1.ADJ_REASON_CODE = t2.ADJ_REASON_CODE
where
	t2.LANGUAGE_CODE = {2}
