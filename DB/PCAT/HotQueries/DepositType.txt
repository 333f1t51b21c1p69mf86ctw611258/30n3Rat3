select
	t1.DEPOSIT_TYPE as depositType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	DEPOSIT_TYPE_REF t1
	inner join DEPOSIT_TYPE_VALUES t2 on t1.DEPOSIT_TYPE = t2.DEPOSIT_TYPE
where
	t2.LANGUAGE_CODE = {2}
