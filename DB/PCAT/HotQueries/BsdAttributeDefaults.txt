select
	t1.ATTRIBUTE_DEFAULT_ID as attributeDefaultId,
	t1.OBJECT_ID as objectId,
	t1.ATTRIBUTE_ID as attributeId,
	t1.OVERRIDE as override,
	t1.PROCESS_SEQUENCE as processSequence,
	t1.DEFAULT_TYPE as defaultType,
	t1.DEFAULT_VALUE as defaultValue,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.UPD_CNT as updCnt,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt
from
	BSD_ATTRIBUTE_DEFAULTS t1
	left outer join BSD_OBJECTS t2 on t1.OBJECT_ID = t2.OBJECT_ID
	left outer join BSD_ATTRIBUTES t3 on t1.ATTRIBUTE_ID = t3.ATTRIBUTE_ID
