select
	t1.FORMAT_TYPE_ID as formatTypeId,
	t1.P_FORMAT_TYPE_ID as pFormatTypeId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_FORMAT_TYPE_REF t1
	inner join CTC_FORMAT_TYPE_VALUES t2 on t1.FORMAT_TYPE_ID = t2.FORMAT_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
