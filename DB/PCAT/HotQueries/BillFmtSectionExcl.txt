select
	t1.SECTION_EXCLUDE as sectionExclude,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_FMT_SECTION_EXCL_REF t1
	inner join BILL_FMT_SECTION_EXCL_VALUES t2 on t1.SECTION_EXCLUDE = t2.SECTION_EXCLUDE
where
	t2.LANGUAGE_CODE = {2}
