select
	t1.LOCATION_DETAIL_ID as locationDetailId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_LOCATION_DETAIL_REF t1
	inner join INVS_LOCATION_DETAIL_VALUES t2 on t1.LOCATION_DETAIL_ID = t2.LOCATION_DETAIL_ID
where
	t2.LANGUAGE_CODE = {2}
