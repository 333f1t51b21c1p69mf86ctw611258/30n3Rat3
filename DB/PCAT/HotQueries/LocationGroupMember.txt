select
	t4.DISPLAY_VALUE as locationGroupKey,
	t1.LOCATION_GROUP_ID as locationGroupId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.GLOBAL_LOCATION_ID as globalLocationId
from
	LOCATION_GROUP_MEMBER t1
	inner join LOCATION_GROUP_KEY t2 on t1.LOCATION_GROUP_ID = t2.LOCATION_GROUP_ID
	inner join LOCATION_GROUP_REF t3 on t2.LOCATION_GROUP_ID = t3.LOCATION_GROUP_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join LOCATION_GROUP_VALUES t4 on t2.LOCATION_GROUP_ID = t4.LOCATION_GROUP_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
