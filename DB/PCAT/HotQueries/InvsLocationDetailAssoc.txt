select
	t3.DISPLAY_VALUE as invsLocationDetailRef,
	t1.LOCATION_DETAIL_ID as locationDetailId,
	t5.DISPLAY_VALUE as invsLocationRef,
	t1.LOCATION_ID as locationId
from
	INVS_LOCATION_DETAIL_ASSOC t1
	inner join INVS_LOCATION_DETAIL_REF t2 on t1.LOCATION_DETAIL_ID = t2.LOCATION_DETAIL_ID
	inner join INVS_LOCATION_DETAIL_VALUES t3 on t2.LOCATION_DETAIL_ID = t3.LOCATION_DETAIL_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_LOCATION_REF t4 on t1.LOCATION_ID = t4.LOCATION_ID
	inner join INVS_LOCATION_VALUES t5 on t4.LOCATION_ID = t5.LOCATION_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
