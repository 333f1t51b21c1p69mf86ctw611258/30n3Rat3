select
	t1.CARD_CARRIER as cardCarrier,
	t1.CARD_TYPE_CODE as cardTypeCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.EPG_CARD_TYPE as epgCardType,
	t1.CHECK_DIGIT_ALG as checkDigitAlg,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CARD_TYPE_CODE_REF t1
	inner join CARD_TYPE_CODE_VALUES t2 on t1.CARD_TYPE_CODE = t2.CARD_TYPE_CODE
where
	t2.LANGUAGE_CODE = {2}
