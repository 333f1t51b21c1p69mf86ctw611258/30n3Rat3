select
	t1.STATUS_TYPE_ID as statusTypeId,
	t1.IS_TIME_SENSITIVE as isTimeSensitive,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	STATUS_TYPE_REF t1
	inner join STATUS_TYPE_VALUES t2 on t1.STATUS_TYPE_ID = t2.STATUS_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
