select
	t1.JOBSTATE_ID as jobstateId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_JOBSTATE_REF t1
	inner join INVS_JOBSTATE_VALUES t2 on t1.JOBSTATE_ID = t2.JOBSTATE_ID
where
	t2.LANGUAGE_CODE = {2}
