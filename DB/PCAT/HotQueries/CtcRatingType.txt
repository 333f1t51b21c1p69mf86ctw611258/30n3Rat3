select
	t1.RATING_TYPE_ID as ratingTypeId,
	t1.P_RATING_TYPE_ID as pRatingTypeId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_RATING_TYPE_REF t1
	inner join CTC_RATING_TYPE_VALUES t2 on t1.RATING_TYPE_ID = t2.RATING_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
