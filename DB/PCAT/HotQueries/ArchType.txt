select
	t1.ARCH_TYPE_ID as archTypeId,
	t1.IS_INTERNAL as isInternal,
	t1.IS_DEFAULT as isDefault,
	t1.IS_ACTIVE as isActive,
	t1.IS_LEGACY as isLegacy,
	t4.DISPLAY_VALUE as outputMode,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARCH_TYPE_REF t1
	inner join ARCH_TYPE_VALUES t2 on t1.ARCH_TYPE_ID = t2.ARCH_TYPE_ID
        left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''output_mode'')
		and t3.value = t1.OUTPUT_MODE
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
