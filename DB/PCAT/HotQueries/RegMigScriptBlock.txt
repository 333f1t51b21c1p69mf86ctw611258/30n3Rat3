select
	t1.BLOCK_ID as blockId,
	t1.MIG_SCRIPT_ID as migScriptId,
	t1.BLOCK_NAME as blockName,
	t1.BLOCK_TYPE as blockType,
	t1.BLOCK_DBMS_TYPE as blockDbmsType,
	t1.BLOCK_DB_TYPE as blockDbType,
	t1.BLOCK_MODULE as blockModule,
	t1.BLOCK_START as blockStart,
	t1.BLOCK_STATUS as blockStatus,
	t1.BLOCK_END as blockEnd
from
	REG_MIG_SCRIPT_BLOCK t1
	inner join REG_MIG_SCRIPT t2 on t1.MIG_SCRIPT_ID = t2.MIG_SCRIPT_ID
