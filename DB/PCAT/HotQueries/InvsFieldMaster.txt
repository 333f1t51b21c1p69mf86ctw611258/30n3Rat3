select
	t1.FIELD_ID as fieldId,
	t1.ATTRIBUTE_NAME as attributeName,
	t1.ATTRIBUTE_TYPE as attributeType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_FIELD_MASTER_REF t1
	inner join INVS_FIELD_MASTER_VALUES t2 on t1.FIELD_ID = t2.FIELD_ID
where
	t2.LANGUAGE_CODE = {2}
