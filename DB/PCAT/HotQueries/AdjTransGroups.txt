select
	t3.DISPLAY_VALUE as adjTransGroupRef,
	t1.ADJ_TRANS_GROUP as adjTransGroup,
	t1.ADJ_TRANS_CODE as adjTransCode,
	t1.PRIORITY as priority,
	t1.TRANS_GROUP_TYPE as transGroupType
from
	ADJ_TRANS_GROUPS t1
	inner join ADJ_TRANS_GROUP_REF t2 on t1.ADJ_TRANS_GROUP = t2.ADJ_TRANS_GROUP
	inner join ADJ_TRANS_GROUP_VALUES t3 on t2.ADJ_TRANS_GROUP = t3.ADJ_TRANS_GROUP
		and t3.LANGUAGE_CODE = {2}
