select
	t1.CONSTRAINT_DESC_ID as constraintDescId,
	t1.CONSTRAINT_DESC_NAME as constraintDescName,
	t1.UPD_CNT as updCnt,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BSD_CONSTRAINT_DESC_REF t1
	inner join BSD_CONSTRAINT_DESC_VALUES t2 on t1.CONSTRAINT_DESC_ID = t2.CONSTRAINT_DESC_ID
where
	t2.LANGUAGE_CODE = {2}
