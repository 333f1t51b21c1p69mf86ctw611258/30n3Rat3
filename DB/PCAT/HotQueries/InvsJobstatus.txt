select
	t1.JOBSTATUS_ID as jobstatusId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_JOBSTATUS_REF t1
	inner join INVS_JOBSTATUS_VALUES t2 on t1.JOBSTATUS_ID = t2.JOBSTATUS_ID
where
	t2.LANGUAGE_CODE = {2}
