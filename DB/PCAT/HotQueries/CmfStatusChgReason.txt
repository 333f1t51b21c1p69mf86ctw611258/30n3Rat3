select
	t1.CMF_STATUS_CHG_REASON as cmfStatusChgReason,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CMF_STATUS_CHG_REASON_REF t1
	inner join CMF_STATUS_CHG_REASON_VALUES t2 on t1.CMF_STATUS_CHG_REASON = t2.CMF_STATUS_CHG_REASON
where
	t2.LANGUAGE_CODE = {2}
