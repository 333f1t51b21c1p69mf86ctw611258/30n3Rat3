select
	t1.THRESHOLD_ID as thresholdId,
	t3.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t5.DISPLAY_VALUE as invsStatusRef,
	t1.STATUS_ID as statusId,
	t1.THRESHOLD_VALUE as thresholdValue,
	t1.TRIGGERTYPE as triggertype,
	t7.DISPLAY_VALUE as period,
	t1.RESETCOUNTER as resetcounter,
	t9.DISPLAY_VALUE as invsSalesChannelRef,
	t1.SALES_CHANNEL_ID as salesChannelId,
	t11.DISPLAY_VALUE as invsGeographicRegionRef,
	t1.GEOGRAPHIC_REGION_ID as geographicRegionId,
	t13.DISPLAY_VALUE as invsNetworkDeviceRef,
	t1.NETWORK_DEVICE_ID as networkDeviceId,
	t1.PRIMARY_CODE as primaryCode,
	t1.SECONDARY_CODE as secondaryCode,
	t1.TERTIARY_CODE as tertiaryCode,
	t1.RESET_VALUE as resetValue
from
	INVS_THRESHOLD t1
	inner join INVS_TYPE_REF t2 on t1.INVENTORY_TYPE_ID = t2.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t3 on t2.INVENTORY_TYPE_ID = t3.INVENTORY_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	left outer join INVS_STATUS_REF t4 on t1.STATUS_ID = t4.STATUS_ID
	left outer join INVS_STATUS_VALUES t5 on t4.STATUS_ID = t5.STATUS_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t6 on t6.enumeration_key = lower(''PERIOD'')
		and t6.value = t1.PERIOD
	left outer join GENERIC_ENUMERATION_VALUES t7 on t7.enumeration_key = t6.enumeration_key
		and t7.value = t6.value
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join INVS_SALES_CHANNEL_REF t8 on t1.SALES_CHANNEL_ID = t8.SALES_CHANNEL_ID
	left outer join INVS_SALES_CHANNEL_VALUES t9 on t8.SALES_CHANNEL_ID = t9.SALES_CHANNEL_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join INVS_GEOGRAPHIC_REGION_REF t10 on t1.GEOGRAPHIC_REGION_ID = t10.GEOGRAPHIC_REGION_ID
	left outer join INVS_GEOGRAPHIC_REGION_VALUES t11 on t10.GEOGRAPHIC_REGION_ID = t11.GEOGRAPHIC_REGION_ID
		and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join INVS_NETWORK_DEVICE_REF t12 on t1.NETWORK_DEVICE_ID = t12.NETWORK_DEVICE_ID
	left outer join INVS_NETWORK_DEVICE_VALUES t13 on t12.NETWORK_DEVICE_ID = t13.NETWORK_DEVICE_ID
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
