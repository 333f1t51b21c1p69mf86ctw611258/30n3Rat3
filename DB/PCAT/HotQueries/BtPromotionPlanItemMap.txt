select
	t1.BT_PROMOTION_PLAN_ITEM_MAP_ID as btPromotionPlanItemMapId,
	t4.DISPLAY_VALUE as btPromotionPlanKey,
	t1.BT_PROMOTION_PLAN_ID as btPromotionPlanId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as btDiscountItemKey,
	t1.DISCOUNT_ID as discountId,
	t10.DISPLAY_VALUE as unitCrItemKey,
	t1.UNIT_CR_ID as unitCrId,
	t1.PLAN_ORDER as planOrder,
	t1.DEF_ORDER as defOrder
from
	BT_PROMOTION_PLAN_ITEM_MAP t1
	inner join BT_PROMOTION_PLAN_KEY t2 on t1.BT_PROMOTION_PLAN_ID = t2.BT_PROMOTION_PLAN_ID
	inner join BT_PROMOTION_PLAN_REF t3 on t2.BT_PROMOTION_PLAN_ID = t3.BT_PROMOTION_PLAN_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BT_PROMOTION_PLAN_VALUES t4 on t2.BT_PROMOTION_PLAN_ID = t4.BT_PROMOTION_PLAN_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join BT_DISCOUNT_ITEM_KEY t5 on t1.DISCOUNT_ID = t5.DISCOUNT_ID
	left outer join BT_DISCOUNT_ITEM_REF t6 on t5.DISCOUNT_ID = t6.DISCOUNT_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join BT_DISCOUNT_ITEM_VALUES t7 on t5.DISCOUNT_ID = t7.DISCOUNT_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join UNIT_CR_ITEM_KEY t8 on t1.UNIT_CR_ID = t8.UNIT_CR_ID
	left outer join UNIT_CR_ITEM_REF t9 on t8.UNIT_CR_ID = t9.UNIT_CR_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join UNIT_CR_ITEM_VALUES t10 on t8.UNIT_CR_ID = t10.UNIT_CR_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
