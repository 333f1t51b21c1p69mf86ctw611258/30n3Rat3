select
	t3.DISPLAY_VALUE as clearingHouseIdRef,
	t1.CLEARING_HOUSE_ID as clearingHouseId,
	t1.MODULE as module,
	t1.EXT_CONTACT_ID as extContactId,
	t1.IS_SEND as isSend,
	t1.CH_FORMAT_TYPE as chFormatType,
	t1.CREDIT_TRANS_TYPE as creditTransType,
	t1.DEBIT_TRANS_TYPE as debitTransType,
	t1.ALLOW_AUTO_POST as allowAutoPost,
	t1.AUTO_POST_DELAY as autoPostDelay,
	t1.CHGB_RESUBMIT_DELAY as chgbResubmitDelay,
	t1.COLLECTION_DELAY as collectionDelay,
	t1.SUPPRESS_PRENOTES as suppressPrenotes
from
	CLEARING_HOUSE_CONTACTS t1
	inner join CLEARING_HOUSE_ID_REF t2 on t1.CLEARING_HOUSE_ID = t2.CLEARING_HOUSE_ID
	inner join CLEARING_HOUSE_ID_VALUES t3 on t2.CLEARING_HOUSE_ID = t3.CLEARING_HOUSE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join EXT_CONTACTS t4 on t1.EXT_CONTACT_ID = t4.EXT_CONTACT_ID
