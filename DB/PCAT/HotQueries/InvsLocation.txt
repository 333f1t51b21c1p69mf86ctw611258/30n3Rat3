select
	t1.LOCATION_ID as locationId,
	t1.IS_INTERNAL as isInternal,
	t1.IS_DEFAULT as isDefault,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_LOCATION_REF t1
	inner join INVS_LOCATION_VALUES t2 on t1.LOCATION_ID = t2.LOCATION_ID
where
	t2.LANGUAGE_CODE = {2}
