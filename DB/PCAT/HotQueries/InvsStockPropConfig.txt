select
	t1.STOCK_PROPERTY_CONFIG_ID as stockPropertyConfigId,
	t1.MIN_QUANTITY as minQuantity,
	t1.MAX_QUANTITY as maxQuantity,
	t1.SUGGESTED_QUANTITY as suggestedQuantity,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_STOCK_PROP_CONFIG_REF t1
	inner join INVS_STOCK_PROP_CONFIG_VALUES t2 on t1.STOCK_PROPERTY_CONFIG_ID = t2.STOCK_PROPERTY_CONFIG_ID
where
	t2.LANGUAGE_CODE = {2}
