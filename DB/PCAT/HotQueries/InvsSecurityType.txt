select
	t1.SECURITY_TYPE_ID as securityTypeId,
	t1.SECURITY_TAG as securityTag,
	t1.SAVE_TO_DATABASE as saveToDatabase,
	t1.MIN_LENGTH as minLength,
	t1.MAX_LENGTH as maxLength,
	t1.IS_VISIBLE as isVisible,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_SECURITY_TYPE_REF t1
	inner join INVS_SECURITY_TYPE_VALUES t2 on t1.SECURITY_TYPE_ID = t2.SECURITY_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
