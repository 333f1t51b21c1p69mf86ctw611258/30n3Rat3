select
	t1.ATTR_CONSTRAINT_COND_ID as attrConstraintCondId,
	t1.ATTR_CONSTRAINT_ID as attrConstraintId,
	t1.COND_ATTRIBUTE_ID as condAttributeId,
	t5.DISPLAY_VALUE as bsdAttrConstrntTypeRef,
	t1.COND_ATTR_CONSTRAINT_TYPE_ID as condAttrConstraintTypeId,
	t1.COND_EXPRESSION as condExpression,
	t1.UPD_CNT as updCnt,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt
from
	BSD_ATTR_CONSTRNT_CONDITIONS t1
	inner join BSD_ATTRIBUTE_CONSTRAINTS t2 on t1.ATTR_CONSTRAINT_ID = t2.ATTR_CONSTRAINT_ID
	left outer join BSD_ATTRIBUTES t3 on t1.COND_ATTRIBUTE_ID = t3.ATTRIBUTE_ID
	inner join BSD_ATTR_CONSTRNT_TYPE_REF t4 on t1.COND_ATTR_CONSTRAINT_TYPE_ID = t4.ATTR_CONSTRAINT_TYPE_ID
	inner join BSD_ATTR_CONSTRNT_TYPE_VALUES t5 on t4.ATTR_CONSTRAINT_TYPE_ID = t5.ATTR_CONSTRAINT_TYPE_ID
		and t5.LANGUAGE_CODE = {2}
