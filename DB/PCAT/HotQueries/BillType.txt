select
	t1.BILL_TYPE as billType,
	t4.DISPLAY_VALUE as adjTransGroupRef,
	t1.ADJ_TRANS_GROUP as adjTransGroup,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.IS_VIEWABLE as isViewable,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	BILL_TYPE_REF t1
	inner join BILL_TYPE_VALUES t2 on t1.BILL_TYPE = t2.BILL_TYPE
	left outer join ADJ_TRANS_GROUP_REF t3 on t1.ADJ_TRANS_GROUP = t3.ADJ_TRANS_GROUP
	left outer join ADJ_TRANS_GROUP_VALUES t4 on t3.ADJ_TRANS_GROUP = t4.ADJ_TRANS_GROUP
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
