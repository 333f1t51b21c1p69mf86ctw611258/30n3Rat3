select
	t3.DISPLAY_VALUE as cellIdRef,
	t1.CELL_ID as cellId,
	t5.DISPLAY_VALUE as zoneRef,
	t1.ZONE_ID as zoneId,
	t7.SERVICE_VERSION_ID as serviceVersionId,
	t8.DISPLAY_VALUE as zoneClassKey,
	t1.ZONE_CLASS as zoneClass
from
	CELL_ID_ZONE_ID_MAP t1
	inner join CELL_ID_REF t2 on t1.CELL_ID = t2.CELL_ID
	inner join CELL_ID_VALUES t3 on t2.CELL_ID = t3.CELL_ID
		and t3.LANGUAGE_CODE = {2}
	inner join ZONE_REF t4 on t1.ZONE_ID = t4.ZONE_ID
	inner join ZONE_VALUES t5 on t4.ZONE_ID = t5.ZONE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join ZONE_CLASS_KEY t6 on t1.ZONE_CLASS = t6.ZONE_CLASS
	inner join ZONE_CLASS_REF t7 on t6.ZONE_CLASS = t7.ZONE_CLASS
		and t7.SERVICE_VERSION_ID = {1}
	inner join ZONE_CLASS_VALUES t8 on t6.ZONE_CLASS = t8.ZONE_CLASS
		and t8.SERVICE_VERSION_ID = t7.SERVICE_VERSION_ID
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
