select
	t3.DISPLAY_VALUE as bsdBusinessUnitRef,
	t1.BUSINESS_UNIT_ID as businessUnitId,
	t5.DISPLAY_VALUE as bsdApplicationRef,
	t1.APPLICATION_ID as applicationId,
	t1.CONSTRAINT_ID as constraintId,
	t1.IS_RELATIONAL_CONSTRAINT as isRelationalConstraint,
	t1.DEPLOYMENT_MODE as deploymentMode,
	t1.ACTION_ID as actionId
from
	BSD_APPLICATION_CONSTRAINTS t1
	inner join BSD_BUSINESS_UNIT_REF t2 on t1.BUSINESS_UNIT_ID = t2.BUSINESS_UNIT_ID
	inner join BSD_BUSINESS_UNIT_VALUES t3 on t2.BUSINESS_UNIT_ID = t3.BUSINESS_UNIT_ID
		and t3.LANGUAGE_CODE = {2}
	inner join BSD_APPLICATION_REF t4 on t1.APPLICATION_ID = t4.APPLICATION_ID
	inner join BSD_APPLICATION_VALUES t5 on t4.APPLICATION_ID = t5.APPLICATION_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join BSD_ACTIONS t6 on t1.ACTION_ID = t6.ACTION_ID
