select
	t1.BLACKLIST_ID as blacklistId,
	t1.IS_ASSIGNABLE as isAssignable,
	t1.IS_PROVISIONABLE as isProvisionable,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode
from
	INVS_BLACKLIST_REF t1
	inner join INVS_BLACKLIST_VALUES t2 on t1.BLACKLIST_ID = t2.BLACKLIST_ID
where
	t2.LANGUAGE_CODE = {2}
