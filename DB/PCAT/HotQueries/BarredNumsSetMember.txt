select
	t1.BARRED_NUM_SET_MEM_ID as barredNumSetMemId,
	t4.DISPLAY_VALUE as barredNumsSetIdKey,
	t1.BARRED_NUM_SET_ID as barredNumSetId,
	t1.BARRED_NUMBER as barredNumber,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	BARRED_NUMS_SET_MEMBER t1
	inner join BARRED_NUMS_SET_ID_KEY t2 on t1.BARRED_NUM_SET_ID = t2.BARRED_NUM_SET_ID
	inner join BARRED_NUMS_SET_ID_REF t3 on t2.BARRED_NUM_SET_ID = t3.BARRED_NUM_SET_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BARRED_NUMS_SET_ID_VALUES t4 on t2.BARRED_NUM_SET_ID = t4.BARRED_NUM_SET_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
