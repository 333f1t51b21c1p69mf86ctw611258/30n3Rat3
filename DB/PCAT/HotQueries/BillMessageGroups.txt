select
	t3.DISPLAY_VALUE as msgGrpIdRef,
	t1.MSG_GRP_ID as msgGrpId,
	t1.MESSAGE_ID as messageId,
	t5.DISPLAY_VALUE as msgTriggerIdRef,
	t1.MSG_TRIGGER_ID as msgTriggerId,
	t1.MSG_SEQUENCE_NUM as msgSequenceNum,
	t1.GEO_KEY as geoKey,
	t7.DISPLAY_VALUE as mktCodeRef,
	t1.MKT_CODE as mktCode,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.SERVICE_CENTER_ID as serviceCenterId,
	t1.MSG_FOREIGN_KEY as msgForeignKey
from
	BILL_MESSAGE_GROUPS t1
	inner join MSG_GRP_ID_REF t2 on t1.MSG_GRP_ID = t2.MSG_GRP_ID
	inner join MSG_GRP_ID_VALUES t3 on t2.MSG_GRP_ID = t3.MSG_GRP_ID
		and t3.LANGUAGE_CODE = {2}
	inner join MSG_TRIGGER_ID_REF t4 on t1.MSG_TRIGGER_ID = t4.MSG_TRIGGER_ID
	inner join MSG_TRIGGER_ID_VALUES t5 on t4.MSG_TRIGGER_ID = t5.MSG_TRIGGER_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join MKT_CODE_REF t6 on t1.MKT_CODE = t6.MKT_CODE
	inner join MKT_CODE_VALUES t7 on t6.MKT_CODE = t7.MKT_CODE
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
