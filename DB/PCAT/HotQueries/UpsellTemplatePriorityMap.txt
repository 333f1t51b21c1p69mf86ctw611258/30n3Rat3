select
	t1.UPSELL_TEMPLATE_PRIORITY_ID as upsellTemplatePriorityId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as upsellTemplateKey,
	t1.DEFAULT_UPSELL_TEMPLATE_ID as defaultUpsellTemplateId,
	t1.ACCOUNT_CATEGORY_PRIORITY as accountCategoryPriority,
	t1.REGULATORY_ID_PRIORITY as regulatoryIdPriority,
	t1.VIP_CODE_PRIORITY as vipCodePriority,
	t1.MKT_CODE_PRIORITY as mktCodePriority
from
	UPSELL_TEMPLATE_PRIORITY_MAP t1
	inner join UPSELL_TEMPLATE_KEY t2 on t1.DEFAULT_UPSELL_TEMPLATE_ID = t2.UPSELL_TEMPLATE_ID
	inner join UPSELL_TEMPLATE_REF t3 on t2.UPSELL_TEMPLATE_ID = t3.UPSELL_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join UPSELL_TEMPLATE_VALUES t4 on t2.UPSELL_TEMPLATE_ID = t4.UPSELL_TEMPLATE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
