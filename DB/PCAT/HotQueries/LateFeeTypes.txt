select
	t1.LATE_FEE_TYPE_ID as lateFeeTypeId,
	t1.GOOD_PAYER as goodPayer,
	t1.ACTIVE_DT as activeDt,
	t1.CALCULATION_TYPE as calculationType,
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t3.ISO_CODE as currencyIsoCode,
	t3.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t4.DISPLAY_VALUE as rateCurrencyKey,
	t1.CURRENCY_CODE as currencyCode,
	t1.MIN_AMT as minAmt,
	t1.MAX_AMT as maxAmt,
	t6.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as nrcTermKey,
	t1.NRC_TERM_ID as nrcTermId,
	t1.GRACE_PERIOD as gracePeriod,
	t1.MIN_BAL_THRESH as minBalThresh,
	t1.INACTIVE_DT as inactiveDt
from
	LATE_FEE_TYPES t1
	inner join RATE_CURRENCY_KEY t2 on t1.CURRENCY_CODE = t2.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t3 on t2.CURRENCY_CODE = t3.CURRENCY_CODE
		and t3.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t4 on t2.CURRENCY_CODE = t4.CURRENCY_CODE
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join NRC_TERM_KEY t5 on t1.NRC_TERM_ID = t5.NRC_TERM_ID
	inner join NRC_TERM_REF t6 on t5.NRC_TERM_ID = t6.NRC_TERM_ID
		and t6.RESELLER_VERSION_ID = {0}
	inner join NRC_TERM_VALUES t7 on t5.NRC_TERM_ID = t7.NRC_TERM_ID
		and t7.RESELLER_VERSION_ID = t6.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
