select
	t1.TEMPLATE_CODE as templateCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_FMT_TEMPLATE_CODE_REF t1
	inner join BILL_FMT_TEMPLATE_CODE_VALUES t2 on t1.TEMPLATE_CODE = t2.TEMPLATE_CODE
where
	t2.LANGUAGE_CODE = {2}
