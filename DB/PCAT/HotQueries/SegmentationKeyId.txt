select
	t1.SEGMENTATION_KEY_ID as segmentationKeyId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.DATATYPE as datatype,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	SEGMENTATION_KEY_ID_KEY t1
	inner join SEGMENTATION_KEY_ID_REF t2 on t1.SEGMENTATION_KEY_ID = t2.SEGMENTATION_KEY_ID
	inner join SEGMENTATION_KEY_ID_VALUES t3 on t2.SEGMENTATION_KEY_ID = t3.SEGMENTATION_KEY_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
