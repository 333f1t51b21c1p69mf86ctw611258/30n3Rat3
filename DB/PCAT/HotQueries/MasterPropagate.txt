select
	t1.REPID as repid,
	t1.TABLE_NAME as tableName,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.TRIG_TYPE as trigType,
	t1.ROW_ID as rowId,
	t1.WHERE_STR as whereStr,
	t1.SET_STR as setStr,
	t1.SET_STR2 as setStr2,
	t1.TIME_STAMP as timeStamp,
	t1.ERROR_REASON as errorReason
from
	MASTER_PROPAGATE t1
where
	t1.RESELLER_VERSION_ID = {0}
	and t1.SERVICE_VERSION_ID = {1}
