select
	t4.DISPLAY_VALUE as geoRegionKey,
	t1.GEO_REGION_ID as geoRegionId,
	t6.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t1.SERVICE_VERSION_ID as serviceVersionId
from
	GEO_REGION_INVS_TYPE_MAP t1
	inner join GEO_REGION_KEY t2 on t1.GEO_REGION_ID = t2.GEO_REGION_ID
	inner join GEO_REGION_REF t3 on t2.GEO_REGION_ID = t3.GEO_REGION_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join GEO_REGION_VALUES t4 on t2.GEO_REGION_ID = t4.GEO_REGION_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join INVS_TYPE_REF t5 on t1.INVENTORY_TYPE_ID = t5.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t6 on t5.INVENTORY_TYPE_ID = t6.INVENTORY_TYPE_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
