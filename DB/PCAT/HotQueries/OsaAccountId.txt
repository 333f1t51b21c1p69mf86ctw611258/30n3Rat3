select
	t1.OSA_ACCOUNT_ID as osaAccountId,
	t1.OSA_MERCHANT_ACCT_ID as osaMerchantAcctId,
	t2.OSA_MERCHANT_ACCT_ID as osaMerchantAcctId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	OSA_ACCOUNT_ID_KEY t1
	inner join OSA_ACCOUNT_ID_REF t2 on t1.OSA_ACCOUNT_ID = t2.OSA_ACCOUNT_ID
	inner join OSA_ACCOUNT_ID_VALUES t3 on t2.OSA_ACCOUNT_ID = t3.OSA_ACCOUNT_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
