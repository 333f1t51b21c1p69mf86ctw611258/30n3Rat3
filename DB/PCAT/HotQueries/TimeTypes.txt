select
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.SEQNUM as seqnum,
	t1.OFFER_ID as offerId,
	t1.RC_TERM_ID as rcTermId,
	t1.USAGE_PLAN_ID as usagePlanId,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt,
	t1.CREATE_DT as createDt,
	t1.PRIORITY as priority,
	t1.DISCOUNT_FACTOR as discountFactor,
	t4.DISPLAY_VALUE as timeTypeKey,
	t1.TIME_TYPE_ID as timeTypeId,
	t1.ENDHHMM as endhhmm,
	t1.STARTHHMM as starthhmm,
	t1.DAYOFWEEK as dayofweek,
	t1.DAYOFMONTH as dayofmonth,
	t1.MONTH as month,
	t1.YEAR as year,
	t6.DISPLAY_VALUE as classOfServiceCodeRef,
	t1.CLASS_OF_SERVICE_CODE as classOfServiceCode,
	t8.DISPLAY_VALUE as equipClassCodeRef,
	t1.EQUIP_CLASS_CODE as equipClassCode,
	t10.DISPLAY_VALUE as equipTypeCodeRef,
	t1.EQUIP_TYPE_CODE as equipTypeCode,
	t1.RATE_CLASS as rateClass,
	t13.DISPLAY_VALUE as autFinalKey,
	t1.AUT_ID as autId,
	t15.SERVICE_VERSION_ID as serviceVersionId,
	t16.DISPLAY_VALUE as providerClassKey,
	t1.PROVIDER_CLASS as providerClass,
	t19.DISPLAY_VALUE as billClassKey,
	t1.BILL_CLASS as billClass,
	t22.DISPLAY_VALUE as jurisdictionsKey,
	t1.JURISDICTION as jurisdiction,
	t25.DISPLAY_VALUE as pointClassKey,
	t1.POINT_CLASS_ORIGIN as pointClassOrigin,
	t28.DISPLAY_VALUE as pointClassKeyByPointClassTarge,
	t1.POINT_CLASS_TARGET as pointClassTarget
from
	TIME_TYPES t1
	inner join TIME_TYPE_KEY t2 on t1.TIME_TYPE_ID = t2.TIME_TYPE_ID
	inner join TIME_TYPE_REF t3 on t2.TIME_TYPE_ID = t3.TIME_TYPE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TIME_TYPE_VALUES t4 on t2.TIME_TYPE_ID = t4.TIME_TYPE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join CLASS_OF_SERVICE_CODE_REF t5 on t1.CLASS_OF_SERVICE_CODE = t5.CLASS_OF_SERVICE_CODE
	left outer join CLASS_OF_SERVICE_CODE_VALUES t6 on t5.CLASS_OF_SERVICE_CODE = t6.CLASS_OF_SERVICE_CODE
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join EQUIP_CLASS_CODE_REF t7 on t1.EQUIP_CLASS_CODE = t7.EQUIP_CLASS_CODE
	left outer join EQUIP_CLASS_CODE_VALUES t8 on t7.EQUIP_CLASS_CODE = t8.EQUIP_CLASS_CODE
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join EQUIP_TYPE_CODE_REF t9 on t1.EQUIP_TYPE_CODE = t9.EQUIP_TYPE_CODE
	left outer join EQUIP_TYPE_CODE_VALUES t10 on t9.EQUIP_TYPE_CODE = t10.EQUIP_TYPE_CODE
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join AUT_FINAL_KEY t11 on t1.AUT_ID = t11.AUT_ID
	left outer join AUT_FINAL_REF t12 on t11.AUT_ID = t12.AUT_ID
		and t12.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join AUT_FINAL_VALUES t13 on t11.AUT_ID = t13.AUT_ID
		and t13.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t13.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join PROVIDER_CLASS_KEY t14 on t1.PROVIDER_CLASS = t14.PROVIDER_CLASS
	left outer join PROVIDER_CLASS_REF t15 on t14.PROVIDER_CLASS = t15.PROVIDER_CLASS
		and t15.SERVICE_VERSION_ID = {1}
	left outer join PROVIDER_CLASS_VALUES t16 on t14.PROVIDER_CLASS = t16.PROVIDER_CLASS
		and t16.SERVICE_VERSION_ID = {1}
		and t16.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join BILL_CLASS_KEY t17 on t1.BILL_CLASS = t17.BILL_CLASS
	left outer join BILL_CLASS_REF t18 on t17.BILL_CLASS = t18.BILL_CLASS
		and t18.SERVICE_VERSION_ID = {1}
	left outer join BILL_CLASS_VALUES t19 on t17.BILL_CLASS = t19.BILL_CLASS
		and t19.SERVICE_VERSION_ID = {1}
		and t19.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join JURISDICTIONS_KEY t20 on t1.JURISDICTION = t20.JURISDICTION
	left outer join JURISDICTIONS_REF t21 on t20.JURISDICTION = t21.JURISDICTION
		and t21.SERVICE_VERSION_ID = {1}
	left outer join JURISDICTIONS_VALUES t22 on t20.JURISDICTION = t22.JURISDICTION
		and t22.SERVICE_VERSION_ID = {1}
		and t22.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join POINT_CLASS_KEY t23 on t1.POINT_CLASS_ORIGIN = t23.POINT_CLASS
	left outer join POINT_CLASS_REF t24 on t23.POINT_CLASS = t24.POINT_CLASS
		and t24.SERVICE_VERSION_ID = {1}
	left outer join POINT_CLASS_VALUES t25 on t23.POINT_CLASS = t25.POINT_CLASS
		and t25.SERVICE_VERSION_ID = {1}
		and t25.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join POINT_CLASS_KEY t26 on t1.POINT_CLASS_TARGET = t26.POINT_CLASS
	left outer join POINT_CLASS_REF t27 on t26.POINT_CLASS = t27.POINT_CLASS
		and t27.SERVICE_VERSION_ID = {1}
	left outer join POINT_CLASS_VALUES t28 on t26.POINT_CLASS = t28.POINT_CLASS
		and t28.SERVICE_VERSION_ID = {1}
		and t28.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
