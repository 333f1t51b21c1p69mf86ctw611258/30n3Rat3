select
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as segmentationKeyIdKey,
	t1.SEGMENTATION_KEY_ID as segmentationKeyId,
	t7.DISPLAY_VALUE as segmentationKeyAttrKey,
	t1.SEGMENTATION_KEY_ATTRIBUTE_ID as segmentationKeyAttributeId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.PRIORITY as priority,
	t1.IS_USED as isUsed
from
	SEGMENTATION_KEY_ATTR_MAP t1
	inner join SEGMENTATION_KEY_ID_KEY t2 on t1.SEGMENTATION_KEY_ID = t2.SEGMENTATION_KEY_ID
	inner join SEGMENTATION_KEY_ID_REF t3 on t2.SEGMENTATION_KEY_ID = t3.SEGMENTATION_KEY_ID
		and t3.SERVICE_VERSION_ID = {1}
	inner join SEGMENTATION_KEY_ID_VALUES t4 on t2.SEGMENTATION_KEY_ID = t4.SEGMENTATION_KEY_ID
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join SEGMENTATION_KEY_ATTR_KEY t5 on t1.SEGMENTATION_KEY_ATTRIBUTE_ID = t5.ATTRIBUTE_ID
	inner join SEGMENTATION_KEY_ATTR_REF t6 on t5.ATTRIBUTE_ID = t6.ATTRIBUTE_ID
		and t6.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join SEGMENTATION_KEY_ATTR_VALUES t7 on t5.ATTRIBUTE_ID = t7.ATTRIBUTE_ID
		and t7.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
