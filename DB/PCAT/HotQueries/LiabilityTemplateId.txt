select
	t1.LR_TEMPLATE_ID as lrTemplateId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.SUGGESTED_PRIMARY_OFFER_ID as suggestedPrimaryOfferId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	LIABILITY_TEMPLATE_ID_KEY t1
	inner join LIABILITY_TEMPLATE_ID_REF t2 on t1.LR_TEMPLATE_ID = t2.LR_TEMPLATE_ID
	inner join LIABILITY_TEMPLATE_ID_VALUES t3 on t2.LR_TEMPLATE_ID = t3.LR_TEMPLATE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
