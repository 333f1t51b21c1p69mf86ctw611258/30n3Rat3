select
	t4.DISPLAY_VALUE as usageCategoryIdKey,
	t1.CATEGORY_ID as categoryId,
	t7.DISPLAY_VALUE as autInitialKey,
	t1.INITIAL_AUT_ID as initialAutId,
	t1.SERVICE_VERSION_ID as serviceVersionId
from
	USAGE_CATEGORY_MEMBER t1
	inner join USAGE_CATEGORY_ID_KEY t2 on t1.CATEGORY_ID = t2.USAGE_CATEGORY_ID
	inner join USAGE_CATEGORY_ID_REF t3 on t2.USAGE_CATEGORY_ID = t3.USAGE_CATEGORY_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join USAGE_CATEGORY_ID_VALUES t4 on t2.USAGE_CATEGORY_ID = t4.USAGE_CATEGORY_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join AUT_INITIAL_KEY t5 on t1.INITIAL_AUT_ID = t5.INITIAL_AUT_ID
	inner join AUT_INITIAL_REF t6 on t5.INITIAL_AUT_ID = t6.INITIAL_AUT_ID
		and t6.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join AUT_INITIAL_VALUES t7 on t5.INITIAL_AUT_ID = t7.INITIAL_AUT_ID
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
