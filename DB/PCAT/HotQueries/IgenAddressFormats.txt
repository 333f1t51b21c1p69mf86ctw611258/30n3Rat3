select
	t3.DISPLAY_VALUE as igenAddressFormatRef,
	t1.ADDRESS_FORMAT_ID as addressFormatId,
	t1.COUNTRY_CODE as countryCode,
	t5.DISPLAY_VALUE as languageCodeRef,
	t1.LANGUAGE_CODE as languageCode,
	t1.ADDRESS_LINE as addressLine
from
	IGEN_ADDRESS_FORMATS t1
	inner join IGEN_ADDRESS_FORMAT_REF t2 on t1.ADDRESS_FORMAT_ID = t2.ADDRESS_FORMAT_ID
	inner join IGEN_ADDRESS_FORMAT_VALUES t3 on t2.ADDRESS_FORMAT_ID = t3.ADDRESS_FORMAT_ID
		and t3.LANGUAGE_CODE = {2}
	inner join LANGUAGE_CODE_REF t4 on t1.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join LANGUAGE_CODE_VALUES t5 on t4.LANGUAGE_CODE = t5.INPUT_LANGUAGE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
