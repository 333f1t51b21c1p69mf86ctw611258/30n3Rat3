select
	t1.BMF_TRANS_TYPE as bmfTransType,
	t1.BMF_TRANS_CATEGORY as bmfTransCategory,
	t1.TRANS_SIGN as transSign,
	t3.DISPLAY_VALUE as billingCategory,
	t1.DESCRIPTION_CODE as descriptionCode,
	t1.ALLOW_MANUAL_ENTRY as allowManualEntry,
	t1.ALLOW_SUSPENSE_ACNT as allowSuspenseAcnt,
	t1.IS_DISPLAYED_ONBILL as isDisplayedOnbill,
	t1.IS_JOURNALABLE as isJournalable,
	t1.ALLOW_AUTOTRANSFER as allowAutotransfer
from
	BMF_TRANS_DESCR t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''BILLING_CATEGORY'')
		and t2.value = t1.BILLING_CATEGORY
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
