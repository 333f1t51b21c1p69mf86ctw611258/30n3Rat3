select
	t1.APPLICATION_ID as applicationId,
	t1.NAME as name,
	t1.LOCATION as location,
	t1.UPD_CNT as updCnt,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BSD_APPLICATION_REF t1
	inner join BSD_APPLICATION_VALUES t2 on t1.APPLICATION_ID = t2.APPLICATION_ID
where
	t2.LANGUAGE_CODE = {2}
