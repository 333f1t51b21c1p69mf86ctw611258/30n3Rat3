select
	t1.SERVER_ID as serverId,
	t1.DEVICE_ID as deviceId,
	t1.NDEVICE_ID as ndeviceId,
	t1.DEVICE_NAME as deviceName,
	t1.DEVICE_TYPE as deviceType,
	t1.USAGE_TYPE as usageType,
	t1.DEVICE_CLASS as deviceClass,
	t1.OPER_ASSIST as operAssist,
	t1.TASK_ID as taskId,
	t1.START_TIME as startTime,
	t1.END_TIME as endTime
from
	DEVICES t1
