select
	t1.PREREQ_RULE_MAP_ID as prereqRuleMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as prereqRuleKey,
	t1.PREREQ_RULE_ID as prereqRuleId,
	t7.DISPLAY_VALUE as bundleKey,
	t1.BUNDLE_ID as bundleId,
	t10.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId
from
	PREREQ_RULE_MEMBER_MAP t1
	inner join PREREQ_RULE_KEY t2 on t1.PREREQ_RULE_ID = t2.PREREQ_RULE_ID
	inner join PREREQ_RULE_REF t3 on t2.PREREQ_RULE_ID = t3.PREREQ_RULE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join PREREQ_RULE_VALUES t4 on t2.PREREQ_RULE_ID = t4.PREREQ_RULE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join BUNDLE_KEY t5 on t1.BUNDLE_ID = t5.BUNDLE_ID
	left outer join BUNDLE_REF t6 on t5.BUNDLE_ID = t6.BUNDLE_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join BUNDLE_VALUES t7 on t5.BUNDLE_ID = t7.BUNDLE_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join OFFER_KEY t8 on t1.OFFER_ID = t8.OFFER_ID
	left outer join OFFER_REF t9 on t8.OFFER_ID = t9.OFFER_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join OFFER_VALUES t10 on t8.OFFER_ID = t10.OFFER_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
