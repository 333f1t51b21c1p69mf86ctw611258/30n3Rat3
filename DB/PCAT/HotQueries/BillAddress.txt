select
	t3.DISPLAY_VALUE as billFmtOptRef,
	t1.BILL_FMT_OPT as billFmtOpt,
	t5.DISPLAY_VALUE as countryCodeRef,
	t1.COUNTRY_CODE as countryCode,
	t1.ADDRESS_FIELD as addressField,
	t1.ADDRESS_LINE as addressLine,
	t1.ADDRESS_LINE_ORDER as addressLineOrder,
	t1.FIELD_STATUS as fieldStatus,
	t1.IS_DEFAULT as isDefault
from
	BILL_ADDRESS t1
	inner join BILL_FMT_OPT_REF t2 on t1.BILL_FMT_OPT = t2.BILL_FMT_OPT
	inner join BILL_FMT_OPT_VALUES t3 on t2.BILL_FMT_OPT = t3.BILL_FMT_OPT
		and t3.LANGUAGE_CODE = {2}
	inner join COUNTRY_CODE_REF t4 on t1.COUNTRY_CODE = t4.COUNTRY_CODE
	inner join COUNTRY_CODE_VALUES t5 on t4.COUNTRY_CODE = t5.COUNTRY_CODE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
