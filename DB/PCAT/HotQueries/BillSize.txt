select
	t3.DISPLAY_VALUE as billFmtOptRef,
	t1.BILL_FMT_OPT as billFmtOpt,
	t5.DISPLAY_VALUE as paperSize,
	t1.IS_DEFAULT as isDefault
from
	BILL_SIZE t1
	inner join BILL_FMT_OPT_REF t2 on t1.BILL_FMT_OPT = t2.BILL_FMT_OPT
	inner join BILL_FMT_OPT_VALUES t3 on t2.BILL_FMT_OPT = t3.BILL_FMT_OPT
		and t3.LANGUAGE_CODE = {2}
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''PAPER_SIZE'')
		and t4.value = t1.PAPER_SIZE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
