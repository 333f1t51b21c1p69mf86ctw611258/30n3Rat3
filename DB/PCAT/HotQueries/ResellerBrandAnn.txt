select
	t1.RESELLER_BRAND_ANN_ID as resellerBrandAnnId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.PROMPT_ID as promptId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	RESELLER_BRAND_ANN_KEY t1
	inner join RESELLER_BRAND_ANN_REF t2 on t1.RESELLER_BRAND_ANN_ID = t2.RESELLER_BRAND_ANN_ID
	inner join RESELLER_BRAND_ANN_VALUES t3 on t2.RESELLER_BRAND_ANN_ID = t3.RESELLER_BRAND_ANN_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
