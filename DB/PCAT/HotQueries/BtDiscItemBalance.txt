select
	t4.DISPLAY_VALUE as btDiscItemBalTempltKey,
	t1.BAL_TEMPLATE_ID as balTemplateId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as balanceKey,
	t1.BALANCE_ID as balanceId,
	t1.AMOUNT as amount,
	t1.AMOUNT_IMPLIED_DECIMAL as amountImpliedDecimal,
	t1.CONVERSION_FACTOR as conversionFactor
from
	BT_DISC_ITEM_BALANCE t1
	inner join BT_DISC_ITEM_BAL_TEMPLT_KEY t2 on t1.BAL_TEMPLATE_ID = t2.BAL_TEMPLATE_ID
	inner join BT_DISC_ITEM_BAL_TEMPLT_REF t3 on t2.BAL_TEMPLATE_ID = t3.BAL_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BT_DISC_ITEM_BAL_TEMPLT_VALUES t4 on t2.BAL_TEMPLATE_ID = t4.BAL_TEMPLATE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join BALANCE_KEY t5 on t1.BALANCE_ID = t5.BALANCE_ID
	inner join BALANCE_REF t6 on t5.BALANCE_ID = t6.BALANCE_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BALANCE_VALUES t7 on t5.BALANCE_ID = t7.BALANCE_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
