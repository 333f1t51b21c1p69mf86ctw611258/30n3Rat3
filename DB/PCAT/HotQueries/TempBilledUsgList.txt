select
	t1.TYPE_ID_USG as typeIdUsg,
	t1.BILL_REF_NO as billRefNo,
	t1.BILL_REF_RESETS as billRefResets,
	t1.BILL_INVOICE_ROW as billInvoiceRow,
	t1.SUBSCR_NO as subscrNo,
	t1.SUBSCR_NO_RESETS as subscrNoResets,
	t1.EQUIP_EXTERNAL_ID as equipExternalId,
	t1.PROVIDER_ID as providerId,
	t1.ELEMENT_ID as elementId,
	t1.TRANSDATE as transdate,
	t1.BI_CURRENCY_CODE as biCurrencyCode,
	t1.AMOUNT as amount,
	t1.RATE_CURRENCY_CODE as rateCurrencyCode,
	t1.RATED_AMOUNT as ratedAmount,
	t1.UNITS as units,
	t1.TAX as tax,
	t1.TAX_RATE as taxRate,
	t1.DISCOUNT as discount,
	t1.ANNOTATION as annotation,
	t1.PROFILE_ID as profileId,
	t1.GEOCODE as geocode,
	t1.AUX_TAX_INFO as auxTaxInfo,
	t1.JURISDICTION as jurisdiction,
	t1.MSG_ID as msgId,
	t1.MSG_ID2 as msgId2,
	t1.SPLIT_ROW_NUM as splitRowNum,
	t1.BILL_CLASS as billClass,
	t1.POINT_TARGET as pointTarget,
	t1.UNITS_CREDITED as unitsCredited,
	t1.AMOUNT_CREDITED as amountCredited,
	t1.TIME_TYPE_ID as timeTypeId,
	t3.DISPLAY_VALUE as typeCode,
	t1.ACCOUNT_NO as accountNo,
	t1.ACCT_EXTERNAL_ID as acctExternalId,
	t1.TAX_CLASS as taxClass,
	t5.DISPLAY_VALUE as taxLocationUsg,
	t1.TAX_PKG_INST_ID as taxPkgInstId,
	t1.IS_ADJUSTABLE as isAdjustable,
	t1.RATABLE_UNIT_CLASS as ratableUnitClass,
	t1.POINT_ORIGIN as pointOrigin,
	t1.DISPLAY_VALUE as displayValue,
	t1.DESCRIPTION_CODE as descriptionCode,
	t1.AGGR_USAGE_ID as aggrUsageId,
	t1.DESCRIPTION_TEXT as descriptionText,
	t1.REV_RCV_COST_CTR as revRcvCostCtr,
	t1.B_REV_RCV_COST_CTR as bRevRcvCostCtr,
	t1.USG_AGG_AMOUNT as usgAggAmount,
	t1.BID_TAX as bidTax,
	t1.BID_DISCOUNT as bidDiscount,
	t1.BID_TRACKING_ID as bidTrackingId,
	t1.BID_SECONDARY_AMOUNT as bidSecondaryAmount,
	t1.OPEN_ITEM_ID as openItemId,
	t1.CELL_ID_ORIGIN as cellIdOrigin,
	t7.DISPLAY_VALUE as origType,
	t1.TRANS_SIGN as transSign,
	t1.INCLUSIVE_TAX as inclusiveTax
from
	TEMP_BILLED_USG_LIST t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''TYPE_CODE'')
		and t2.value = t1.TYPE_CODE
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''TAX_LOCATION_USG'')
		and t4.value = t1.TAX_LOCATION_USG
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = {2}
	left outer join GENERIC_ENUMERATION_REF t6 on t6.enumeration_key = lower(''ORIG_TYPE'')
		and t6.value = t1.ORIG_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t7 on t7.enumeration_key = t6.enumeration_key
		and t7.value = t6.value
		and t7.LANGUAGE_CODE = {2}
