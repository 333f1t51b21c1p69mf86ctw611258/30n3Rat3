select
	t1.ITEM_URL_ID as itemUrlId,
	t1.ITEM_ID as itemId,
	t4.DISPLAY_VALUE as ctcUrlTypeRef,
	t1.URL_TYPE_ID as urlTypeId,
	t1.URL as url
from
	CTC_ITEM_URL t1
	inner join CTC_ITEM t2 on t1.ITEM_ID = t2.ITEM_ID
	inner join CTC_URL_TYPE_REF t3 on t1.URL_TYPE_ID = t3.URL_TYPE_ID
	inner join CTC_URL_TYPE_VALUES t4 on t3.URL_TYPE_ID = t4.URL_TYPE_ID
		and t4.LANGUAGE_CODE = {2}
