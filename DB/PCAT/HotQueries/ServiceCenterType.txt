select
	t1.SERVICE_CENTER_TYPE as serviceCenterType,
	t1.IS_REQUIRED as isRequired,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	SERVICE_CENTER_TYPE_REF t1
	inner join SERVICE_CENTER_TYPE_VALUES t2 on t1.SERVICE_CENTER_TYPE = t2.SERVICE_CENTER_TYPE
where
	t2.LANGUAGE_CODE = {2}
