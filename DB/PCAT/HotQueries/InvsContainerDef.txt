select
	t1.CONTAINER_DEF_ID as containerDefId,
	t3.DISPLAY_VALUE as invsContainertypeRef,
	t1.CONTAINERTYPE_ID as containertypeId,
	t5.DISPLAY_VALUE as invsContainertypeRefByContaine,
	t1.CONTAINED_CONTAINERTYPE_ID as containedContainertypeId,
	t7.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t1.MINIMUM_COUNT as minimumCount,
	t1.MAXIMUM_COUNT as maximumCount
from
	INVS_CONTAINER_DEF t1
	inner join INVS_CONTAINERTYPE_REF t2 on t1.CONTAINERTYPE_ID = t2.CONTAINERTYPE_ID
	inner join INVS_CONTAINERTYPE_VALUES t3 on t2.CONTAINERTYPE_ID = t3.CONTAINERTYPE_ID
		and t3.LANGUAGE_CODE = {2}
	left outer join INVS_CONTAINERTYPE_REF t4 on t1.CONTAINED_CONTAINERTYPE_ID = t4.CONTAINERTYPE_ID
	left outer join INVS_CONTAINERTYPE_VALUES t5 on t4.CONTAINERTYPE_ID = t5.CONTAINERTYPE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join INVS_TYPE_REF t6 on t1.INVENTORY_TYPE_ID = t6.INVENTORY_TYPE_ID
	left outer join INVS_TYPE_VALUES t7 on t6.INVENTORY_TYPE_ID = t7.INVENTORY_TYPE_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
