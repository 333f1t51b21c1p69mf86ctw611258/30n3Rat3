select
	t1.THRESHOLD_ID as thresholdId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId,
	t6.DISPLAY_VALUE as thresholdType,
	t1.THRESHOLD_NUMBER as thresholdNumber,
	t1.THRESHOLD_VALUE as thresholdValue,
	t1.OFFER_BRAND_ANN_ID as offerBrandAnnId,
	t1.MAX_CALLS as maxCalls
from
	OFFER_PRE_THRESHOLD t1
	inner join OFFER_KEY t2 on t1.OFFER_ID = t2.OFFER_ID
	inner join OFFER_REF t3 on t2.OFFER_ID = t3.OFFER_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join OFFER_VALUES t4 on t2.OFFER_ID = t4.OFFER_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join GUI_INDICATOR_REF t5 on t5.table_name = ''OFFER_PRE_THRESHOLD''
		and t5.field_name = lower(''THRESHOLD_TYPE'')
		and t5.integer_value = t1.THRESHOLD_TYPE
	left outer join GUI_INDICATOR_VALUES t6 on t6.table_name = t5.table_name
		and t6.field_name = t5.field_name
		and t6.integer_value = t5.integer_value
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
