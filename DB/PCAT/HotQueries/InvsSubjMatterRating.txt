select
	t1.SUBJECT_MATTER_RATING as subjectMatterRating,
	t1.IS_INTERNAL as isInternal,
	t1.IS_DEFAULT as isDefault,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_SUBJ_MATTER_RATING_REF t1
	inner join INVS_SUBJ_MATTER_RATING_VALUES t2 on t1.SUBJECT_MATTER_RATING = t2.SUBJECT_MATTER_RATING
where
	t2.LANGUAGE_CODE = {2}
