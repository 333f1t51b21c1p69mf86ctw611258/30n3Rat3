select
	t1.HDR_TYPE as hdrType,
	t1.HDR_POSITION as hdrPosition,
	t1.JNL_FEED_CODE as jnlFeedCodeCode,
	t1.HDR_TABLE as hdrTable,
	t1.HDR_FIELD as hdrField,
	t1.HDR_FIELD_SIZE as hdrFieldSize,
	t1.HDR_HARDCODED as hdrHardcoded,
	t1.HDR_FIELD_FORMAT as hdrFieldFormat,
	t1.HDR_FIELD_TRIM as hdrFieldTrim,
	t1.HDR_FIELD_DESCRIPTION as hdrFieldDescription,
	t1.IN_FEED_REPORT as inFeedReport,
	t1.HDR_IS_PACK as hdrIsPack
from
	JNL_FEED_HEADER t1
	inner join JNL_FEED_CODE t2 on t1.JNL_FEED_CODE = t2.JNL_FEED_CODE
