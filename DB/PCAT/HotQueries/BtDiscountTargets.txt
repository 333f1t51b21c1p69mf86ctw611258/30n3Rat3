select
	t1.BT_DISCOUNT_TARGET_ID as btDiscountTargetId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as btDiscountItemKey,
	t1.DISCOUNT_ID as discountId,
	t6.DISPLAY_VALUE as targetType,
	t8.DISPLAY_VALUE as targetDomain,
	t1.TARGET_ID as targetId,
	t1.IS_EXCLUSION as isExclusion,
	t1.TARGET_ID2 as targetId2
from
	BT_DISCOUNT_TARGETS t1
	inner join BT_DISCOUNT_ITEM_KEY t2 on t1.DISCOUNT_ID = t2.DISCOUNT_ID
	inner join BT_DISCOUNT_ITEM_REF t3 on t2.DISCOUNT_ID = t3.DISCOUNT_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BT_DISCOUNT_ITEM_VALUES t4 on t2.DISCOUNT_ID = t4.DISCOUNT_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join GUI_INDICATOR_REF t5 on t5.table_name = ''BT_DISCOUNT_TARGETS''
		and t5.field_name = lower(''TARGET_TYPE'')
		and t5.integer_value = t1.TARGET_TYPE
	left outer join GUI_INDICATOR_VALUES t6 on t6.table_name = t5.table_name
		and t6.field_name = t5.field_name
		and t6.integer_value = t5.integer_value
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t7 on t7.table_name = ''BT_DISCOUNT_TARGETS''
		and t7.field_name = lower(''TARGET_DOMAIN'')
		and t7.integer_value = t1.TARGET_DOMAIN
	left outer join GUI_INDICATOR_VALUES t8 on t8.table_name = t7.table_name
		and t8.field_name = t7.field_name
		and t8.integer_value = t7.integer_value
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
