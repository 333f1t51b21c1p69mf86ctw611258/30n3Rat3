select
	t1.DEPENDENCY_TYPE_ID as dependencyTypeId,
	t1.DEPENDENCY_CATEGORY as dependencyCategory,
	t1.CANCEL_ALLOWED as cancelAllowed,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ORD_DEPENDENCY_TYPE_REF t1
	inner join ORD_DEPENDENCY_TYPE_VALUES t2 on t1.DEPENDENCY_TYPE_ID = t2.DEPENDENCY_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
