select
	t3.DISPLAY_VALUE as serviceCenterTypeRef,
	t1.SERVICE_CENTER_TYPE as serviceCenterType,
	t1.SERVICE_CENTER_ID as serviceCenterId
from
	SERVICE_CENTER_TYPES t1
	inner join SERVICE_CENTER_TYPE_REF t2 on t1.SERVICE_CENTER_TYPE = t2.SERVICE_CENTER_TYPE
	inner join SERVICE_CENTER_TYPE_VALUES t3 on t2.SERVICE_CENTER_TYPE = t3.SERVICE_CENTER_TYPE
		and t3.LANGUAGE_CODE = {2}
	inner join SERVICE_CENTERS t4 on t1.SERVICE_CENTER_ID = t4.SERVICE_CENTER_ID
