select
	t1.FIELD_NAME as fieldName,
	t1.TABLE_NAME as tableName,
	t1.INTEGER_VALUE as integerValue,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	GUI_INDICATOR_REF t1
	inner join GUI_INDICATOR_VALUES t2 on t1.FIELD_NAME = t2.FIELD_NAME
		and t1.TABLE_NAME = t2.TABLE_NAME
		and t1.INTEGER_VALUE = t2.INTEGER_VALUE
where
	t2.LANGUAGE_CODE = {2}
