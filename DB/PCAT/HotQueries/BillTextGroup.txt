select
	t1.TEXT_GROUP_ID as textGroupId,
	t4.DISPLAY_VALUE as billTextGroupRef,
	t1.PARENT_GROUP_ID as parentGroupId,
	t1.IS_INTERNAL as isInternal,
	t1.IS_HIDDEN as isHidden,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_TEXT_GROUP_REF t1
	inner join BILL_TEXT_GROUP_VALUES t2 on t1.TEXT_GROUP_ID = t2.TEXT_GROUP_ID
	left outer join BILL_TEXT_GROUP_REF t3 on t1.PARENT_GROUP_ID = t3.TEXT_GROUP_ID
	left outer join BILL_TEXT_GROUP_VALUES t4 on t3.TEXT_GROUP_ID = t4.TEXT_GROUP_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
