select
	t1.GEOGRAPHIC_REGION_ID as geographicRegionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_GEOGRAPHIC_REGION_REF t1
	inner join INVS_GEOGRAPHIC_REGION_VALUES t2 on t1.GEOGRAPHIC_REGION_ID = t2.GEOGRAPHIC_REGION_ID
where
	t2.LANGUAGE_CODE = {2}
