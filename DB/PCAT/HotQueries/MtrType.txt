select
	t1.MTR_TYPE_ID as mtrTypeId,
	t1.MTR_SUB_TYPE_ID as mtrSubTypeId,
	t4.DISPLAY_VALUE as mtrCategoryRef,
	t1.MTR_CATEGORY_ID as mtrCategoryId,
	t1.IS_JOURNALABLE as isJournalable,
	t1.NO_BILL as noBill,
	t1.SAVE_DB as saveDb,
	t1.IS_INTERNAL as isInternal,
	t1.IS_DEFAULT as isDefault,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	MTR_TYPE_REF t1
	inner join MTR_TYPE_VALUES t2 on t1.MTR_TYPE_ID = t2.MTR_TYPE_ID
		and t1.MTR_SUB_TYPE_ID = t2.MTR_SUB_TYPE_ID
	inner join MTR_CATEGORY_REF t3 on t1.MTR_CATEGORY_ID = t3.MTR_CATEGORY_ID
	inner join MTR_CATEGORY_VALUES t4 on t3.MTR_CATEGORY_ID = t4.MTR_CATEGORY_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
