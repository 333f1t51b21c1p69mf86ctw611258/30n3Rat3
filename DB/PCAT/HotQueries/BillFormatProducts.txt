select
	t1.FORMAT_CODE as formatCode,
	t3.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as productLineKey,
	t1.PRODUCT_LINE_ID as productLineId,
	t1.MSG_TRIGGER_ID as msgTriggerId,
	t1.PRINTER_TEXT as printerText,
	t1.SORT_PRIORITY as sortPriority,
	t1.ALT_PRINTER_TEXT as altPrinterText
from
	BILL_FORMAT_PRODUCTS t1
	inner join PRODUCT_LINE_KEY t2 on t1.PRODUCT_LINE_ID = t2.PRODUCT_LINE_ID
	inner join PRODUCT_LINE_REF t3 on t2.PRODUCT_LINE_ID = t3.PRODUCT_LINE_ID
		and t3.RESELLER_VERSION_ID = {0}
	inner join PRODUCT_LINE_VALUES t4 on t2.PRODUCT_LINE_ID = t4.PRODUCT_LINE_ID
		and t4.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
