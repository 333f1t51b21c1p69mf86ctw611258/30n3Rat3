select
	t1.BONUS_POINT_ID as bonusPointId,
	t4.DISPLAY_VALUE as roundingMethod,
	t1.USE_BY as useBy,
	t1.USE_BY_UNITS as useByUnits,
	t1.USE_BY_DATE as useByDate,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	BONUS_POINT_ID_REF t1
	inner join BONUS_POINT_ID_VALUES t2 on t1.BONUS_POINT_ID = t2.BONUS_POINT_ID
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''ROUNDING_METHOD'')
		and t3.value = t1.ROUNDING_METHOD
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
