select
	t1.FORMAT_CODE as formatCode,
	t3.DISPLAY_VALUE as billFmtSectionTypeRef,
	t1.SECTION_TYPE as sectionType,
	t1.CODE_VALUE as codeValue,
	t1.SORT_PRIORITY as sortPriority,
	t1.RC_TERM_ID as rcTermId,
	t5.DISPLAY_VALUE as billFormatCodeTypeRef,
	t1.CODE_TYPE as codeType,
	t7.DISPLAY_VALUE as equipTypeCodeRef,
	t1.EQUIP_TYPE_CODE as equipTypeCode,
	t9.DISPLAY_VALUE as billFmtTemplateCodeRef,
	t1.TEMPLATE_CODE as templateCode,
	t11.DISPLAY_VALUE as equipClassCodeRef,
	t1.EQUIP_CLASS_CODE as equipClassCode,
	t13.DISPLAY_VALUE as classOfServiceCodeRef,
	t1.CLASS_OF_SERVICE_CODE as classOfServiceCode,
	t15.SERVICE_VERSION_ID as serviceVersionId,
	t16.DISPLAY_VALUE as serviceProvidersKey,
	t1.PROVIDER_ID as providerId,
	t19.DISPLAY_VALUE as jurisdictionsKey,
	t1.JURISDICTION as jurisdiction,
	t1.USAGE_PLAN_ID as usagePlanId,
	t1.OFFER_ID as offerId,
	t1.TABLE_ORDER as tableOrder,
	t1.FORMAT_DISPLAY as formatDisplay
from
	BILL_FORMAT_TEMPLATE t1
	inner join BILL_FMT_SECTION_TYPE_REF t2 on t1.SECTION_TYPE = t2.SECTION_TYPE
	inner join BILL_FMT_SECTION_TYPE_VALUES t3 on t2.SECTION_TYPE = t3.SECTION_TYPE
		and t3.LANGUAGE_CODE = {2}
	inner join BILL_FORMAT_CODE_TYPE_REF t4 on t1.CODE_TYPE = t4.CODE_TYPE
	inner join BILL_FORMAT_CODE_TYPE_VALUES t5 on t4.CODE_TYPE = t5.CODE_TYPE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EQUIP_TYPE_CODE_REF t6 on t1.EQUIP_TYPE_CODE = t6.EQUIP_TYPE_CODE
	left outer join EQUIP_TYPE_CODE_VALUES t7 on t6.EQUIP_TYPE_CODE = t7.EQUIP_TYPE_CODE
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join BILL_FMT_TEMPLATE_CODE_REF t8 on t1.TEMPLATE_CODE = t8.TEMPLATE_CODE
	inner join BILL_FMT_TEMPLATE_CODE_VALUES t9 on t8.TEMPLATE_CODE = t9.TEMPLATE_CODE
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EQUIP_CLASS_CODE_REF t10 on t1.EQUIP_CLASS_CODE = t10.EQUIP_CLASS_CODE
	left outer join EQUIP_CLASS_CODE_VALUES t11 on t10.EQUIP_CLASS_CODE = t11.EQUIP_CLASS_CODE
		and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join CLASS_OF_SERVICE_CODE_REF t12 on t1.CLASS_OF_SERVICE_CODE = t12.CLASS_OF_SERVICE_CODE
	left outer join CLASS_OF_SERVICE_CODE_VALUES t13 on t12.CLASS_OF_SERVICE_CODE = t13.CLASS_OF_SERVICE_CODE
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join SERVICE_PROVIDERS_KEY t14 on t1.PROVIDER_ID = t14.PROVIDER_ID
	left outer join SERVICE_PROVIDERS_REF t15 on t14.PROVIDER_ID = t15.PROVIDER_ID
		and t15.SERVICE_VERSION_ID = {1}
	left outer join SERVICE_PROVIDERS_VALUES t16 on t14.PROVIDER_ID = t16.PROVIDER_ID
		and t16.SERVICE_VERSION_ID = {1}
		and t16.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join JURISDICTIONS_KEY t17 on t1.JURISDICTION = t17.JURISDICTION
	left outer join JURISDICTIONS_REF t18 on t17.JURISDICTION = t18.JURISDICTION
		and t18.SERVICE_VERSION_ID = {1}
	left outer join JURISDICTIONS_VALUES t19 on t17.JURISDICTION = t19.JURISDICTION
		and t19.SERVICE_VERSION_ID = {1}
		and t19.LANGUAGE_CODE = t3.LANGUAGE_CODE
