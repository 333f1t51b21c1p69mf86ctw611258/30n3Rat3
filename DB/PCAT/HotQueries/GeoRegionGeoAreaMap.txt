select
	t4.DISPLAY_VALUE as geoAreaKey,
	t1.GEO_AREA_ID as geoAreaId,
	t7.DISPLAY_VALUE as geoRegionKey,
	t1.GEO_REGION_ID as geoRegionId,
	t1.SERVICE_VERSION_ID as serviceVersionId
from
	GEO_REGION_GEO_AREA_MAP t1
	inner join GEO_AREA_KEY t2 on t1.GEO_AREA_ID = t2.GEO_AREA_ID
	inner join GEO_AREA_REF t3 on t2.GEO_AREA_ID = t3.GEO_AREA_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join GEO_AREA_VALUES t4 on t2.GEO_AREA_ID = t4.GEO_AREA_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join GEO_REGION_KEY t5 on t1.GEO_REGION_ID = t5.GEO_REGION_ID
	inner join GEO_REGION_REF t6 on t5.GEO_REGION_ID = t6.GEO_REGION_ID
		and t6.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join GEO_REGION_VALUES t7 on t5.GEO_REGION_ID = t7.GEO_REGION_ID
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
