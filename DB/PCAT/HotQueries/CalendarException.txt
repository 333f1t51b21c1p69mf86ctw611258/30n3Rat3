select
	t1.CALENDAR_EXCEPTION_ID as calendarExceptionId,
	t4.DISPLAY_VALUE as calendarKey,
	t1.CALENDAR_ID as calendarId,
	t1.EXCEPTION_DATE as exceptionDate,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as dayTypeIdKey,
	t1.DAY_TYPE_ID as dayTypeId
from
	CALENDAR_EXCEPTION t1
	inner join CALENDAR_KEY t2 on t1.CALENDAR_ID = t2.CALENDAR_ID
	inner join CALENDAR_REF t3 on t2.CALENDAR_ID = t3.CALENDAR_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join CALENDAR_VALUES t4 on t2.CALENDAR_ID = t4.CALENDAR_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join DAY_TYPE_ID_KEY t5 on t1.DAY_TYPE_ID = t5.DAY_TYPE_ID
	inner join DAY_TYPE_ID_REF t6 on t5.DAY_TYPE_ID = t6.DAY_TYPE_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join DAY_TYPE_ID_VALUES t7 on t5.DAY_TYPE_ID = t7.DAY_TYPE_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
