select
	t1.CDR_ADJ_ERROR as cdrAdjError,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CDR_ADJ_ERROR_REF t1
	inner join CDR_ADJ_ERROR_VALUES t2 on t1.CDR_ADJ_ERROR = t2.CDR_ADJ_ERROR
where
	t2.LANGUAGE_CODE = {2}
