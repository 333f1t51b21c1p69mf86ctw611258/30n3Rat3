select
	t1.STATUS_REASON_ID as statusReasonId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	STATUS_REASON_REF t1
	inner join STATUS_REASON_VALUES t2 on t1.STATUS_REASON_ID = t2.STATUS_REASON_ID
where
	t2.LANGUAGE_CODE = {2}
