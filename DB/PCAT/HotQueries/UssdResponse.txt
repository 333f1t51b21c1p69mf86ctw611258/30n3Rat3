select
	t1.USSD_RESPONSE_ID as ussdResponseId,
        t4.DISPLAY_VALUE as responseCode,
        t1.RESPONSE_CODE as responseCodeNum,
	t1.TONE_ID as toneId,
	t1.NOTIF_IVR_LANGUAGE_CODE as notifIvrLanguageCode,
        t6.DISPLAY_VALUE as notifIvrLanguageRef,
	t1.DEFAULT_MSG as defaultMsg,
	t1.UNICODE_FLAG as unicodeFlag,
        t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	USSD_RESPONSE_REF t1
        inner join USSD_RESPONSE_VALUES t2 on t1.USSD_RESPONSE_ID = t2.USSD_RESPONSE_ID
        left outer join GUI_INDICATOR_REF t3 on t3.table_name = ''USSD_RESPONSE_REF''
		and t3.field_name = lower(''RESPONSE_CODE'')
		and t3.integer_value = t1.RESPONSE_CODE
	left outer join GUI_INDICATOR_VALUES t4 on t4.table_name = t3.table_name
		and t4.field_name = t3.field_name
		and t4.integer_value = t3.integer_value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
        inner join NOTIF_IVR_LANGUAGE_REF t5 on t1.NOTIF_IVR_LANGUAGE_CODE = t5.NOTIF_IVR_LANGUAGE_ID
        inner join NOTIF_IVR_LANGUAGE_VALUES t6 on t5.NOTIF_IVR_LANGUAGE_ID = t6.NOTIF_IVR_LANGUAGE_ID
                and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
