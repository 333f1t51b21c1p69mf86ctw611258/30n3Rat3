select
	t1.SECTION_TYPE as sectionType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_FMT_SECTION_TYPE_REF t1
	inner join BILL_FMT_SECTION_TYPE_VALUES t2 on t1.SECTION_TYPE = t2.SECTION_TYPE
where
	t2.LANGUAGE_CODE = {2}
