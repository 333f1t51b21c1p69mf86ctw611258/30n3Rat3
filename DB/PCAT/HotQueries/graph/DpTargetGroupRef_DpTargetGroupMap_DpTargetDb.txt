select
	t1.TARGET_DB_ID as targetDbId,
	t1.TARGET_GROUP_ID as targetGroupId,
	t1.REGION_ID as regionId,
	t2.HOSTNAME as hostname,
	t2.DSQUERY as dsQuery,
	t2.DS_DATABASE as dsDatabase,
	t2.TARGET_DB_TYPE as targetDbType,
	t5.DISPLAY_VALUE as regionName
from
	DP_TARGET_GROUP_MAP t1
	inner join DP_TARGET_DB t2 on t1.TARGET_DB_ID = t2.TARGET_DB_ID
	inner join DP_TARGET_GROUP_REF t3 on t1.TARGET_GROUP_ID = t3.TARGET_GROUP_ID
	inner join DP_TARGET_GROUP_VALUES t4 on t1.TARGET_GROUP_ID = t4.TARGET_GROUP_ID
		and t4.LANGUAGE_CODE = {2}
	inner join REGION_VALUES t5 on t1.REGION_ID = t5.REGION_ID
		and t5.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t5.SERVICE_VERSION_ID = {1}
