select
	t1.RC_TERM_ID as rcTermId,
	t1.BUNDLE_ID as bundleId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as name,
	t1.OFFER_OVERRIDE as isOverride,
	t9.DISPLAY_VALUE as assocType,
    t15.DISPLAY_VALUE as frequency,
    ''{3}'' as termType
from
	BUNDLE_RC_TERM_MAP t1
	inner join BUNDLE_VALUES t4 on t1.BUNDLE_ID = t4.BUNDLE_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join RC_TERM_VALUES t6 on t1.RC_TERM_ID = t6.RC_TERM_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join GUI_INDICATOR_REF t8 on t8.table_name = ''BUNDLE_RC_TERM_MAP''
		and t8.field_name = lower(''ASSOCIATION_TYPE'')
		and t8.integer_value = t1.ASSOCIATION_TYPE
	left outer join GUI_INDICATOR_VALUES t9 on t9.table_name = t8.table_name
		and t9.field_name = t8.field_name
		and t9.integer_value = t8.integer_value
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join BILLING_FREQUENCY_REF t14 on t1.PERIOD_FREQUENCY = t14.BILLING_FREQUENCY
	left outer join BILLING_FREQUENCY_VALUES t15 on t1.PERIOD_FREQUENCY = t15.BILLING_FREQUENCY
		and t15.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
