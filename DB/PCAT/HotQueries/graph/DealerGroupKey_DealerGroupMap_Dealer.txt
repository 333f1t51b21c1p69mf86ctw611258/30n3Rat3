select
	t1.DEALER_GROUP_ID as dealerGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.DEALER_ID as dealerId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
	t4.DISPLAY_VALUE as dealerType
from
	DEALER_GROUP_MAP t1
	inner join DEALER_REF t2 on t1.DEALER_ID = t2.DEALER_ID
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join DEALER_VALUES t3 on t1.DEALER_ID = t3.DEALER_ID
		and t3.LANGUAGE_CODE = {2}
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join DEALER_TYPE_VALUES t4 on t2.DEALER_TYPE_ID = t4.DEALER_TYPE_ID
		and t4.LANGUAGE_CODE = t3.LANGUAGE_CODE
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
