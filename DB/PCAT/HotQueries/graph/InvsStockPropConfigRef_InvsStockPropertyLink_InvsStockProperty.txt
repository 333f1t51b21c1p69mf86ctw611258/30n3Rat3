select
	t1.STOCK_PROPERTY_CONFIG_ID as stockPropertyConfigId,
	t1.STOCK_PROPERTY_ID as stockPropertyId,
	t5.DISPLAY_VALUE as name,
	t1.STOCK_PROPERTY_VALUE as stockPropertyValue
from
	INVS_STOCK_PROPERTY_LINK t1
	inner join INVS_STOCK_PROP_CONFIG_REF t2 on t1.STOCK_PROPERTY_CONFIG_ID = t2.STOCK_PROPERTY_CONFIG_ID
	inner join INVS_STOCK_PROP_CONFIG_VALUES t3 on t2.STOCK_PROPERTY_CONFIG_ID = t3.STOCK_PROPERTY_CONFIG_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_STOCK_PROPERTY_REF t4 on t1.STOCK_PROPERTY_ID = t4.STOCK_PROPERTY_ID
	inner join INVS_STOCK_PROPERTY_VALUES t5 on t4.STOCK_PROPERTY_ID = t5.STOCK_PROPERTY_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
