select
	t1.RESELLER_ID as resellerId,
	t1.REGION_ID as regionId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t2.DISPLAY_VALUE as name,
	t2.DESCRIPTION as description
from
	RESELLER_REGION_MAP t1
	inner join REGION_VALUES t2 on t1.REGION_ID = t2.REGION_ID
		and t2.LANGUAGE_CODE = {2}
		and t2.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
where
	t1.SERVICE_VERSION_ID = {1}
