select
        t1.EQUIP_CHARGE_CODE as equipChargeCode,
        t1.RC_TERM_ID as rcTermId,
        t1.RESELLER_VERSION_ID as resellerVersionId,
        t2.DISPLAY_VALUE as name,
        t2.DESCRIPTION as description
from
        EQUIP_CHARGE_CODE_RC_TERM_MAP t1
        inner join INVS_EQUIP_CHARGE_CODE_VALUES t2 on t1.EQUIP_CHARGE_CODE = t2.EQUIP_CHARGE_CODE
                and t2.LANGUAGE_CODE = {2}
        inner join RC_TERM_VALUES t3 on t1.RC_TERM_ID = t3.RC_TERM_ID
                and t3.LANGUAGE_CODE = t2.LANGUAGE_CODE
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
