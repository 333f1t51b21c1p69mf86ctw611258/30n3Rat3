select
	t1.INVENTORY_LINE_ID as inventoryLineId,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t5.DISPLAY_VALUE as name,
	t1.IS_CSR_INSERT as isCsrInsert,
	t1.MANAGE_AS_PLACEHOLDER as manageAsPlaceholder,
	t1.IS_ANNOTATION_REQUIRED as isAnnotationRequired
from
	INVS_LINE_TYPE_MAP t1
	inner join INVS_LINE_REF t2 on t1.INVENTORY_LINE_ID = t2.INVENTORY_LINE_ID
	inner join INVS_LINE_VALUES t3 on t2.INVENTORY_LINE_ID = t3.INVENTORY_LINE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_TYPE_REF t4 on t1.INVENTORY_TYPE_ID = t4.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t5 on t4.INVENTORY_TYPE_ID = t5.INVENTORY_TYPE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
