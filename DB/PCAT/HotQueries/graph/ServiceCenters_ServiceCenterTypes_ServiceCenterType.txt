select
	t1.SERVICE_CENTER_ID as serviceCenterId,
	t2.SC_NAME as scName,
	t4.DISPLAY_VALUE as name
from
	SERVICE_CENTER_TYPES t1
	inner join SERVICE_CENTERS t2 on t1.SERVICE_CENTER_ID = t2.SERVICE_CENTER_ID
	inner join SERVICE_CENTER_TYPE_REF t3 on t1.SERVICE_CENTER_TYPE = t3.SERVICE_CENTER_TYPE
	inner join SERVICE_CENTER_TYPE_VALUES t4 on t1.SERVICE_CENTER_TYPE = t4.SERVICE_CENTER_TYPE
		and t4.LANGUAGE_CODE = {2}
