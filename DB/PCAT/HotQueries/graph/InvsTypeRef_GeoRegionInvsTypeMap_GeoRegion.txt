select
	t1.GEO_REGION_ID as geoRegionId,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t1.SERVICE_VERSION_ID serviceVersionId,
	t5.DISPLAY_VALUE as name,
	t5.DESCRIPTION as description	
from
	GEO_REGION_INVS_TYPE_MAP t1
	inner join INVS_TYPE_REF t2 on t1.INVENTORY_TYPE_ID = t2.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t3 on t1.INVENTORY_TYPE_ID = t3.INVENTORY_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join GEO_REGION_REF t4 on t1.GEO_REGION_ID = t4.GEO_REGION_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join GEO_REGION_VALUES t5 on t1.GEO_REGION_ID = t5.GEO_REGION_ID
		and t5.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
