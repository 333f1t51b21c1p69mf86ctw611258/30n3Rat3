select
	t1.AUT_GROUP_MAP_ID as autGroupMapId,
	t1.AUT_GROUP_ID as autGroupId,
	t1.AUT_ID as autId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t2.DISPLAY_VALUE as name,
	t4.DISPLAY_VALUE as application,
	t5.DISPLAY_VALUE as subType,
	t6.DISPLAY_VALUE as unitType
from
	AUT_GROUP_MAP t1
	inner join AUT_FINAL_VALUES t2 on t1.AUT_ID = t2.AUT_ID
		and t2.LANGUAGE_CODE = {2}
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join AUT_FINAL_REF t3 on t1.AUT_ID = t3.AUT_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join APPLICATION_ID_VALUES t4 on t3.APPLICATION_ID = t4.APPLICATION_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
		and t4.SERVICE_VERSION_ID = {1}
	left outer join SUB_TYPE_ID_VALUES t5 on t3.SUB_TYPE_ID = t5.SUB_TYPE_ID
		and t5.LANGUAGE_CODE = t2.LANGUAGE_CODE
		and t5.SERVICE_VERSION_ID = t4.SERVICE_VERSION_ID
	left outer join UNITS_TYPE_VALUES t6 on t3.UNIT_TYPE = t6.UNIT_TYPE
		and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
		and t6.SERVICE_VERSION_ID = t4.SERVICE_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
