select
	t1.OFFER_ID as offerId,
	t1.ACCUMULATOR_ID as accumulatorId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as name,
	t11.DISPLAY_VALUE as accumulatorType,
	t13.DISPLAY_VALUE as countType,
	t7.DISPLAY_VALUE as unitType,
	t9.DISPLAY_VALUE as period
from
	OFFER_ACCUMULATOR_MAP t1
	inner join ACCUMULATOR_KEY t2 on t1.ACCUMULATOR_ID = t2.ACCUMULATOR_ID
	inner join ACCUMULATOR_REF t3 on t1.ACCUMULATOR_ID = t3.ACCUMULATOR_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join ACCUMULATOR_VALUES t4 on t1.ACCUMULATOR_ID = t4.ACCUMULATOR_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join UNITS_TYPE_KEY t5 on t3.UNIT_TYPE = t5.UNIT_TYPE
	inner join UNITS_TYPE_REF t6 on t3.UNIT_TYPE = t6.UNIT_TYPE
		and t6.SERVICE_VERSION_ID = {1}
	inner join UNITS_TYPE_VALUES t7 on t3.UNIT_TYPE = t7.UNIT_TYPE
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
		and t7.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
	left outer join GUI_INDICATOR_REF t8 on t8.field_name = lower(''PERIOD'')
		and t8.integer_value = t3.PERIOD
	left outer join GUI_INDICATOR_VALUES t9 on t9.field_name = t8.field_name
		and t9.integer_value = t8.integer_value
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t10 on t10.enumeration_key = lower(''ACCUMULATOR_TYPE'')
		and t10.value = t3.ACCUMULATOR_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t11 on t11.enumeration_key = t10.enumeration_key
		and t11.value = t10.value
		and t11.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t12 on t12.enumeration_key = lower(''COUNT_TYPE'')
		and t12.value = t3.COUNT_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t13 on t13.enumeration_key = t12.enumeration_key
		and t13.value = t12.value
		and t13.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
