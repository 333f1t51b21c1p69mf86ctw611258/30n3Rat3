select
        t1.MARKETING_CATEGORY_MAP_ID as marketingCategoryMapId,
        t1.RESELLER_VERSION_ID as resellerVersionId,
        t1.PC_ENTITY_TYPE as pcEntityType,
        t1.PC_ENTITY_ID as pcEntityId,
        t6.DISPLAY_VALUE as name,
        t6.DESCRIPTION as description
from
        marketing_category_map t1
        inner join MARKETING_CATEGORY_VALUES t4 on t1.MARKETING_CATEGORY_ID = t4.MARKETING_CATEGORY_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
        inner join AUT_FINAL_VALUES t6 on t1.PC_ENTITY_ID = t6.AUT_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
        left outer join AUT_FINAL_REF t9 on t9.AUT_ID = t1.PC_ENTITY_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join AUT_FINAL_VALUES t10 on t10.AUT_ID = t9.AUT_ID
		and t10.LANGUAGE_CODE = {2}
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID  
where
	t1.RESELLER_VERSION_ID = {0}
		and t1.PC_ENTITY_TYPE = ''AUT''
