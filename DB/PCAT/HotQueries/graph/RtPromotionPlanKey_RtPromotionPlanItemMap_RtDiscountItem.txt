select
	t1.RT_PROMO_PI_MAP_ID as rtPromoPiMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.RT_PROMOTION_PLAN_ID as rtPromotionPlanId,
	t1.DISCOUNT_ITEM_ID as discountItemId,
	t2.DISPLAY_VALUE as name,
	t2.DESCRIPTION as description
from
	RT_PROMOTION_PLAN_ITEM_MAP t1
	left outer join RT_DISCOUNT_ITEM_VALUES t2 on t1.DISCOUNT_ITEM_ID = t2.DISCOUNT_ITEM_ID
		and t2.LANGUAGE_CODE = {2}
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
		and t1.DISCOUNT_ITEM_ID is not null
