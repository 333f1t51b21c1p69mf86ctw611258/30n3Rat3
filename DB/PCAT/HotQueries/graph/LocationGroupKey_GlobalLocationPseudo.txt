select
	t1.LOCATION_GROUP_ID as locationGroupId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t2.LOCATION as name,
	''{3}'' as locationType
from
	LOCATION_GROUP_MEMBER t1
        inner join GLOBAL_LOCATION t2 on t2.GLOBAL_LOCATION_ID = t1.GLOBAL_LOCATION_ID
		and t2.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
where
	t1.SERVICE_VERSION_ID = {1}
and     t2.LI like ''2%''
