select
	t1.USAGE_PLAN_ID as usagePlanId,
	t1.OFFER_ID as offerId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as name,
	t6.DESCRIPTION as description,
    t7.RECOMMENDED_RC_VALUE as recRcVal,
    ''{3}'' as planType
from
	OFFER_USAGE_PLAN_MAP t1
	inner join OFFER_VALUES t4 on t1.OFFER_ID = t4.OFFER_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join USAGE_PLAN_KEY t5 on t1.USAGE_PLAN_ID = t5.USAGE_PLAN_ID
	inner join USAGE_PLAN_VALUES t6 on t1.USAGE_PLAN_ID = t6.USAGE_PLAN_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join USAGE_PLAN_REF t7 on t1.USAGE_PLAN_ID = t7.USAGE_PLAN_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
