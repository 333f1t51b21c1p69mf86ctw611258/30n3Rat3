select
	t4.GEO_AVAIL_RULE_ID as geoAvailRuleId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
	t1.BUNDLE_ID as bundleId,
    ''{3}'' as ruleType
from
	GEO_AVAIL_RULE_KEY t4
	inner join GEO_AVAIL_RULE_REF t2 on t4.GEO_AVAIL_RULE_ID = t2.GEO_AVAIL_RULE_ID
	inner join GEO_AVAIL_RULE_VALUES t3 on t2.GEO_AVAIL_RULE_ID = t3.GEO_AVAIL_RULE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join BUNDLE_REF t1 on t4.GEO_AVAIL_RULE_ID = t1.GEO_AVAIL_RULE_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
