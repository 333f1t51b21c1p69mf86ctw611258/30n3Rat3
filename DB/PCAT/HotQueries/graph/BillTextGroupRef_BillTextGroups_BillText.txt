select
	t1.TEXT_GROUP_ID as textGroupId,
        t4.DISPLAY_VALUE as textGroupName,
        t1.TEXT_ID as textId,
        t6.DISPLAY_VALUE as textName
from
	BILL_TEXT_GROUPS t1
	inner join BILL_TEXT_GROUP_REF t3 on t1.TEXT_GROUP_ID = t3.TEXT_GROUP_ID
	inner join BILL_TEXT_GROUP_VALUES t4 on t1.TEXT_GROUP_ID = t4.TEXT_GROUP_ID
		and t4.LANGUAGE_CODE = {2}
	inner join BILL_TEXT_REF t5 on t1.TEXT_ID = t5.TEXT_ID
	inner join BILL_TEXT_VALUES t6 on t1.TEXT_ID = t6.TEXT_ID
		and t5.TEST_FLAG = t6.TEST_FLAG
		and t6.LANGUAGE_CODE = {2}
