select
        t1.BUNDLE_ID as bundleId,
        t2.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
        t4.CAMPAIGN_TREATMENT_ID as campaignTreatmentId,
        ''{3}'' as itemType
from
        BUNDLE_KEY t1
        inner join BUNDLE_REF t2 on t1.BUNDLE_ID = t2.BUNDLE_ID
	inner join BUNDLE_VALUES t3 on t2.BUNDLE_ID = t3.BUNDLE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join CAMPAIGN_TREATMENT_REF t4 on t1.BUNDLE_ID = t4.BUNDLE_ID
		and t2.RESELLER_VERSION_ID = t4.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
