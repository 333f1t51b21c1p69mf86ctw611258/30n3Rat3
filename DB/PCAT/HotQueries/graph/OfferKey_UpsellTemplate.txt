select
	t4.UPSELL_TEMPLATE_ID as upsellTemplateId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
    ''{3}'' as templateType
from
	UPSELL_TEMPLATE_KEY t4
	inner join UPSELL_TEMPLATE_REF t2 on t4.UPSELL_TEMPLATE_ID = t2.UPSELL_TEMPLATE_ID
	inner join UPSELL_TEMPLATE_VALUES t3 on t2.UPSELL_TEMPLATE_ID = t3.UPSELL_TEMPLATE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join OFFER_REF t1 on t4.UPSELL_TEMPLATE_ID = t1.UPSELL_TEMPLATE_ID
        	and t1.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
