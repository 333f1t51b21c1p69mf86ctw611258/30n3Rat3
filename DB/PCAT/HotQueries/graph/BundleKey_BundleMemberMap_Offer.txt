select
	t1.BUNDLE_MEMBER_MAP_ID as bundleMemberMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.MEMBER_ID as memberId,
	t1.MEMBER_ID as offerId,
	t10.DISPLAY_VALUE as offerName,
	t16.DISPLAY_VALUE as offerType,
	t8.DISPLAY_VALUE as association,
	t1.IS_DEFAULT as isDefault,
	t13.DISPLAY_VALUE as paymentMode,
	t14.DISPLAY_VALUE as serviceCategory,
	t1.BALANCE_PRIORITY as balPriOrder,
	t1.TARIFF_PRIORITY as tariffPlanPriOrder,
	t1.DISCOUNT_PRIORITY as discountPriOrder,
	t1.RC_PRIORITY as rcTermPriOrder,
	t1.DISPLAY_PRIORITY as displayOrder
from
	BUNDLE_MEMBER_MAP t1
	inner join BUNDLE_VALUES t4 on t1.BUNDLE_ID = t4.BUNDLE_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join GUI_INDICATOR_REF t7 on t7.table_name = ''BUNDLE_MEMBER_MAP''
		and t7.field_name = lower(''MEMBER_SELECTION_TYPE'')
		and t7.integer_value = t1.MEMBER_SELECTION_TYPE
	left outer join GUI_INDICATOR_VALUES t8 on t8.table_name = t7.table_name
		and t8.field_name = t7.field_name
		and t8.integer_value = t7.integer_value
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join OFFER_REF t9 on t9.OFFER_ID = t1.MEMBER_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join OFFER_VALUES t10 on t10.OFFER_ID = t9.OFFER_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t12 on t12.enumeration_key = lower(''PAYMENT_MODE'')
		and t12.value = t9.PAYMENT_MODE
	left outer join GENERIC_ENUMERATION_VALUES t13 on t13.enumeration_key = t12.enumeration_key
		and t13.value = t12.value
		and t13.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join SERVICE_CATEGORY_VALUES t14 on t14.SERVICE_CATEGORY_ID = t9.SERVICE_CATEGORY_ID
		and t14.LANGUAGE_CODE = t4.LANGUAGE_CODE
		and t14.SERVICE_VERSION_ID = {1}
	left outer join GENERIC_ENUMERATION_REF t15 on t15.enumeration_key = lower(''OFFER_TYPE'')
		and t15.value = t9.OFFER_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t16 on t16.enumeration_key = t15.enumeration_key
		and t16.value = t15.value
		and t16.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
		and t1.MEMBER_TYPE = 2
