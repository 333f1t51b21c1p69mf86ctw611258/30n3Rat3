select
	t1.RESELLER_ID as resellerId,
	t1.RESELLER_LOGIN as resellerLogin,
	t1.MAX_QUANTITY as maxQuantity,
	t1.MAX_LOT_SIZE as maxLotSize,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_SIM_RESELLER_REF t1
	inner join INVS_SIM_RESELLER_VALUES t2 on t1.RESELLER_ID = t2.RESELLER_ID
where
	t2.LANGUAGE_CODE = {2}
