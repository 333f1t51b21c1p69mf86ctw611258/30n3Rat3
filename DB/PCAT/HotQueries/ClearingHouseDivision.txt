select
	t3.DISPLAY_VALUE as clearingHouseIdRef,
	t1.CLEARING_HOUSE_ID as clearingHouseId,
	t5.SERVICE_VERSION_ID as serviceVersionId,
	t5.ISO_CODE as currencyIsoCode,
	t5.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t6.DISPLAY_VALUE as rateCurrencyKey,
	t1.CURRENCY_CODE as currencyCode,
	t8.DISPLAY_VALUE as revRcvCostCenterRef,
	t1.REV_RCV_COST_CTR as revRcvCostCtr,
	t1.DIVISION_NUMBER as divisionNumber
from
	CLEARING_HOUSE_DIVISION t1
	inner join CLEARING_HOUSE_ID_REF t2 on t1.CLEARING_HOUSE_ID = t2.CLEARING_HOUSE_ID
	inner join CLEARING_HOUSE_ID_VALUES t3 on t2.CLEARING_HOUSE_ID = t3.CLEARING_HOUSE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join RATE_CURRENCY_KEY t4 on t1.CURRENCY_CODE = t4.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t5 on t4.CURRENCY_CODE = t5.CURRENCY_CODE
		and t5.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t6 on t4.CURRENCY_CODE = t6.CURRENCY_CODE
		and t6.SERVICE_VERSION_ID = t5.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join REV_RCV_COST_CENTER_REF t7 on t1.REV_RCV_COST_CTR = t7.REV_RCV_COST_CTR
	inner join REV_RCV_COST_CENTER_VALUES t8 on t7.REV_RCV_COST_CTR = t8.REV_RCV_COST_CTR
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
