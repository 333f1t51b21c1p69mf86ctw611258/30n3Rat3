select
	t1.ACTION as action,
	t4.DISPLAY_VALUE as armPluginProcessRef,
	t1.ARM_PLUGIN_ID as armPluginId,
	t6.DISPLAY_VALUE as armPluginProcessRefByArmPlugin,
	t1.PROCESS_ID as processId,
	t2.LANGUAGE_CODE as languageCode
from
	ARM_ACTION_PLUGIN_MAP_REF t1
	inner join ARM_ACTION_PLUGIN_MAP_VALUES t2 on t1.ACTION = t2.ACTION
	inner join ARM_PLUGIN_PROCESS_REF t3 on t1.ARM_PLUGIN_ID = t3.ARM_PLUGIN_ID
		and t1.PROCESS_ID = t3.PROCESS_ID
	inner join ARM_PLUGIN_PROCESS_VALUES t4 on t3.ARM_PLUGIN_ID = t4.ARM_PLUGIN_ID
		and t3.PROCESS_ID = t4.PROCESS_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
	inner join ARM_PLUGIN_PROCESS_REF t5 on t1.ARM_PLUGIN_ID = t5.ARM_PLUGIN_ID
		and t1.PROCESS_ID = t5.PROCESS_ID
	inner join ARM_PLUGIN_PROCESS_VALUES t6 on t5.ARM_PLUGIN_ID = t6.ARM_PLUGIN_ID
		and t5.PROCESS_ID = t6.PROCESS_ID
		and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
