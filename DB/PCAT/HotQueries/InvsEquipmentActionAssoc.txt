select
	t3.DISPLAY_VALUE as invsEquipmentActionRef,
	t1.EQUIPMENT_ACTION_ID as equipmentActionId,
	t5.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t1.WP_PROCESS_ID as wpProcessId
from
	INVS_EQUIPMENT_ACTION_ASSOC t1
	inner join INVS_EQUIPMENT_ACTION_REF t2 on t1.EQUIPMENT_ACTION_ID = t2.EQUIPMENT_ACTION_ID
	inner join INVS_EQUIPMENT_ACTION_VALUES t3 on t2.EQUIPMENT_ACTION_ID = t3.EQUIPMENT_ACTION_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_TYPE_REF t4 on t1.INVENTORY_TYPE_ID = t4.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t5 on t4.INVENTORY_TYPE_ID = t5.INVENTORY_TYPE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
