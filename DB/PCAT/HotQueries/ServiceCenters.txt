select
	t1.SERVICE_CENTER_ID as serviceCenterId,
	t1.SC_NAME as scName,
	t1.SC_CONTACT_FNAME as scContactFname,
	t1.SC_CONTACT_MINIT as scContactMinit,
	t1.SC_CONTACT_LNAME as scContactLname,
	t1.SC_ADDRESS1 as scAddress1,
	t1.SC_ADDRESS2 as scAddress2,
	t1.SC_ADDRESS3 as scAddress3,
	t1.SC_CITY as scCity,
	t1.SC_STATE as scState,
	t1.SC_ZIP as scZip,
	t3.DISPLAY_VALUE as countryCodeRef,
	t1.SC_COUNTRY_CODE as scCountryCode,
	t1.SC_COUNTY as scCounty,
	t1.SC_PHONE as scPhone,
	t1.TELE_MSG as teleMsg,
	t1.FAX_MSG as faxMsg,
	t1.EMAIL_MSG as emailMsg,
	t1.CHG_DATE as chgDate,
	t1.CHG_WHO as chgWho,
	t1.PRINTER_DEVICE_ID as printerDeviceId
from
	SERVICE_CENTERS t1
	left outer join COUNTRY_CODE_REF t2 on t1.SC_COUNTRY_CODE = t2.COUNTRY_CODE
	left outer join COUNTRY_CODE_VALUES t3 on t2.COUNTRY_CODE = t3.COUNTRY_CODE
		and t3.LANGUAGE_CODE = {2}
