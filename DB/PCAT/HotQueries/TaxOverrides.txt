select
	t1.TAX_OVERRIDE_ID as taxOverrideId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as chargeEltType,
	t1.CHARGE_ELT_ID as chargeEltId,
	t1.TAX_PKG_INST_ID as taxPkgInstId,
        CASE
            WHEN t1.TAX_PKG_INST_ID=0 THEN {3}
            ELSE t6.DISPLAY_VALUE
        END as taxPkgInstIdValues,
        CASE
            WHEN t1.TAX_TYPE_CODE=0 THEN {3}
            ELSE t14.DISPLAY_VALUE
        END as taxTypeCodeValues,
	t1.TAX_TYPE_CODE as taxTypeCode,
	t1.COUNTRY_CODE as countryCode,
        t8.DISPLAY_VALUE as countryCodeRef,
	t1.FRANCHISE_CODE as franchiseCode,
        CASE
            WHEN t1.FRANCHISE_CODE=0 THEN {3}
            ELSE t11.DISPLAY_VALUE
        END as franchiseCodeValues,
	t1.STATE_CODE as stateCode,
	t1.COUNTY_CODE as countyCode,
	t1.CITY_CODE as cityCode,
	t1.PRIORITY as priority,
	t1.FEDERAL_RATE as federalRate,
	t1.STATE_RATE as stateRate,
	t1.COUNTY_RATE as countyRate,
	t1.CITY_RATE as cityRate,
	t1.OTHER_RATE as otherRate
from
	TAX_OVERRIDES t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''CHARGE_ELT_TYPE'')
		and t2.value = t1.CHARGE_ELT_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
        left outer join TAX_PKG_INST_ID_KEY t4 on t4.TAX_PKG_INST_ID = t1.TAX_PKG_INST_ID
        left outer join TAX_PKG_INST_ID_REF t5 on t5.TAX_PKG_INST_ID = t1.TAX_PKG_INST_ID
                and t5.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
        left outer join TAX_PKG_INST_ID_VALUES t6 on t6.TAX_PKG_INST_ID = t1.TAX_PKG_INST_ID
                and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
                and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
        left outer join COUNTRY_CODE_REF t7 on t7.COUNTRY_CODE = t1.COUNTRY_CODE
        left outer join COUNTRY_CODE_VALUES t8 on t8.COUNTRY_CODE = t1.COUNTRY_CODE
                and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
        left outer join FRANCHISE_CODE_KEY t9 on t9.FRANCHISE_CODE = t1.FRANCHISE_CODE
        left outer join FRANCHISE_CODE_REF t10 on t10.FRANCHISE_CODE = t1.FRANCHISE_CODE
                and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
        left outer join FRANCHISE_CODE_VALUES t11 on t11.FRANCHISE_CODE = t1.FRANCHISE_CODE
                and t11.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
                and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
        left outer join TAX_TYPE_CODE_KEY t12 on t12.TAX_TYPE_CODE = t1.TAX_TYPE_CODE
        left outer join TAX_TYPE_CODE_REF t13 on t13.TAX_TYPE_CODE = t1.TAX_TYPE_CODE
                and t13.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
        left outer join TAX_TYPE_CODE_VALUES t14 on t14.TAX_TYPE_CODE = t1.TAX_TYPE_CODE
                and t14.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
                and t14.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
