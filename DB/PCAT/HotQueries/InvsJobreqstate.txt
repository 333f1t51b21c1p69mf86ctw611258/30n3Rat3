select
	t1.JOBREQSTATE_ID as jobreqstateId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_JOBREQSTATE_REF t1
	inner join INVS_JOBREQSTATE_VALUES t2 on t1.JOBREQSTATE_ID = t2.JOBREQSTATE_ID
where
	t2.LANGUAGE_CODE = {2}
