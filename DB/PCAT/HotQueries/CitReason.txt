select
	t1.CIT_REASON_ID as citReasonId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CIT_REASON_REF t1
	inner join CIT_REASON_VALUES t2 on t1.CIT_REASON_ID = t2.CIT_REASON_ID
where
	t2.LANGUAGE_CODE = {2}
