select
	t1.CLEARING_HOUSE_ID as clearingHouseId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CLEARING_HOUSE_ID_REF t1
	inner join CLEARING_HOUSE_ID_VALUES t2 on t1.CLEARING_HOUSE_ID = t2.CLEARING_HOUSE_ID
where
	t2.LANGUAGE_CODE = {2}
