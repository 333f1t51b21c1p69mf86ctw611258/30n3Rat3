select
	t1.WORKORDER_ITEM_TYPE_DEF as workorderItemTypeDef,
	t3.DISPLAY_VALUE as ordServiceOrderTypeRef,
	t1.SERVICE_ORDER_TYPE_ID as serviceOrderTypeId,
	t5.DISPLAY_VALUE as memberType,
	t1.MEMBER_ID as memberId,
	t7.DISPLAY_VALUE as ordItemActionRef,
	t1.ITEM_ACTION_ID as itemActionId,
	t1.GROUPING_LEVEL as groupingLevel,
	t1.WORK_UNITS as workUnits,
	t1.IS_OPTIONAL as isOptional,
	t1.IS_CUSTOMER_NEGOTIATED as isCustomerNegotiated,
	t9.RESELLER_VERSION_ID as resellerVersionId,
	t10.DISPLAY_VALUE as nrcTermKey,
	t1.NRC_TERM_ID as nrcTermId
from
	WO_WORKORDER_ITEM_TYPE_DEF t1
	inner join ORD_SERVICE_ORDER_TYPE_REF t2 on t1.SERVICE_ORDER_TYPE_ID = t2.SERVICE_ORDER_TYPE_ID
	inner join ORD_SERVICE_ORDER_TYPE_VALUES t3 on t2.SERVICE_ORDER_TYPE_ID = t3.SERVICE_ORDER_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join ORD_ITEM_MEMBER_TYPE_REF t4 on t1.MEMBER_TYPE = t4.ITEM_MEMBER_TYPE_ID		
	inner join ORD_ITEM_MEMBER_TYPE_VALUES t5 on t4.ITEM_MEMBER_TYPE_ID = t5.ITEM_MEMBER_TYPE_ID		
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join ORD_ITEM_ACTION_REF t6 on t1.ITEM_ACTION_ID = t6.ITEM_ACTION_ID
	inner join ORD_ITEM_ACTION_VALUES t7 on t6.ITEM_ACTION_ID = t7.ITEM_ACTION_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join NRC_TERM_KEY t8 on t1.NRC_TERM_ID = t8.NRC_TERM_ID
	inner join NRC_TERM_REF t9 on t8.NRC_TERM_ID = t9.NRC_TERM_ID
		and t9.RESELLER_VERSION_ID = {0}
	inner join NRC_TERM_VALUES t10 on t8.NRC_TERM_ID = t10.NRC_TERM_ID
		and t10.RESELLER_VERSION_ID = t9.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t3.LANGUAGE_CODE
