select
	t1.FILE_CLASS as fileClass,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	USAGE_FILE_CLASS_REF t1
	inner join USAGE_FILE_CLASS_VALUES t2 on t1.FILE_CLASS = t2.FILE_CLASS
where
	t2.LANGUAGE_CODE = {2}
