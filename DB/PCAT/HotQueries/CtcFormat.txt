select
	t1.FORMAT_ID as formatId,
	t4.DISPLAY_VALUE as ctcFormatTypeRef,
	t1.FORMAT_TYPE_ID as formatTypeId,
	t1.P_FORMAT_ID as pFormatId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_FORMAT_REF t1
	inner join CTC_FORMAT_VALUES t2 on t1.FORMAT_ID = t2.FORMAT_ID
	inner join CTC_FORMAT_TYPE_REF t3 on t1.FORMAT_TYPE_ID = t3.FORMAT_TYPE_ID
	inner join CTC_FORMAT_TYPE_VALUES t4 on t3.FORMAT_TYPE_ID = t4.FORMAT_TYPE_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
