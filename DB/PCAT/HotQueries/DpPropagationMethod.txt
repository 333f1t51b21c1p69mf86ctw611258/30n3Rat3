select
	t1.PROPAGATION_METHOD as propagationMethod,
	t1.IS_BLOCKING as isBlocking,
	t1.CLASS_NAME as className,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t4.DISPLAY_VALUE as targetDbType,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	DP_PROPAGATION_METHOD_REF t1
	inner join DP_PROPAGATION_METHOD_VALUES t2 on t1.PROPAGATION_METHOD = t2.PROPAGATION_METHOD
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''TARGET_DB_TYPE'')
		and t3.value = t1.TARGET_DB_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
