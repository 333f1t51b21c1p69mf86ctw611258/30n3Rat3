select
	t3.DISPLAY_VALUE as billFmtOptRef,
	t1.BILL_FMT_OPT as billFmtOpt,
	t5.DISPLAY_VALUE as billFmtSectionExclRef,
	t1.SECTION_EXCLUDE as sectionExclude
from
	BILL_SECTIONS t1
	inner join BILL_FMT_OPT_REF t2 on t1.BILL_FMT_OPT = t2.BILL_FMT_OPT
	inner join BILL_FMT_OPT_VALUES t3 on t2.BILL_FMT_OPT = t3.BILL_FMT_OPT
		and t3.LANGUAGE_CODE = {2}
	inner join BILL_FMT_SECTION_EXCL_REF t4 on t1.SECTION_EXCLUDE = t4.SECTION_EXCLUDE
	inner join BILL_FMT_SECTION_EXCL_VALUES t5 on t4.SECTION_EXCLUDE = t5.SECTION_EXCLUDE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
