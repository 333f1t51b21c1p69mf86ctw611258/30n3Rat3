select
	t1.SO_WF_MAP_ID as soWfMapId,
	t3.DISPLAY_VALUE as ordServiceOrderTypeRef,
	t1.SERVICE_ORDER_TYPE_ID as serviceOrderTypeId,
	t1.IS_SERVICE_LEVEL as isServiceLevel,
	t5.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as offerKey,
	t1.PRIMARY_OFFER_ID as primaryOfferId,
	t1.WP_PROCESS_ID as wpProcessId,
	t1.DURATION_UNITS as durationUnits,
	t1.ALIGN_START as alignStart,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.PRE_PIVOT_DURATION as prePivotDuration,
	t1.POST_PIVOT_DURATION as postPivotDuration
from
	ORD_SERVICE_ORDER_WORKFLOW_MAP t1
	inner join ORD_SERVICE_ORDER_TYPE_REF t2 on t1.SERVICE_ORDER_TYPE_ID = t2.SERVICE_ORDER_TYPE_ID
	inner join ORD_SERVICE_ORDER_TYPE_VALUES t3 on t2.SERVICE_ORDER_TYPE_ID = t3.SERVICE_ORDER_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	left outer join OFFER_KEY t4 on t1.PRIMARY_OFFER_ID = t4.OFFER_ID
	left outer join OFFER_REF t5 on t4.OFFER_ID = t5.OFFER_ID
		and t5.RESELLER_VERSION_ID = {0}
	left outer join OFFER_VALUES t6 on t4.OFFER_ID = t6.OFFER_ID
		and t6.RESELLER_VERSION_ID = {0}
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
