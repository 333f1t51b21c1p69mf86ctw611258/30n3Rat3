select
	t1.WORKORDER_TYPE_DEF as workorderTypeDef,
	t3.DISPLAY_VALUE as woWorkorderTypeRef,
	t1.WORKORDER_TYPE as workorderType,
	t5.DISPLAY_VALUE as ordServiceOrderTypeRef,
	t1.SERVICE_ORDER_TYPE_ID as serviceOrderTypeId,
	t1.PRIMARY_OFFER_ID as primaryOfferId,
	t1.SKILL_CODE as skillCode,
	t1.IS_SERVICE_CALL as isServiceCall,
	t7.DISPLAY_VALUE as ttSubTypeRef,
	t1.TT_SUB_TYPE_ID as ttSubTypeId
from
	WO_WORKORDER_TYPE_DEF t1
	inner join WO_WORKORDER_TYPE_REF t2 on t1.WORKORDER_TYPE = t2.WORKORDER_TYPE
	inner join WO_WORKORDER_TYPE_VALUES t3 on t2.WORKORDER_TYPE = t3.WORKORDER_TYPE
		and t3.LANGUAGE_CODE = {2}
	inner join ORD_SERVICE_ORDER_TYPE_REF t4 on t1.SERVICE_ORDER_TYPE_ID = t4.SERVICE_ORDER_TYPE_ID
	inner join ORD_SERVICE_ORDER_TYPE_VALUES t5 on t4.SERVICE_ORDER_TYPE_ID = t5.SERVICE_ORDER_TYPE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join TT_SUB_TYPE_REF t6 on t1.TT_SUB_TYPE_ID = t6.TT_SUB_TYPE_ID
	left outer join TT_SUB_TYPE_VALUES t7 on t6.TT_SUB_TYPE_ID = t7.TT_SUB_TYPE_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
