select
	t1.NOTE_TEMPLATE_ID as noteTemplateId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.NOTE_TEMPLATE_CODE as noteTemplateCode,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	NOTE_TEMPLATE_REF t1
	inner join NOTE_TEMPLATE_VALUES t2 on t1.NOTE_TEMPLATE_ID = t2.NOTE_TEMPLATE_ID
where
	t2.LANGUAGE_CODE = {2}
