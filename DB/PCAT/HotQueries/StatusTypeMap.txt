select
	t3.DISPLAY_VALUE as statusTypeRef,
	t1.STATUS_TYPE_ID as statusTypeId,
	t5.DISPLAY_VALUE as statusRef,
	t1.STATUS_ID as statusId
from
	STATUS_TYPE_MAP t1
	inner join STATUS_TYPE_REF t2 on t1.STATUS_TYPE_ID = t2.STATUS_TYPE_ID
	inner join STATUS_TYPE_VALUES t3 on t2.STATUS_TYPE_ID = t3.STATUS_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join STATUS_REF t4 on t1.STATUS_ID = t4.STATUS_ID
	inner join STATUS_VALUES t5 on t4.STATUS_ID = t5.STATUS_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
