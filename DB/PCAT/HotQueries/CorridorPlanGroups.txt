select
	t1.CORRIDOR_PLAN_GROUP_ID as corridorPlanGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as corridorPlanIdKey,
	t1.CORRIDOR_PLAN_ID as corridorPlanId,
	t1.POINT_ORIGIN as pointOrigin,
	t6.SERVICE_VERSION_ID as serviceVersionId,
	t7.DISPLAY_VALUE as pointCategoryKey,
	t1.POINT_CATEGORY as pointCategory,
	t1.ORIGIN_IS_XACT as originIsXact,
	t9.DISPLAY_VALUE as countryCodeRef,
	t1.COUNTRY_CODE_ORIGIN as countryCodeOrigin,
	t1.POINT_TARGET as pointTarget,
	t1.TARGET_IS_XACT as targetIsXact,
	t11.DISPLAY_VALUE as countryCodeRefByCountryCodeTar,
	t1.COUNTRY_CODE_TARGET as countryCodeTarget,
	t1.ACCESS_REGION_ORIGIN as accessRegionOrigin,
	t1.ACCESS_REGION_TARGET as accessRegionTarget
from
	CORRIDOR_PLAN_GROUPS t1
	inner join CORRIDOR_PLAN_ID_KEY t2 on t1.CORRIDOR_PLAN_ID = t2.CORRIDOR_PLAN_ID
	inner join CORRIDOR_PLAN_ID_REF t3 on t2.CORRIDOR_PLAN_ID = t3.CORRIDOR_PLAN_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join CORRIDOR_PLAN_ID_VALUES t4 on t2.CORRIDOR_PLAN_ID = t4.CORRIDOR_PLAN_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join POINT_CATEGORY_KEY t5 on t1.POINT_CATEGORY = t5.POINT_CATEGORY
	inner join POINT_CATEGORY_REF t6 on t5.POINT_CATEGORY = t6.POINT_CATEGORY
		and t6.SERVICE_VERSION_ID = {1}
	inner join POINT_CATEGORY_VALUES t7 on t5.POINT_CATEGORY = t7.POINT_CATEGORY
		and t7.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join COUNTRY_CODE_REF t8 on t1.COUNTRY_CODE_ORIGIN = t8.COUNTRY_CODE
	left outer join COUNTRY_CODE_VALUES t9 on t8.COUNTRY_CODE = t9.COUNTRY_CODE
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join COUNTRY_CODE_REF t10 on t1.COUNTRY_CODE_TARGET = t10.COUNTRY_CODE
	left outer join COUNTRY_CODE_VALUES t11 on t10.COUNTRY_CODE = t11.COUNTRY_CODE
		and t11.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
