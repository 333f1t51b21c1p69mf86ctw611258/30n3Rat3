select
	t1.DENIAL_REASON as denialReason,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	DENIAL_REASON_REF t1
	inner join DENIAL_REASON_VALUES t2 on t1.DENIAL_REASON = t2.DENIAL_REASON
where
	t2.LANGUAGE_CODE = {2}
