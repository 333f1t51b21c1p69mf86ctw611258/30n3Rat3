select
	t1.SUB_TYPE_ID as subTypeId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.INSTANCE_FLAG as instanceFlag,
	t2.SUB_TYPE_CODE as subTypeCode,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	SUB_TYPE_ID_KEY t1
	inner join SUB_TYPE_ID_REF t2 on t1.SUB_TYPE_ID = t2.SUB_TYPE_ID
	inner join SUB_TYPE_ID_VALUES t3 on t2.SUB_TYPE_ID = t3.SUB_TYPE_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
