select
	t3.DISPLAY_VALUE as invsLineRef,
	t1.INVENTORY_LINE_ID as inventoryLineId,
	t5.DISPLAY_VALUE as disconnectReasonRef,
	t1.DISCONNECT_REASON as disconnectReason,
	t1.BLACKLIST_REASON_ID as blacklistReasonId,
	t1.IS_TO_BE_PORTED as isToBePorted
from
	INVS_DISCONNECT_BEHAVIOR t1
	inner join INVS_LINE_REF t2 on t1.INVENTORY_LINE_ID = t2.INVENTORY_LINE_ID
	inner join INVS_LINE_VALUES t3 on t2.INVENTORY_LINE_ID = t3.INVENTORY_LINE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join DISCONNECT_REASON_REF t4 on t1.DISCONNECT_REASON = t4.DISCONNECT_REASON
	inner join DISCONNECT_REASON_VALUES t5 on t4.DISCONNECT_REASON = t5.DISCONNECT_REASON
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
