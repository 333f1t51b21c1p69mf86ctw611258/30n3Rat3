select
	t3.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t5.DISPLAY_VALUE as invsSubjMatterRatingRef,
	t1.SUBJECT_MATTER_RATING as subjectMatterRating,
	t1.IS_TYPE_DEFAULT as isTypeDefault
from
	INVS_SMR_INV_ASSOC t1
	inner join INVS_TYPE_REF t2 on t1.INVENTORY_TYPE_ID = t2.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t3 on t2.INVENTORY_TYPE_ID = t3.INVENTORY_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_SUBJ_MATTER_RATING_REF t4 on t1.SUBJECT_MATTER_RATING = t4.SUBJECT_MATTER_RATING
	inner join INVS_SUBJ_MATTER_RATING_VALUES t5 on t4.SUBJECT_MATTER_RATING = t5.SUBJECT_MATTER_RATING
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
