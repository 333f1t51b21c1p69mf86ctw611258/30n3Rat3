select
	t1.MARKETING_CATEGORY_ID as marketingCategoryId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as marketingCategoryKey,
	t2.PARENT_MARKETING_CATEGORY_ID as parentMarketingCategoryId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	MARKETING_CATEGORY_KEY t1
	inner join MARKETING_CATEGORY_REF t2 on t1.MARKETING_CATEGORY_ID = t2.MARKETING_CATEGORY_ID
	inner join MARKETING_CATEGORY_VALUES t3 on t2.MARKETING_CATEGORY_ID = t3.MARKETING_CATEGORY_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join MARKETING_CATEGORY_KEY t4 on t2.PARENT_MARKETING_CATEGORY_ID = t4.MARKETING_CATEGORY_ID
	left outer join MARKETING_CATEGORY_REF t5 on t4.MARKETING_CATEGORY_ID = t5.MARKETING_CATEGORY_ID
		and t5.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join MARKETING_CATEGORY_VALUES t6 on t4.MARKETING_CATEGORY_ID = t6.MARKETING_CATEGORY_ID
		and t6.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
