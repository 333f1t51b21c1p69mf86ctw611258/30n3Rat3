select
	t1.BILLING_FREQUENCY as billingFrequency,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	BILLING_FREQUENCY_REF t1
	inner join BILLING_FREQUENCY_VALUES t2 on t1.BILLING_FREQUENCY = t2.BILLING_FREQUENCY
where
	t2.LANGUAGE_CODE = {2}
