select
	t1.ATTRIBUTE_ID as attributeId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.DATA_TYPE as dataType,
	t6.DISPLAY_VALUE as segmentationKeyIdKey,
	t2.SEGMENTATION_KEY_ID as segmentationKeyId,
	t2.ATTRIBUTE_TYPE as attributeType,
	t2.ATTRIBUTE_SOURCE as attributeSource,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	SEGMENTATION_KEY_ATTR_KEY t1
	inner join SEGMENTATION_KEY_ATTR_REF t2 on t1.ATTRIBUTE_ID = t2.ATTRIBUTE_ID
	inner join SEGMENTATION_KEY_ATTR_VALUES t3 on t2.ATTRIBUTE_ID = t3.ATTRIBUTE_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join SEGMENTATION_KEY_ID_KEY t4 on t2.SEGMENTATION_KEY_ID = t4.SEGMENTATION_KEY_ID
	inner join SEGMENTATION_KEY_ID_REF t5 on t4.SEGMENTATION_KEY_ID = t5.SEGMENTATION_KEY_ID
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join SEGMENTATION_KEY_ID_VALUES t6 on t4.SEGMENTATION_KEY_ID = t6.SEGMENTATION_KEY_ID
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
