select
	t1.OPERATOR_ID as operatorId,
	t1.OPERATOR_TYPE as operatorType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.TAX_GEOCODE as taxGeocode,
	t1.SERVER_ID as serverId,
	t1.MNC_LENGTH as mncLength,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ROAMING_OPERATOR_REF t1
	inner join ROAMING_OPERATOR_VALUES t2 on t1.OPERATOR_ID = t2.OPERATOR_ID
where
	t2.LANGUAGE_CODE = {2}
