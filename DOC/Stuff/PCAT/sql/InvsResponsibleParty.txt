select
	t1.RESPONSIBLE_PARTY_ID as responsiblePartyId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_RESPONSIBLE_PARTY_REF t1
	inner join INVS_RESPONSIBLE_PARTY_VALUES t2 on t1.RESPONSIBLE_PARTY_ID = t2.RESPONSIBLE_PARTY_ID
where
	t2.LANGUAGE_CODE = {2}
