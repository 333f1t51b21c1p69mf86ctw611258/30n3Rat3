select
	t3.DISPLAY_VALUE as paymentMerchantIdRef,
	t1.PAYMENT_MERCHANT_ID as paymentMerchantId,
	t1.PAY_METHOD as payMethod,
	t5.DISPLAY_VALUE as cardTypeCodeRef,
	t1.CARD_TYPE_CODE as cardTypeCode,
	t1.IS_DEFAULT_MERCHANT_ID as isDefaultMerchantId
from
	PAYMENT_MERCHANT_ID_ASSIGN t1
	inner join PAYMENT_MERCHANT_ID_REF t2 on t1.PAYMENT_MERCHANT_ID = t2.PAYMENT_MERCHANT_ID
	inner join PAYMENT_MERCHANT_ID_VALUES t3 on t2.PAYMENT_MERCHANT_ID = t3.PAYMENT_MERCHANT_ID
		and t3.LANGUAGE_CODE = {2}
	inner join CARD_TYPE_CODE_REF t4 on t1.CARD_TYPE_CODE = t4.CARD_TYPE_CODE
	inner join CARD_TYPE_CODE_VALUES t5 on t4.CARD_TYPE_CODE = t5.CARD_TYPE_CODE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
