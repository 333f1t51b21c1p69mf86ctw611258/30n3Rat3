select
	t1.CLIENT_ACTION_ID as clientActionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.ACTION_CATEGORY as actionCategory,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CAS_CLIENT_ACTION_REF t1
	inner join CAS_CLIENT_ACTION_VALUES t2 on t1.CLIENT_ACTION_ID = t2.CLIENT_ACTION_ID
where
	t2.LANGUAGE_CODE = {2}
