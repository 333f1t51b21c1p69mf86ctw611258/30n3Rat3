select
	t1.OSA_URL_ID as osaUrlId,
	t1.URL as url,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t3.DISPLAY_VALUE as serviceType,
	t1.CALLBACK_URL as callbackUrl
from
	OSA_URL t1
	left outer join GUI_INDICATOR_REF t2 on t2.table_name = ''OSA_URL''
		and t2.field_name = lower(''SERVICE_TYPE'')
		and t2.integer_value = t1.SERVICE_TYPE
	left outer join GUI_INDICATOR_VALUES t3 on t3.table_name = t2.table_name
		and t3.field_name = t2.field_name
		and t3.integer_value = t2.integer_value
		and t3.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
