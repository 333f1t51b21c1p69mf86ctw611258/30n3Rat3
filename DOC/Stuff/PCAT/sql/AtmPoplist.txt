select
	t1.TABLE_NAME as tableName,
	t1.FIELD_NAME as fieldName,
	t1.INT_VALUE as intValue,
	t1.CHAR_VALUE as charValue,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ATM_POPLIST_REF t1
	inner join ATM_POPLIST_VALUES t2 on t1.TABLE_NAME = t2.TABLE_NAME
		and t1.FIELD_NAME = t2.FIELD_NAME
		and (t1.INT_VALUE = t2.INT_VALUE
		or NVL(t1.CHAR_VALUE,'''') = NVL(t2.CHAR_VALUE,''''))
where
	t2.LANGUAGE_CODE = {2}
