select
	t1.RESELLER_VERSION_CHG_ID as resellerVersionChgId,
	t1.PREV_RESELLER_VERSION_CHG_ID as prevResellerVersionChgId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.TABLE_NAME as tableName,
	t1.CHANGE_ID1 as changeId1,
	t1.CHANGE_ID2 as changeId2,
	t1.MAJOR_PC_VERSION_NUM_START as majorPcVersionNumStart,
	t1.MINOR_PC_VERSION_NUM as minorPcVersionNum,
	t1.LOCK_USER as lockUser,
	t1.CHG_WHO as chgWho,
	t1.CHG_DT as chgDt,
	t1.MAJOR_PC_VERSION_NUM_END as majorPcVersionNumEnd
from
	RESELLER_VERSION_CHG_LOG t1
	left outer join RESELLER_VERSION_CHG_LOG t2 on t1.PREV_RESELLER_VERSION_CHG_ID = t2.RESELLER_VERSION_CHG_ID
where
	t1.RESELLER_VERSION_ID = {0}
