select
	t3.DISPLAY_VALUE as lbxSourceIdRef,
	t1.SOURCE_ID as sourceId,
	t1.EXT_CATEGORY as extCategory,
	t1.BMF_TRANS_TYPE as bmfTransType
from
	LBX_PAYMENT_TYPES t1
	inner join LBX_SOURCE_ID_REF t2 on t1.SOURCE_ID = t2.SOURCE_ID
	inner join LBX_SOURCE_ID_VALUES t3 on t2.SOURCE_ID = t3.SOURCE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join BMF_TRANS_DESCR t4 on t1.BMF_TRANS_TYPE = t4.BMF_TRANS_TYPE
