select
	t1.GEO_AREA_MEMBER_ID as geoAreaMemberId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as geoAreaKey,
	t1.GEO_AREA_ID as geoAreaId,
	t1.GEO_AREA_TYPE_ID as geoAreaTypeId,
	t1.GEO_AREA_RULE as geoAreaRule
from
	GEO_AREA_MEMBER t1
	inner join GEO_AREA_KEY t2 on t1.GEO_AREA_ID = t2.GEO_AREA_ID
	inner join GEO_AREA_REF t3 on t2.GEO_AREA_ID = t3.GEO_AREA_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join GEO_AREA_VALUES t4 on t2.GEO_AREA_ID = t4.GEO_AREA_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
