select
	t1.CIT_AUTO_NOTE_ACTION_ID as citAutoNoteActionId,
	t4.DISPLAY_VALUE as noteTemplateRef,
	t1.NOTE_TEMPLATE_ID as noteTemplateId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.IS_ENABLED as isEnabled,
	t1.DEFAULT_PERMANENCE as defaultPermanence,
	t6.DISPLAY_VALUE as noteTypeRef,
	t1.DEFAULT_NOTE_TYPE_ID as defaultNoteTypeId,
	t8.DISPLAY_VALUE as noteReasonRef,
	t1.DEFAULT_NOTE_REASON_ID as defaultNoteReasonId,
	t10.DISPLAY_VALUE as citReasonRef,
	t1.DEFAULT_CIT_REASON_ID as defaultCitReasonId,
	t12.DISPLAY_VALUE as citChannelRef,
	t1.DEFAULT_CIT_CHANNEL_ID as defaultCitChannelId,
	t14.DISPLAY_VALUE as citImportanceRef,
	t1.DEFAULT_CIT_IMPORTANCE_ID as defaultCitImportanceId,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CIT_AUTO_NOTE_ACTION_REF t1
	inner join CIT_AUTO_NOTE_ACTION_VALUES t2 on t1.CIT_AUTO_NOTE_ACTION_ID = t2.CIT_AUTO_NOTE_ACTION_ID
	inner join NOTE_TEMPLATE_REF t3 on t1.NOTE_TEMPLATE_ID = t3.NOTE_TEMPLATE_ID
	inner join NOTE_TEMPLATE_VALUES t4 on t3.NOTE_TEMPLATE_ID = t4.NOTE_TEMPLATE_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join NOTE_TYPE_REF t5 on t1.DEFAULT_NOTE_TYPE_ID = t5.NOTE_TYPE_ID
	left outer join NOTE_TYPE_VALUES t6 on t5.NOTE_TYPE_ID = t6.NOTE_TYPE_ID
		and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join NOTE_REASON_REF t7 on t1.DEFAULT_NOTE_REASON_ID = t7.NOTE_REASON_ID
	left outer join NOTE_REASON_VALUES t8 on t7.NOTE_REASON_ID = t8.NOTE_REASON_ID
		and t8.LANGUAGE_CODE = t2.LANGUAGE_CODE
	inner join CIT_REASON_REF t9 on t1.DEFAULT_CIT_REASON_ID = t9.CIT_REASON_ID
	inner join CIT_REASON_VALUES t10 on t9.CIT_REASON_ID = t10.CIT_REASON_ID
		and t10.LANGUAGE_CODE = t2.LANGUAGE_CODE
	inner join CIT_CHANNEL_REF t11 on t1.DEFAULT_CIT_CHANNEL_ID = t11.CIT_CHANNEL_ID
	inner join CIT_CHANNEL_VALUES t12 on t11.CIT_CHANNEL_ID = t12.CIT_CHANNEL_ID
		and t12.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join CIT_IMPORTANCE_REF t13 on t1.DEFAULT_CIT_IMPORTANCE_ID = t13.CIT_IMPORTANCE_ID
	left outer join CIT_IMPORTANCE_VALUES t14 on t13.CIT_IMPORTANCE_ID = t14.CIT_IMPORTANCE_ID
		and t14.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
