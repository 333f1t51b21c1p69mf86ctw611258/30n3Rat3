select
        t1.TAX_CHARGE_ELEMENTS_ID as taxChargeElementsId,
	t3.DISPLAY_VALUE as chargeEltType,
	t1.CHARGE_ELT_ID as chargeEltId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.TAXABLE_BUS_IND as taxableBusInd,
	t1.COMPLIANCE_TRANS_CODE as complianceTransCode,
	t1.SERVICE_GROUP as serviceGroup,
	t1.FREQUENCY as frequency,
	t1.TYPE_OF_CHARGE as typeOfCharge,
	t1.TELETAX_CALC_IND as teletaxCalcInd,
	t1.DIVISION as division,
	t1.PRODUCT_CODE as productCode,
	t1.PRODUCT_GROUP as productGroup,
	t1.TERMS_CODE as termsCode,
	t1.USAGE_OF_PRODUCT as usageOfProduct
from
	TAX_CHARGE_ELEMENTS t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''CHARGE_ELT_TYPE'')
		and t2.value = t1.CHARGE_ELT_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
