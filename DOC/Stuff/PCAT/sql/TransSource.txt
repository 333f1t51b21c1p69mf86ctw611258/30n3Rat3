select
	t1.TRANS_SOURCE as transSource,
	t1.SOURCE_ID as sourceId,
	t1.MODULE as module,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	TRANS_SOURCE_REF t1
	inner join TRANS_SOURCE_VALUES t2 on t1.TRANS_SOURCE = t2.TRANS_SOURCE
where
	t2.LANGUAGE_CODE = {2}
