select
	t1.OFFER_ID as offerId,
	t1.FA_LIST_TEMPLATE_ID as faListTemplateId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as name,
        t4.DESCRIPTION as description,
        t3.USE_LOCATION_GROUP as useLocationGroup,
	t6.DISPLAY_VALUE as listType,
        ''{3}'' as templateType
from
	OFFER_FA_LIST_TEMPLATE_MAP t1
	inner join FA_LIST_TEMPLATE_KEY t2 on t1.FA_LIST_TEMPLATE_ID = t2.FA_LIST_TEMPLATE_ID
	inner join FA_LIST_TEMPLATE_REF t3 on t1.FA_LIST_TEMPLATE_ID = t3.FA_LIST_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join FA_LIST_TEMPLATE_VALUES t4 on t1.FA_LIST_TEMPLATE_ID = t4.FA_LIST_TEMPLATE_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
        left outer join GUI_INDICATOR_REF t5 on t5.FIELD_NAME = lower(''LIST_TYPE'')
                and t5.TABLE_NAME = ''FA_LIST_TEMPLATE_REF''
		and t5.INTEGER_VALUE = t3.LIST_TYPE
	left outer join GUI_INDICATOR_VALUES t6 on t6.FIELD_NAME = t5.FIELD_NAME
                and t6.TABLE_NAME = t5.TABLE_NAME
		and t6.INTEGER_VALUE = t5.INTEGER_VALUE
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
