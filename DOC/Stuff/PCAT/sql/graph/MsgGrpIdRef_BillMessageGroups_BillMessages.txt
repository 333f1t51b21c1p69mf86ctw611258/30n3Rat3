select
        t1.MSG_GRP_ID as msgGrpId,
        t1.MESSAGE_ID as messageId,
	t2.LINE_NUM as lineNum,
        t2.MESSAGE_TEXT as billMsgText,
	t4.DISPLAY_VALUE as name
from
	BILL_MESSAGE_GROUPS t1
	inner join BILL_MESSAGES t2 on t1.MESSAGE_ID = t2.MESSAGE_ID
	inner join MSG_GRP_ID_REF t3 on t1.MSG_GRP_ID = t3.MSG_GRP_ID
	inner join MSG_GRP_ID_VALUES t4 on t1.MSG_GRP_ID = t4.MSG_GRP_ID
		and t4.LANGUAGE_CODE = {2}
