select
	t1.OFFER_ID as offerId,
	t1.EXTERNAL_ID_TYPE as externalIdType,
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as name,
	t4.DESCRIPTION as description
from
	OFFER_EXT_ID_COREQ t1
	inner join EXTERNAL_ID_TYPE_KEY t2 on t1.EXTERNAL_ID_TYPE = t2.EXTERNAL_ID_TYPE
	inner join EXTERNAL_ID_TYPE_REF t3 on t1.EXTERNAL_ID_TYPE = t3.EXTERNAL_ID_TYPE
		and t3.SERVICE_VERSION_ID = {1}
	inner join EXTERNAL_ID_TYPE_VALUES t4 on t1.EXTERNAL_ID_TYPE = t4.EXTERNAL_ID_TYPE
		and t4.LANGUAGE_CODE = {2}
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
        and t1.AUTO_GENERATE=0
        and t1.IS_AUT_GENERATED=0
