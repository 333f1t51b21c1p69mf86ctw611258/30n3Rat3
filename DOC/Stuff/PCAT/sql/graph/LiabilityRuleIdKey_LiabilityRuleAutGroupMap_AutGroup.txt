select
	t1.LIABILITY_RULE_ID as liabilityRuleId,
	t1.AUT_GROUP_ID as autGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description
from
	LIABILITY_RULE_AUT_GROUP_MAP t1
	inner join AUT_GROUP_REF t2 on t1.AUT_GROUP_ID = t2.AUT_GROUP_ID
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join AUT_GROUP_VALUES t3 on t1.AUT_GROUP_ID = t3.AUT_GROUP_ID
		and t3.LANGUAGE_CODE = {2}
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
