select
	t1.INITIAL_AUT_ID as initialAutId,
	t1.EXTERNAL_ID_TYPE as externalIdType,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t7.DISPLAY_VALUE as name,
	t7.DESCRIPTION as description
from
	AUT_EXT_ID_TYPE_MAP t1
	inner join EXTERNAL_ID_TYPE_VALUES t7 on t1.EXTERNAL_ID_TYPE = t7.EXTERNAL_ID_TYPE
		and t7.LANGUAGE_CODE = {2}
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
where
	t1.SERVICE_VERSION_ID = {1}
