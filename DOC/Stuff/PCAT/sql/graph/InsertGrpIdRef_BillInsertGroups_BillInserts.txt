select
	t1.INSERT_ID as insertId,
	t1.INSERT_GRP_ID as insertGrpId,
	t2.DESCRIPTION as billInsertDescription,
	t4.DISPLAY_VALUE as name
from
	BILL_INSERT_GROUPS t1
	inner join BILL_INSERTS t2 on t1.INSERT_ID = t2.INSERT_ID
		and t2.LANGUAGE_CODE = {2}
	inner join INSERT_GRP_ID_REF t3 on t1.INSERT_GRP_ID = t3.INSERT_GRP_ID
	inner join INSERT_GRP_ID_VALUES t4 on t1.INSERT_GRP_ID = t4.INSERT_GRP_ID
		and t4.LANGUAGE_CODE = {2}
