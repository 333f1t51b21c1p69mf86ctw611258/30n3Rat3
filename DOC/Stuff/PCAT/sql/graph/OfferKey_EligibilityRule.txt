select
	t4.ELIGIBILITY_RULE_ID as eligibilityRuleId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
        t1.OFFER_ID as offerId,
    ''{3}'' as ruleType
from
	ELIGIBILITY_RULE_KEY t4
	inner join ELIGIBILITY_RULE_REF t2 on t4.ELIGIBILITY_RULE_ID = t2.ELIGIBILITY_RULE_ID
	inner join ELIGIBILITY_RULE_VALUES t3 on t2.ELIGIBILITY_RULE_ID = t3.ELIGIBILITY_RULE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join OFFER_REF t1 on t4.ELIGIBILITY_RULE_ID = t1.ELIGIBILITY_RULE_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
