select
	t1.CURRENCY_GROUP as currencyGroup,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.CURRENCY_CODE as currencyCode,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description
from
	RATE_CURRENCY_GROUP_KEY t1
	inner join RATE_CURRENCY_GROUP t2 on t1.CURRENCY_GROUP = t2.CURRENCY_GROUP
	left outer join RATE_CURRENCY_VALUES t3 on t2.CURRENCY_CODE = t3.CURRENCY_CODE
		and t3.SERVICE_VERSION_ID = {1}
		and t3.LANGUAGE_CODE = {2}
where
	t2.RESELLER_VERSION_ID = {0}
