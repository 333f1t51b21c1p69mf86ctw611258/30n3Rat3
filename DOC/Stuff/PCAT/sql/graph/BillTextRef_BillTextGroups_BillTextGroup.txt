select
	t1.GEO_AREA_ID as geoAreaId,
	t7.DISPLAY_VALUE as geoRegionKey,
	t1.GEO_REGION_ID as geoRegionId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as name,
	t4.DESCRIPTION as description
from
	BILL_TEXT_GROUPS t1
	inner join BILL_TEXT_GROUP_REF t3 on t1.TEXT_GROUP_ID = t3.TEXT_GROUP_ID
	inner join BILL_TEXT_GROUP_VALUES t4 on t1.TEXT_GROUP_ID = t4.TEXT_GROUP_ID
		and t4.LANGUAGE_CODE = {2}
	inner join BILL_TEXT_REF t6 on t1.TEXT_ID = t6.TEXT_ID
	inner join BILL_TEXT_VALUES t7 on t1.TEXT_ID = t6.TEXT_ID
		and t7.TEST_FLAG = t6.TEST_FLAG
		and t7.LANGUAGE_CODE = {2}
