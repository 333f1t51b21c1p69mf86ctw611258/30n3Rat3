select
	t4.TRANSITION_RULE_ID as transitionRuleId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
	t1.OFFER_ID as offerId,
    ''{3}'' as ruleType
from
	TRANSITION_RULE_KEY t4
	inner join TRANSITION_RULE_REF t2 on t4.TRANSITION_RULE_ID = t2.TRANSITION_RULE_ID
	inner join TRANSITION_RULE_VALUES t3 on t2.TRANSITION_RULE_ID = t3.TRANSITION_RULE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	inner join OFFER_REF t1 on t4.TRANSITION_RULE_ID = t1.TRANSITION_RULE_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
