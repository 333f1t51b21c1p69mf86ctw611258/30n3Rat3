select
	t1.BALANCE_ID as balanceId,
	t1.OFFER_ID as offerId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
	t5.DISPLAY_VALUE as unitType,
	t1.IS_CORE as isCore,
	t8.DISPLAY_VALUE as classification,
	t10.DISPLAY_VALUE as paymentMode,
	t1.BALANCE_ORDER as balanceOrder,
	t1.GRANT_AMOUNT as initialValue
from
	OFFER_BALANCE_MAP t1
	inner join BALANCE_REF t2 on t1.BALANCE_ID = t2.BALANCE_ID
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BALANCE_VALUES t3 on t1.BALANCE_ID = t3.BALANCE_ID
		and t3.LANGUAGE_CODE = {2}
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join UNITS_TYPE_REF t4 on t2.UNIT_TYPE = t4.UNIT_TYPE
		and t4.SERVICE_VERSION_ID = {1}
	inner join UNITS_TYPE_VALUES t5 on t2.UNIT_TYPE = t5.UNIT_TYPE
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
		and t5.SERVICE_VERSION_ID = t4.SERVICE_VERSION_ID
	inner join OFFER_REF t6 on t1.OFFER_ID = t6.OFFER_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t7 on t7.enumeration_key = lower(''SHADOW_REAL_OPT'')
		and t7.value = t1.SHADOW_REAL_OPT
	left outer join GENERIC_ENUMERATION_VALUES t8 on t8.enumeration_key = t7.enumeration_key
		and t8.value = t7.value
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t9 on t9.enumeration_key = lower(''BALANCE_PAYMENT_MODE'')
		and t9.value = t1.BALANCE_PAYMENT_MODE
	left outer join GENERIC_ENUMERATION_VALUES t10 on t10.enumeration_key = t9.enumeration_key
		and t10.value = t9.value
		and t10.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
