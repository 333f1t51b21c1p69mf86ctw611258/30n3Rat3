select
	t1.USAGE_PLAN_ID as usagePlanId,
	t1.USAGE_ITEM_ID as usageItemId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description
from
	USAGE_PLAN_ITEM_MAP t1
	inner join USAGE_PLAN_VALUES t2 on t1.USAGE_PLAN_ID = t2.USAGE_PLAN_ID
		and t2.LANGUAGE_CODE = {2}
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join USAGE_ITEM_VALUES t3 on t1.USAGE_ITEM_ID = t3.USAGE_ITEM_ID
		and t3.LANGUAGE_CODE = t2.LANGUAGE_CODE
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
