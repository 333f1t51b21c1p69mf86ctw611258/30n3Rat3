select
	t1.OFFER_ID as offerId,
	t1.FF_LIST_TEMPLATE_ID as ffListTemplateId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as name,
        t4.DESCRIPTION as description,
	t7.DISPLAY_VALUE as ffListTypeKey,
        ''{3}'' as templateType
from
	OFFER_FF_LIST_TEMPLATE_MAP t1
	inner join FF_LIST_TEMPLATE_KEY t2 on t1.FF_LIST_TEMPLATE_ID = t2.FF_LIST_TEMPLATE_ID
	inner join FF_LIST_TEMPLATE_REF t3 on t1.FF_LIST_TEMPLATE_ID = t3.FF_LIST_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join FF_LIST_TEMPLATE_VALUES t4 on t1.FF_LIST_TEMPLATE_ID = t4.FF_LIST_TEMPLATE_ID
		and t4.LANGUAGE_CODE = {2}
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join FF_LIST_TYPE_KEY t5 on t5.FF_LIST_TYPE_ID = t3.FF_LIST_TYPE_ID
	left outer join FF_LIST_TYPE_REF t6 on t6.FF_LIST_TYPE_ID = t3.FF_LIST_TYPE_ID
		and t6.SERVICE_VERSION_ID = {1}
	left outer join FF_LIST_TYPE_VALUES t7 on t7.FF_LIST_TYPE_ID = t3.FF_LIST_TYPE_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
                and t7.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
