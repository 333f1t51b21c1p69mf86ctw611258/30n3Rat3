select
	t1.EXCLUSION_RULE_MAP_ID as exclusionRuleMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.EXCLUSION_RULE_ID as exclusionRuleId,
	t1.BUNDLE_ID as bundleId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description,
	t5.DISPLAY_VALUE as bundleType
from
	EXCLUSION_RULE_MEMBER_MAP t1
	left outer join BUNDLE_REF t2 on t1.BUNDLE_ID = t2.BUNDLE_ID
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join BUNDLE_VALUES t3 on t1.BUNDLE_ID = t3.BUNDLE_ID
		and t3.LANGUAGE_CODE = {2}
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''LEVEL_CODE'')
		and t4.value = t2.BUNDLE_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
		and t1.BUNDLE_ID is not null
