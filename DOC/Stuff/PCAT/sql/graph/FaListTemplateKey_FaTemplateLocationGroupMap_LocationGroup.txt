select
	t1.LOCATION_GROUP_ID as locationGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description
from
	FA_TEMPLATE_LOCATION_GROUP_MAP t1
        inner join LOCATION_GROUP_REF t2 on t2.LOCATION_GROUP_ID = t1.LOCATION_GROUP_ID
		and t2.SERVICE_VERSION_ID = {1}
	inner join LOCATION_GROUP_VALUES t3 on t3.LOCATION_GROUP_ID = t1.LOCATION_GROUP_ID
		and t3.LANGUAGE_CODE = {2}
		and t3.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
