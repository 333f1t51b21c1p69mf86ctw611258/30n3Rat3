select
	t1.BUNDLE_GROUP_ID as bundleGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.BUNDLE_ID as bundleId,
	t4.DISPLAY_VALUE as name,
	t1.DISPLAY_ORDER as displayOrder,
	t6.DISPLAY_VALUE bundleType
from
	BUNDLE_GROUP_MEMBER t1
	inner join BUNDLE_GROUP_ID_VALUES t2 on t1.BUNDLE_GROUP_ID = t2.BUNDLE_GROUP_ID
		and t2.LANGUAGE_CODE = {2}
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BUNDLE_REF t3 on t1.BUNDLE_ID = t3.BUNDLE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BUNDLE_VALUES t4 on t1.BUNDLE_ID = t4.BUNDLE_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t5 on t5.enumeration_key = lower(''LEVEL_CODE'')
		and t5.value = t3.BUNDLE_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t6 on t6.enumeration_key = t5.enumeration_key
		and t6.value = t5.value
		and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
