select
	t1.SERV_PLAN_ITEM_MAP_ID as servPlanItemMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.SERVICE_PLAN_ID as servicePlanId,
	t1.SERVICE_ITEM_ID as serviceItemId,
	t2.DISPLAY_VALUE as name,
	t2.DESCRIPTION as description
from
	SERVICE_PLAN_ITEM_MAP t1
	left outer join SERVICE_ITEM_VALUES t2 on t1.SERVICE_ITEM_ID = t2.SERVICE_ITEM_ID
		and t2.LANGUAGE_CODE = {2}
		and t2.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
where
	t1.RESELLER_VERSION_ID = {0}
