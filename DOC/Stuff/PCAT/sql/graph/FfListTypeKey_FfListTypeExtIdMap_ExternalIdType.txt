select
	t1.FF_LIST_TYPE_ID as ffListTypeId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t3.DISPLAY_VALUE as name,
	t3.DESCRIPTION as description
from
	FF_LIST_TYPE_EXT_ID_MAP t1
        inner join EXTERNAL_ID_TYPE_REF t2 on t2.EXTERNAL_ID_TYPE = t1.EXTERNAL_ID_TYPE
		and t2.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join EXTERNAL_ID_TYPE_VALUES t3 on t3.EXTERNAL_ID_TYPE = t1.EXTERNAL_ID_TYPE
		and t3.LANGUAGE_CODE = {2}
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
where
	t1.SERVICE_VERSION_ID = {1}
