select
	t1.EXPRESSION_ID as expressionId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.EXPRESSION_GROUP_ID as expressionGroupId,
	t2.ATTRIBUTE_ID as attributeId,
	t2.ATTRIBUTE_QUALIFIER as attributeQualifier,
	t5.DISPLAY_VALUE as operator,
	t2.VALUE as value,
	t2.ATTR_SRC as attrSrc
from
	EXPRESSION_KEY t1
	inner join EXPRESSION t2 on t1.EXPRESSION_ID = t2.EXPRESSION_ID
	inner join EXPRESSION_GROUP_KEY t3 on t2.EXPRESSION_GROUP_ID = t3.EXPRESSION_GROUP_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''OPERATOR'')
		and t4.value = t2.OPERATOR
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = {2}
where
	t2.RESELLER_VERSION_ID = {0}
