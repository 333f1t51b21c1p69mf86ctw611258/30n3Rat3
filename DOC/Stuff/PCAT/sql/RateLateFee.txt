select
	t3.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as franchiseCodeKey,
	t1.FRANCHISE_TAX_CODE as franchiseTaxCode,
	t6.DISPLAY_VALUE as billingFrequencyRef,
	t1.BILLING_FREQUENCY as billingFrequency,
	t8.SERVICE_VERSION_ID as serviceVersionId,
	t8.ISO_CODE as currencyIsoCode,
	t8.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t9.DISPLAY_VALUE as rateCurrencyKey,
	t1.CURRENCY_CODE as currencyCode,
	t1.DATE_ACTIVE as dateActive,
	t1.GRACE_PERIOD as gracePeriod,
	t1.FEE_CLASS as feeClass,
	t1.NRC_TERM_ID as nrcTermId,
	t1.FEE_PERCENT as feePercent,
	t1.IMPLIED_DECIMALS as impliedDecimals,
	t1.FEE_AMOUNT as feeAmount,
	t1.MINIMUM_AMOUNT as minimumAmount,
	t1.MAXIMUM_AMOUNT as maximumAmount,
	t1.DATE_INACTIVE as dateInactive,
	t1.DATE_CREATED as dateCreated,
	t1.MIN_BAL_THRESHOLD as minBalThreshold
from
	RATE_LATE_FEE t1
	inner join FRANCHISE_CODE_KEY t2 on t1.FRANCHISE_TAX_CODE = t2.FRANCHISE_CODE
	inner join FRANCHISE_CODE_REF t3 on t2.FRANCHISE_CODE = t3.FRANCHISE_CODE
		and t3.RESELLER_VERSION_ID = {0}
	inner join FRANCHISE_CODE_VALUES t4 on t2.FRANCHISE_CODE = t4.FRANCHISE_CODE
		and t4.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join BILLING_FREQUENCY_REF t5 on t1.BILLING_FREQUENCY = t5.BILLING_FREQUENCY
	inner join BILLING_FREQUENCY_VALUES t6 on t5.BILLING_FREQUENCY = t6.BILLING_FREQUENCY
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join RATE_CURRENCY_KEY t7 on t1.CURRENCY_CODE = t7.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t8 on t7.CURRENCY_CODE = t8.CURRENCY_CODE
		and t8.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t9 on t7.CURRENCY_CODE = t9.CURRENCY_CODE
		and t9.SERVICE_VERSION_ID = t8.SERVICE_VERSION_ID
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
