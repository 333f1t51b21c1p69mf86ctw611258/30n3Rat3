select
	t3.DISPLAY_VALUE as invsEquipChargeCodeRef,
	t1.EQUIP_CHARGE_CODE as equipChargeCode,
	t6.DISPLAY_VALUE as rcTermKey,
	t1.RC_TERM_ID as rcTermId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	EQUIP_CHARGE_CODE_RC_TERM_MAP t1
	inner join INVS_EQUIP_CHARGE_CODE_REF t2 on t1.EQUIP_CHARGE_CODE = t2.EQUIP_CHARGE_CODE
	inner join INVS_EQUIP_CHARGE_CODE_VALUES t3 on t2.EQUIP_CHARGE_CODE = t3.EQUIP_CHARGE_CODE
		and t3.LANGUAGE_CODE = {2}
	inner join RC_TERM_KEY t4 on t1.RC_TERM_ID = t4.RC_TERM_ID
	inner join RC_TERM_REF t5 on t4.RC_TERM_ID = t5.RC_TERM_ID
		and t5.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join RC_TERM_VALUES t6 on t4.RC_TERM_ID = t6.RC_TERM_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
