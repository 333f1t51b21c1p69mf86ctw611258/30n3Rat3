select
	t1.EXTERNAL_INTERFACE_PARAM_ID as externalInterfaceParamId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as externalInterfaceKey,
	t1.INTERFACE_ID as interfaceId,
	t1.SITE_ID as siteId,
	t1.PRIMARY_ADDRESS as primaryAddress,
	t1.SECONDARY_ADDRESS as secondaryAddress,
	t1.PRIMARY_PORT as primaryPort,
	t1.SECONDARY_PORT as secondaryPort
from
	EXTERNAL_INTERFACE_PARAM t1
	inner join EXTERNAL_INTERFACE_KEY t2 on t1.INTERFACE_ID = t2.INTERFACE_ID
	inner join EXTERNAL_INTERFACE_REF t3 on t2.INTERFACE_ID = t3.INTERFACE_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join EXTERNAL_INTERFACE_VALUES t4 on t2.INTERFACE_ID = t4.INTERFACE_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
