select
	t1.BT_PROMOTION_PLAN_ID as btPromotionPlanId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t5.DISPLAY_VALUE as promotionType,
	t2.RECOMMENDED_RC_VALUE as recommendedRcValue,
	t2.FOREIGN_CODE as foreignCode,
	t7.DISPLAY_VALUE as bonusPointIdRef,
	t2.BONUS_POINT_ID as bonusPointId,
	t9.DISPLAY_VALUE as rolloverOrder,
	t11.DISPLAY_VALUE as rolloverGrouping,
	t13.DISPLAY_VALUE as discountBasePeriod,
	t15.DISPLAY_VALUE as billPeriodRef,
	t2.BILL_PERIOD as billPeriod,
	t2.PRORATE_CONTRACT_START as prorateContractStart,
	t2.PRORATE_CONTRACT_END as prorateContractEnd,
	t2.PRORATE_INTERIM_BILL as prorateInterimBill,
	t2.PRORATE_PERIOD_CHANGE as proratePeriodChange,
	t2.PRORATE_ACCOUNT_START as prorateAccountStart,
	t2.PRORATE_ACCOUNT_END as prorateAccountEnd,
	t2.PRORATE_TARGET_THRESHOLD as prorateTargetThreshold,
	t2.PRORATE_TARGET_REBATES as prorateTargetRebates,
	t17.DISPLAY_VALUE as minDurationUnits,
	t2.MIN_DURATION as minDuration,
	t19.DISPLAY_VALUE as minNoticeUnits,
	t2.MIN_NOTICE_DELAY as minNoticeDelay,
	t2.PARTIAL_PERIOD as partialPeriod,
	t2.GLOBAL_CONTRACT_START_DT as globalContractStartDt,
	t2.GLOBAL_CONTRACT_END_DT as globalContractEndDt,
	t2.IS_EXCLUSIVE as isExclusive,
	t2.IS_MULTI_THRESHOLD as isMultiThreshold,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	BT_PROMOTION_PLAN_KEY t1
	inner join BT_PROMOTION_PLAN_REF t2 on t1.BT_PROMOTION_PLAN_ID = t2.BT_PROMOTION_PLAN_ID
	inner join BT_PROMOTION_PLAN_VALUES t3 on t2.BT_PROMOTION_PLAN_ID = t3.BT_PROMOTION_PLAN_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''PROMOTION_TYPE'')
		and t4.value = t2.PROMOTION_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join BONUS_POINT_ID_REF t6 on t2.BONUS_POINT_ID = t6.BONUS_POINT_ID
	left outer join BONUS_POINT_ID_VALUES t7 on t6.BONUS_POINT_ID = t7.BONUS_POINT_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t8 on t8.enumeration_key = lower(''ROLLOVER_ORDER'')
		and t8.value = t2.ROLLOVER_ORDER
	left outer join GENERIC_ENUMERATION_VALUES t9 on t9.enumeration_key = t8.enumeration_key
		and t9.value = t8.value
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t10 on t10.enumeration_key = lower(''ROLLOVER_GROUPING'')
		and t10.value = t2.ROLLOVER_GROUPING
	left outer join GENERIC_ENUMERATION_VALUES t11 on t11.enumeration_key = t10.enumeration_key
		and t11.value = t10.value
		and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t12 on t12.enumeration_key = lower(''DISCOUNT_BASE_PERIOD'')
		and t12.value = t2.DISCOUNT_BASE_PERIOD
	left outer join GENERIC_ENUMERATION_VALUES t13 on t13.enumeration_key = t12.enumeration_key
		and t13.value = t12.value
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join BILL_PERIOD_REF t14 on t2.BILL_PERIOD = t14.BILL_PERIOD
	left outer join BILL_PERIOD_VALUES t15 on t14.BILL_PERIOD = t15.BILL_PERIOD
		and t15.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t16 on t16.enumeration_key = lower(''MIN_DURATION_UNITS'')
		and t16.value = t2.MIN_DURATION_UNITS
	left outer join GENERIC_ENUMERATION_VALUES t17 on t17.enumeration_key = t16.enumeration_key
		and t17.value = t16.value
		and t17.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t18 on t18.enumeration_key = lower(''MIN_NOTICE_UNITS'')
		and t18.value = t2.MIN_NOTICE_UNITS
	left outer join GENERIC_ENUMERATION_VALUES t19 on t19.enumeration_key = t18.enumeration_key
		and t19.value = t18.value
		and t19.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
