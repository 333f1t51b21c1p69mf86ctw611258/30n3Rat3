select
	t4.DISPLAY_VALUE as liabilityRuleIdKey,
	t1.LIABILITY_RULE_ID as liabilityRuleId,
	t7.DISPLAY_VALUE as autGroupKey,
	t1.AUT_GROUP_ID as autGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	LIABILITY_RULE_AUT_GROUP_MAP t1
	inner join LIABILITY_RULE_ID_KEY t2 on t1.LIABILITY_RULE_ID = t2.LIABILITY_RULE_ID
	inner join LIABILITY_RULE_ID_REF t3 on t2.LIABILITY_RULE_ID = t3.LIABILITY_RULE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join LIABILITY_RULE_ID_VALUES t4 on t2.LIABILITY_RULE_ID = t4.LIABILITY_RULE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join AUT_GROUP_KEY t5 on t1.AUT_GROUP_ID = t5.AUT_GROUP_ID
	inner join AUT_GROUP_REF t6 on t5.AUT_GROUP_ID = t6.AUT_GROUP_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join AUT_GROUP_VALUES t7 on t5.AUT_GROUP_ID = t7.AUT_GROUP_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
