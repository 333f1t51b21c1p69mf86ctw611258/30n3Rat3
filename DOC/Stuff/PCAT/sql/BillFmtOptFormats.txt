select
	t3.DISPLAY_VALUE as billFmtOptRef,
	t1.BILL_FMT_OPT as billFmtOpt,
	t1.DOCUMENT_TYPE_ID as documentTypeId,
	t5.DISPLAY_VALUE as signOfAmt,
	t1.FILENAME as filename,
	t7.DISPLAY_VALUE as printDriver,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.IS_DEFAULT as isDefault,
	t1.CREATE_ASCII_BILL_IMAGE as createAsciiBillImage
from
	BILL_FMT_OPT_FORMATS t1
	inner join BILL_FMT_OPT_REF t2 on t1.BILL_FMT_OPT = t2.BILL_FMT_OPT
	inner join BILL_FMT_OPT_VALUES t3 on t2.BILL_FMT_OPT = t3.BILL_FMT_OPT
		and t3.LANGUAGE_CODE = {2}
        left outer join GUI_INDICATOR_REF t4 on t4.table_name = ''BILL_FMT_OPT_FORMATS''
		and t4.field_name = lower(''SIGN_OF_AMT'')
		and t4.integer_value = t1.SIGN_OF_AMT
	left outer join GUI_INDICATOR_VALUES t5 on t5.table_name = t4.table_name
		and t5.field_name = t4.field_name
		and t5.integer_value = t4.integer_value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t6 on t6.enumeration_key = lower(''PRINT_DRIVER'')
		and t6.value = t1.PRINT_DRIVER
	left outer join GENERIC_ENUMERATION_VALUES t7 on t7.enumeration_key = t6.enumeration_key
		and t7.value = t6.value
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join DOCUMENT_TYPE_REF t8 on t1.DOCUMENT_TYPE_ID = t8.DOCUMENT_TYPE_ID
	inner join DOCUMENT_TYPE_VALUES t9 on t8.DOCUMENT_TYPE_ID = t9.DOCUMENT_TYPE_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
