select
	t1.BONUS_POINT_TRANS_TYPE as bonusPointTransType,
	t1.BONUS_POINT_TRANS_CATEGORY as bonusPointTransCategory,
	t1.TRANS_SIGN as transSign,
	t1.IS_DISPLAYED_ONBILL as isDisplayedOnbill,
	t1.IS_VIEWABLE_ON_GUI as isViewableOnGui,
	t1.IS_EXTERNAL as isExternal,
	t1.IS_SYSTEM_GENERATED as isSystemGenerated,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BONUS_POINT_TRANS_TYPE_REF t1
	inner join BONUS_POINT_TRANS_TYPE_VALUES t2 on t1.BONUS_POINT_TRANS_TYPE = t2.BONUS_POINT_TRANS_TYPE
where
	t2.LANGUAGE_CODE = {2}
