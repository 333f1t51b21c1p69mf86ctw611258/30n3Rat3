select
	t4.DISPLAY_VALUE as liabilityTemplateIdKey,
	t1.TEMPLATE_ID as templateId,
	t7.DISPLAY_VALUE as liabilityRuleIdKey,
	t1.LIABILITY_RULE_ID as liabilityRuleId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t10.DISPLAY_VALUE as calendarKey,
	t1.CALENDAR_ID as calendarId,
	t1.PRIORITY as priority
from
	LIABILITY_TEMPLATE_MAPPING t1
	inner join LIABILITY_TEMPLATE_ID_KEY t2 on t1.TEMPLATE_ID = t2.LR_TEMPLATE_ID
	inner join LIABILITY_TEMPLATE_ID_REF t3 on t2.LR_TEMPLATE_ID = t3.LR_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join LIABILITY_TEMPLATE_ID_VALUES t4 on t2.LR_TEMPLATE_ID = t4.LR_TEMPLATE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join LIABILITY_RULE_ID_KEY t5 on t1.LIABILITY_RULE_ID = t5.LIABILITY_RULE_ID
	inner join LIABILITY_RULE_ID_REF t6 on t5.LIABILITY_RULE_ID = t6.LIABILITY_RULE_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join LIABILITY_RULE_ID_VALUES t7 on t5.LIABILITY_RULE_ID = t7.LIABILITY_RULE_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join CALENDAR_KEY t8 on t1.CALENDAR_ID = t8.CALENDAR_ID
	left outer join CALENDAR_REF t9 on t8.CALENDAR_ID = t9.CALENDAR_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join CALENDAR_VALUES t10 on t8.CALENDAR_ID = t10.CALENDAR_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
