select
	t1.BALANCE_EXC_INC_ID as balanceExcIncId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId,
	t7.DISPLAY_VALUE as balanceKey,
	t1.BALANCE_ID as balanceId,
	t10.DISPLAY_VALUE as autFinalKey,
	t1.AUT_ID as autId,
	t13.DISPLAY_VALUE as autGroupKey,
	t1.AUT_GROUP_ID as autGroupId,
	t16.DISPLAY_VALUE as rcTermKey,
	t1.RC_TERM_ID as rcTermId,
	t19.DISPLAY_VALUE as nrcTermKey,
	t1.NRC_TERM_ID as nrcTermId,        
        (case t1.EXCL_INCL_FLAG
          when 0 then {3}
          when 1 then {4}
         end) exclInclFlag,
	t22.DISPLAY_VALUE as timeTypeKey,
	t1.TIME_TYPE_ID as timeTypeId
from
	BALANCE_EXCLUSION_INCLUSION t1
	inner join OFFER_KEY t2 on t1.OFFER_ID = t2.OFFER_ID
	inner join OFFER_REF t3 on t2.OFFER_ID = t3.OFFER_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join OFFER_VALUES t4 on t2.OFFER_ID = t4.OFFER_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join BALANCE_KEY t5 on t1.BALANCE_ID = t5.BALANCE_ID
	inner join BALANCE_REF t6 on t5.BALANCE_ID = t6.BALANCE_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BALANCE_VALUES t7 on t5.BALANCE_ID = t7.BALANCE_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join AUT_FINAL_KEY t8 on t1.AUT_ID = t8.AUT_ID
	left outer join AUT_FINAL_REF t9 on t8.AUT_ID = t9.AUT_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join AUT_FINAL_VALUES t10 on t8.AUT_ID = t10.AUT_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join AUT_GROUP_KEY t11 on t1.AUT_GROUP_ID = t11.AUT_GROUP_ID
	left outer join AUT_GROUP_REF t12 on t11.AUT_GROUP_ID = t12.AUT_GROUP_ID
		and t12.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join AUT_GROUP_VALUES t13 on t11.AUT_GROUP_ID = t13.AUT_GROUP_ID
		and t13.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t13.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join RC_TERM_KEY t14 on t1.RC_TERM_ID = t14.RC_TERM_ID
	left outer join RC_TERM_REF t15 on t14.RC_TERM_ID = t15.RC_TERM_ID
		and t15.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join RC_TERM_VALUES t16 on t14.RC_TERM_ID = t16.RC_TERM_ID
		and t16.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t16.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join NRC_TERM_KEY t17 on t1.NRC_TERM_ID = t17.NRC_TERM_ID
	left outer join NRC_TERM_REF t18 on t17.NRC_TERM_ID = t18.NRC_TERM_ID
		and t18.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join NRC_TERM_VALUES t19 on t17.NRC_TERM_ID = t19.NRC_TERM_ID
		and t19.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t19.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join TIME_TYPE_KEY t20 on t1.TIME_TYPE_ID = t20.TIME_TYPE_ID
	left outer join TIME_TYPE_REF t21 on t20.TIME_TYPE_ID = t21.TIME_TYPE_ID
		and t21.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join TIME_TYPE_VALUES t22 on t20.TIME_TYPE_ID = t22.TIME_TYPE_ID
		and t22.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t22.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
