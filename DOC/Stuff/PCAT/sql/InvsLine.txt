select
	t1.INVENTORY_LINE_ID as inventoryLineId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	INVS_LINE_REF t1
	inner join INVS_LINE_VALUES t2 on t1.INVENTORY_LINE_ID = t2.INVENTORY_LINE_ID
where
	t2.LANGUAGE_CODE = {2}
