select
	t1.MIG_SCRIPT_ID as migScriptId,
	t1.TICKET_ID as ticketId,
	t1.MIG_SCRIPT_NAME as migScriptName,
	t1.MIG_SCRIPT_START as migScriptStart,
	t1.MIG_SCRIPT_STATUS as migScriptStatus,
	t1.MIG_SCRIPT_END as migScriptEnd,
	t1.MIG_SCRIPT_PREREQ as migScriptPrereq,
	t1.MIG_SCRIPT_EQUIV as migScriptEquiv
from
	REG_MIG_SCRIPT t1
	inner join REG_TICKET t2 on t1.TICKET_ID = t2.TICKET_ID
