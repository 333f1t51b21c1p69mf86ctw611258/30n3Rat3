select
	t1.EQUIP_TYPE_CODE as equipTypeCode,
	t1.FOREIGN_CODE as foreignCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	EQUIP_TYPE_CODE_REF t1
	inner join EQUIP_TYPE_CODE_VALUES t2 on t1.EQUIP_TYPE_CODE = t2.EQUIP_TYPE_CODE
where
	t2.LANGUAGE_CODE = {2}
