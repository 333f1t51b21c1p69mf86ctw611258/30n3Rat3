select
	t1.TARIFF_SET_MEMBER_ID as tariffSetMemberId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as tariffSetIdKey,
	t1.TARIFF_SET_ID as tariffSetId,
	t7.DISPLAY_VALUE as tariffKey,
	t1.TARIFF_ID as tariffId,
	t10.DISPLAY_VALUE as timeTypeKey,
	t1.TIME_TYPE_ID as timeTypeId,
	t1.IS_PRIMARY as isPrimary
from
	TARIFF_SET_MEMBER t1
	inner join TARIFF_SET_ID_KEY t2 on t1.TARIFF_SET_ID = t2.TARIFF_SET_ID
	inner join TARIFF_SET_ID_REF t3 on t2.TARIFF_SET_ID = t3.TARIFF_SET_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TARIFF_SET_ID_VALUES t4 on t2.TARIFF_SET_ID = t4.TARIFF_SET_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join TARIFF_KEY t5 on t1.TARIFF_ID = t5.TARIFF_ID
	left outer join TARIFF_REF t6 on t5.TARIFF_ID = t6.TARIFF_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join TARIFF_VALUES t7 on t5.TARIFF_ID = t7.TARIFF_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join TIME_TYPE_KEY t8 on t1.TIME_TYPE_ID = t8.TIME_TYPE_ID
	inner join TIME_TYPE_REF t9 on t8.TIME_TYPE_ID = t9.TIME_TYPE_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TIME_TYPE_VALUES t10 on t8.TIME_TYPE_ID = t10.TIME_TYPE_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
