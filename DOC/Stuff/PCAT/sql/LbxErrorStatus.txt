select
	t1.LBX_ERROR_STATUS as lbxErrorStatus,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	LBX_ERROR_STATUS_REF t1
	inner join LBX_ERROR_STATUS_VALUES t2 on t1.LBX_ERROR_STATUS = t2.LBX_ERROR_STATUS
where
	t2.LANGUAGE_CODE = {2}
