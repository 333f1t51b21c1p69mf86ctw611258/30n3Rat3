select
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as brandKey,
	t1.BRANDING_ID as brandingId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId
from
	BRAND_RESELLER_OFFER_MAP t1
	inner join BRAND_KEY t2 on t1.BRANDING_ID = t2.BRANDING_ID
	inner join BRAND_REF t3 on t2.BRANDING_ID = t3.BRANDING_ID
		and t3.SERVICE_VERSION_ID = {1}
	inner join BRAND_VALUES t4 on t2.BRANDING_ID = t4.BRANDING_ID
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join OFFER_KEY t5 on t1.OFFER_ID = t5.OFFER_ID
	left outer join OFFER_REF t6 on t5.OFFER_ID = t6.OFFER_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join OFFER_VALUES t7 on t5.OFFER_ID = t7.OFFER_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
