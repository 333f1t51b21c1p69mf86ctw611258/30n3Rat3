select
	t1.MSG_TRIGGER_ID as msgTriggerId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	MSG_TRIGGER_ID_REF t1
	inner join MSG_TRIGGER_ID_VALUES t2 on t1.MSG_TRIGGER_ID = t2.MSG_TRIGGER_ID
where
	t2.LANGUAGE_CODE = {2}
