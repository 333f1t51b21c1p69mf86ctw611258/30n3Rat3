select
	t1.COUNTRY_CODE as countryCode,
	t4.DISPLAY_VALUE as countryCodeRef,
	t1.STATE_GEOCODE as stateGeocode,
	t1.STATE_ABBR as stateAbbr,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as languageCodeRef,
	t2.DISPLAY_VALUE as displayValue
from
	STATE_REF t1
	inner join STATE_VALUES t2 on t1.COUNTRY_CODE = t2.COUNTRY_CODE
		and t1.STATE_GEOCODE = t2.STATE_GEOCODE
	inner join LANGUAGE_CODE_VALUES t3 on t2.LANGUAGE_CODE = t3.INPUT_LANGUAGE
	inner join COUNTRY_CODE_VALUES t4 on t2.COUNTRY_CODE = t4.COUNTRY_CODE
