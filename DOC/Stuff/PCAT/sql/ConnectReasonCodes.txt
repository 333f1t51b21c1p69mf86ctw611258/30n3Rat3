select
	t1.CONNECT_REASON as connectReason,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CONNECT_REASON_CODES_REF t1
	inner join CONNECT_REASON_CODES_VALUES t2 on t1.CONNECT_REASON = t2.CONNECT_REASON
where
	t2.LANGUAGE_CODE = {2}
