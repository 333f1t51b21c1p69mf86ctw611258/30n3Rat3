select
	t1.RATING_ID as ratingId,
	t1.P_RATING_ID as pRatingId,
	t4.DISPLAY_VALUE as ctcRatingTypeRef,
	t1.RATING_TYPE_ID as ratingTypeId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_RATING_REF t1
	inner join CTC_RATING_VALUES t2 on t1.RATING_ID = t2.RATING_ID
	inner join CTC_RATING_TYPE_REF t3 on t1.RATING_TYPE_ID = t3.RATING_TYPE_ID
	inner join CTC_RATING_TYPE_VALUES t4 on t3.RATING_TYPE_ID = t4.RATING_TYPE_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
