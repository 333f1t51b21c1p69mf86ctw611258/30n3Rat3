select
	t1.PRIORITY_TAG as priorityTag,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARM_PRIORITY_TAG_REF t1
	inner join ARM_PRIORITY_TAG_VALUES t2 on t1.PRIORITY_TAG = t2.PRIORITY_TAG
where
	t2.LANGUAGE_CODE = {2}
