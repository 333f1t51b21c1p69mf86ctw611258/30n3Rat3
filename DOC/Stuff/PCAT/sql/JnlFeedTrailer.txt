select
	t1.TRL_TYPE as trlType,
	t1.TRL_POSITION as trlPosition,
	t1.JNL_FEED_CODE as jnlFeedCodeCode,
	t1.TRL_TABLE as trlTable,
	t1.TRL_FIELD as trlField,
	t1.TRL_FIELD_SIZE as trlFieldSize,
	t1.TRL_HARDCODED as trlHardcoded,
	t1.TRL_FIELD_FORMAT as trlFieldFormat,
	t1.TRL_FIELD_TRIM as trlFieldTrim,
	t1.TRL_FIELD_DESCRIPTION as trlFieldDescription,
	t1.IN_FEED_REPORT as inFeedReport
from
	JNL_FEED_TRAILER t1
	inner join JNL_FEED_CODE t2 on t1.JNL_FEED_CODE = t2.JNL_FEED_CODE
