select
	t3.DISPLAY_VALUE as statusReasonRef,
	t1.STATUS_REASON_ID as statusReasonId,
	t5.DISPLAY_VALUE as statusRef,
	t1.STATUS_ID as statusId
from
	STATUS_REASON_MAP t1
	inner join STATUS_REASON_REF t2 on t1.STATUS_REASON_ID = t2.STATUS_REASON_ID
	inner join STATUS_REASON_VALUES t3 on t2.STATUS_REASON_ID = t3.STATUS_REASON_ID
		and t3.LANGUAGE_CODE = {2}
	inner join STATUS_REF t4 on t1.STATUS_ID = t4.STATUS_ID
	inner join STATUS_VALUES t5 on t4.STATUS_ID = t5.STATUS_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
