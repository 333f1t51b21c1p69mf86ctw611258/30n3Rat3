select
	t1.TAP_MAPPING_ID as tapMappingId,
	t1.RECORD_TYPE as recordType,
	t1.SENDER as sender,
	t3.DISPLAY_VALUE as serviceType,
	t1.SERVICE_CODE as serviceCode,
	t1.SS_CODE as ssCode,
	t1.SS_ACTION_CODE as ssActionCode,
	t1.SUBSCR_ID_TYPE as subscrIdType,
	t1.CHARGED_PARTY_STATUS as chargedPartyStatus,
	t1.VAS_CODE as vasCode,
	t1.SERVICE_KEY as serviceKey,
	t1.TYPE_ID_USG as typeIdUsg,
	t1.EXT_ID_TYPE as extIdType
from
	TAP_MAPPING t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''SERVICE_TYPE'')
		and t2.value = t1.SERVICE_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
