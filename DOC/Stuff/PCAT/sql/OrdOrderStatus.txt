select
	t1.ORDER_STATUS_ID as orderStatusId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ORD_ORDER_STATUS_REF t1
	inner join ORD_ORDER_STATUS_VALUES t2 on t1.ORDER_STATUS_ID = t2.ORDER_STATUS_ID
where
	t2.LANGUAGE_CODE = {2}
