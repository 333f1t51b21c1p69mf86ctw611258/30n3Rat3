select
	t1.CORRIDOR_PLAN_ID as corridorPlanId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.PLAN_TYPE as planType,
	t2.PLAN_LEVEL as planLevel,
	t2.ORIGIN_RESTRICTION as originRestriction,
	t2.TARGET_RESTRICTION as targetRestriction,
	t2.DISCOUNT_ID as discountId,
	t2.MAX_CORRIDORS as maxCorridors,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	CORRIDOR_PLAN_ID_KEY t1
	inner join CORRIDOR_PLAN_ID_REF t2 on t1.CORRIDOR_PLAN_ID = t2.CORRIDOR_PLAN_ID
	inner join CORRIDOR_PLAN_ID_VALUES t3 on t2.CORRIDOR_PLAN_ID = t3.CORRIDOR_PLAN_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
