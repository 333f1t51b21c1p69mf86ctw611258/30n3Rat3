select
	t1.BUNDLE_ID as bundleId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t5.DISPLAY_VALUE as bundleType,
	t2.SALES_EFFECTIVE_DT as salesEffectiveDt,
	t2.SALES_EXPIRATION_DT as salesExpirationDt,
	t2.OFFSET_DAYS as offsetDays,
	t2.AUTO_EXPIRATION_DT as autoExpirationDt,
	t2.AUTO_EXPIRATION_DURATION as autoExpirationDuration,
	t7.DISPLAY_VALUE as autoExpirationUnits,
	t2.IS_GENERALLY_AVAILABLE as isGenerallyAvailable,
	t2.IS_GLOBAL as isGlobal,
	t2.ALLOW_DUP_ACTIVITIES as allowDupActivities,
	t2.OFFER_SELECTIVE_MIN_COUNT as offerSelectiveMinCount,
	t2.OFFER_SELECTIVE_MAX_COUNT as offerSelectiveMaxCount,
	t2.BUNDLE_SELECTIVE_MIN_COUNT as bundleSelectiveMinCount,
	t2.BUNDLE_SELECTIVE_MAX_COUNT as bundleSelectiveMaxCount,
	t2.FOREIGN_CODE as foreignCode,
	t9.SERVICE_VERSION_ID as serviceVersionId,
	t10.DISPLAY_VALUE as regionKey,
	t2.REGION_ID as regionId,
	t13.DISPLAY_VALUE as contractTermKey,
	t2.CONTRACT_TERM_ID as contractTermId,
	t16.DISPLAY_VALUE as prereqRuleKey,
	t2.PREREQ_RULE_ID as prereqRuleId,
	t19.DISPLAY_VALUE as exclusionRuleKey,
	t2.EXCLUSION_RULE_ID as exclusionRuleId,
	t22.DISPLAY_VALUE as eligibilityRuleKey,
	t2.ELIGIBILITY_RULE_ID as eligibilityRuleId,
	t25.DISPLAY_VALUE as transitionRuleKey,
	t2.TRANSITION_RULE_ID as transitionRuleId,
	t28.DISPLAY_VALUE as overrideTemplateKey,
	t2.OVERRIDE_TARIFF_RULE_ID as overrideTariffRuleId,
	t31.DISPLAY_VALUE as overrideTemplateKeyByOverrideB,
	t2.OVERRIDE_BONUS_RULE_ID as overrideBonusRuleId,
	t34.DISPLAY_VALUE as overrideTemplateKeyByOverrideD,
	t2.OVERRIDE_DISCOUNT_RULE_ID as overrideDiscountRuleId,
	t36.DISPLAY_VALUE as insfBalanceMethod,
	t38.DISPLAY_VALUE as autoActivation,
	t2.SUPPRESS_BUNDLE_STATE_NOTIF as suppressBundleStateNotif,
	t2.IS_SYNCHRONIZED_TO_SALES as isSynchronizedToSales,
	t2.SALES_CURRENCY_CODE as salesCurrencyCode,
	t2.SALES_VALUE as salesValue,
	t2.COST_OF_GOODS_VALUE as costOfGoodsValue,
	t2.IS_CAMPAIGN_ONLY as isCampaignOnly,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	BUNDLE_KEY t1
	inner join BUNDLE_REF t2 on t1.BUNDLE_ID = t2.BUNDLE_ID
	inner join BUNDLE_VALUES t3 on t2.BUNDLE_ID = t3.BUNDLE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''BUNDLE_TYPE'')
		and t4.value = t2.BUNDLE_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t6 on t6.table_name = ''BUNDLE_REF''
		and t6.field_name = lower(''AUTO_EXPIRATION_UNITS'')
		and t6.integer_value = t2.AUTO_EXPIRATION_UNITS
	left outer join GUI_INDICATOR_VALUES t7 on t7.table_name = t6.table_name
		and t7.field_name = t6.field_name
		and t7.integer_value = t6.integer_value
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join REGION_KEY t8 on t2.REGION_ID = t8.REGION_ID
	left outer join REGION_REF t9 on t8.REGION_ID = t9.REGION_ID
		and t9.SERVICE_VERSION_ID = {1}
	left outer join REGION_VALUES t10 on t8.REGION_ID = t10.REGION_ID
		and t10.SERVICE_VERSION_ID = {1}
		and t10.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join CONTRACT_TERM_KEY t11 on t2.CONTRACT_TERM_ID = t11.CONTRACT_TERM_ID
	left outer join CONTRACT_TERM_REF t12 on t11.CONTRACT_TERM_ID = t12.CONTRACT_TERM_ID
		and t12.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join CONTRACT_TERM_VALUES t13 on t11.CONTRACT_TERM_ID = t13.CONTRACT_TERM_ID
		and t13.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t13.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join PREREQ_RULE_KEY t14 on t2.PREREQ_RULE_ID = t14.PREREQ_RULE_ID
	left outer join PREREQ_RULE_REF t15 on t14.PREREQ_RULE_ID = t15.PREREQ_RULE_ID
		and t15.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join PREREQ_RULE_VALUES t16 on t14.PREREQ_RULE_ID = t16.PREREQ_RULE_ID
		and t16.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t16.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join EXCLUSION_RULE_KEY t17 on t2.EXCLUSION_RULE_ID = t17.EXCLUSION_RULE_ID
	left outer join EXCLUSION_RULE_REF t18 on t17.EXCLUSION_RULE_ID = t18.EXCLUSION_RULE_ID
		and t18.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join EXCLUSION_RULE_VALUES t19 on t17.EXCLUSION_RULE_ID = t19.EXCLUSION_RULE_ID
		and t19.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t19.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ELIGIBILITY_RULE_KEY t20 on t2.ELIGIBILITY_RULE_ID = t20.ELIGIBILITY_RULE_ID
	left outer join ELIGIBILITY_RULE_REF t21 on t20.ELIGIBILITY_RULE_ID = t21.ELIGIBILITY_RULE_ID
		and t21.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join ELIGIBILITY_RULE_VALUES t22 on t20.ELIGIBILITY_RULE_ID = t22.ELIGIBILITY_RULE_ID
		and t22.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t22.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join TRANSITION_RULE_KEY t23 on t2.TRANSITION_RULE_ID = t23.TRANSITION_RULE_ID
	left outer join TRANSITION_RULE_REF t24 on t23.TRANSITION_RULE_ID = t24.TRANSITION_RULE_ID
		and t24.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join TRANSITION_RULE_VALUES t25 on t23.TRANSITION_RULE_ID = t25.TRANSITION_RULE_ID
		and t25.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t25.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join OVERRIDE_TEMPLATE_KEY t26 on t2.OVERRIDE_TARIFF_RULE_ID = t26.OVERRIDE_TEMPLATE_ID
	left outer join OVERRIDE_TEMPLATE_REF t27 on t26.OVERRIDE_TEMPLATE_ID = t27.OVERRIDE_TEMPLATE_ID
		and t27.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join OVERRIDE_TEMPLATE_VALUES t28 on t26.OVERRIDE_TEMPLATE_ID = t28.OVERRIDE_TEMPLATE_ID
		and t28.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t28.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join OVERRIDE_TEMPLATE_KEY t29 on t2.OVERRIDE_BONUS_RULE_ID = t29.OVERRIDE_TEMPLATE_ID
	left outer join OVERRIDE_TEMPLATE_REF t30 on t29.OVERRIDE_TEMPLATE_ID = t30.OVERRIDE_TEMPLATE_ID
		and t30.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join OVERRIDE_TEMPLATE_VALUES t31 on t29.OVERRIDE_TEMPLATE_ID = t31.OVERRIDE_TEMPLATE_ID
		and t31.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t31.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join OVERRIDE_TEMPLATE_KEY t32 on t2.OVERRIDE_DISCOUNT_RULE_ID = t32.OVERRIDE_TEMPLATE_ID
	left outer join OVERRIDE_TEMPLATE_REF t33 on t32.OVERRIDE_TEMPLATE_ID = t33.OVERRIDE_TEMPLATE_ID
		and t33.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join OVERRIDE_TEMPLATE_VALUES t34 on t32.OVERRIDE_TEMPLATE_ID = t34.OVERRIDE_TEMPLATE_ID
		and t34.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t34.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t35 on t35.enumeration_key = lower(''INSF_BALANCE_METHOD'')
		and t35.value = t2.INSF_BALANCE_METHOD
	left outer join GENERIC_ENUMERATION_VALUES t36 on t36.enumeration_key = t35.enumeration_key
		and t36.value = t35.value
		and t36.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t37 on t37.table_name = ''BUNDLE_REF''
		and t37.field_name = lower(''AUTO_ACTIVATION'')
		and t37.integer_value = t2.AUTO_ACTIVATION
	left outer join GUI_INDICATOR_VALUES t38 on t38.table_name = t37.table_name
		and t38.field_name = t37.field_name
		and t38.integer_value = t37.integer_value
		and t38.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
