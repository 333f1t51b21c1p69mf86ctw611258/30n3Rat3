select
	t1.BILL_FMT_OPT as billFmtOpt,
	t1.FORMAT_CODE as formatCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_FMT_OPT_REF t1
	inner join BILL_FMT_OPT_VALUES t2 on t1.BILL_FMT_OPT = t2.BILL_FMT_OPT
where
	t2.LANGUAGE_CODE = {2}
