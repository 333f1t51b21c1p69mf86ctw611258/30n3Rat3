select
	t1.LANGUAGE_CODE as languageCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.INPUT_LANGUAGE as inputLanguage,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	LANGUAGE_CODE_REF t1
        inner join LANGUAGE_CODE_VALUES t2 on t1.LANGUAGE_CODE = t2.INPUT_LANGUAGE
where
	t2.LANGUAGE_CODE = {2}
