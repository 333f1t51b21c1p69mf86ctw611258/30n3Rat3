select
	t1.ADJ_TRANS_CODE as adjTransCode,
	t4.DISPLAY_VALUE as billTypeRef,
	t1.BILL_TYPE as billType,
	t6.DISPLAY_VALUE as adjReasonCodeRef,
	t1.ADJ_REASON_CODE as adjReasonCode,
        t1.ADJ_TRANS_CODE || '' '' || t7.DESCRIPTION_TEXT as adjTransDescr
from
	ADJ_REASON_CODE_MAP t1
	inner join ADJ_TRANS_DESCR t2 on t1.ADJ_TRANS_CODE = t2.ADJ_TRANS_CODE
	inner join BILL_TYPE_REF t3 on t1.BILL_TYPE = t3.BILL_TYPE
	inner join BILL_TYPE_VALUES t4 on t3.BILL_TYPE = t4.BILL_TYPE
		and t4.LANGUAGE_CODE = {2}
	inner join ADJ_REASON_CODE_REF t5 on t1.ADJ_REASON_CODE = t5.ADJ_REASON_CODE
	inner join ADJ_REASON_CODE_VALUES t6 on t5.ADJ_REASON_CODE = t6.ADJ_REASON_CODE
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
        left outer join DESCRIPTIONS t7 on t2.DESCRIPTION_CODE = t7.DESCRIPTION_CODE
                and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
