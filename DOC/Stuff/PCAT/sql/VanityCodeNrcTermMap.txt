select
	t3.DISPLAY_VALUE as invsVanityRef,
	t1.VANITY_CODE as vanityCode,
	t6.DISPLAY_VALUE as nrcTermKey,
	t1.NRC_TERM_ID as nrcTermId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	VANITY_CODE_NRC_TERM_MAP t1
	inner join INVS_VANITY_REF t2 on t1.VANITY_CODE = t2.VANITY_CODE
	inner join INVS_VANITY_VALUES t3 on t2.VANITY_CODE = t3.VANITY_CODE
		and t3.LANGUAGE_CODE = {2}
	inner join NRC_TERM_KEY t4 on t1.NRC_TERM_ID = t4.NRC_TERM_ID
	inner join NRC_TERM_REF t5 on t4.NRC_TERM_ID = t5.NRC_TERM_ID
		and t5.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join NRC_TERM_VALUES t6 on t4.NRC_TERM_ID = t6.NRC_TERM_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
