select
	t1.EQUIP_CLASS_CODE as equipClassCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	EQUIP_CLASS_CODE_REF t1
	inner join EQUIP_CLASS_CODE_VALUES t2 on t1.EQUIP_CLASS_CODE = t2.EQUIP_CLASS_CODE
where
	t2.LANGUAGE_CODE = {2}
