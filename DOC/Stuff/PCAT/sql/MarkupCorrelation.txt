select
	t1.MARKUP_CORRELATION_ID as markupCorrelationId,
	t4.DISPLAY_VALUE as markupRatingKey,
	t1.MARKUP_ID as markupId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as tapGsmProviderIdRef,
	t1.TAP_GSM_PROVIDER_ID as tapGsmProviderId,
	t1.CHARGE_CODE as chargeCode,
	t9.DISPLAY_VALUE as tariffPlanKey,
	t1.TARIFF_PLAN_ID as tariffPlanId
from
	MARKUP_CORRELATION t1
	inner join MARKUP_RATING_KEY t2 on t1.MARKUP_ID = t2.MARKUP_ID
	inner join MARKUP_RATING_REF t3 on t2.MARKUP_ID = t3.MARKUP_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join MARKUP_RATING_VALUES t4 on t2.MARKUP_ID = t4.MARKUP_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join TAP_GSM_PROVIDER_ID_REF t5 on t1.TAP_GSM_PROVIDER_ID = t5.TAP_GSM_PROVIDER_ID
	inner join TAP_GSM_PROVIDER_ID_VALUES t6 on t5.TAP_GSM_PROV_ID = t6.TAP_GSM_PROV_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join TARIFF_PLAN_KEY t7 on t1.TARIFF_PLAN_ID = t7.TARIFF_PLAN_ID
	inner join TARIFF_PLAN_REF t8 on t7.TARIFF_PLAN_ID = t8.TARIFF_PLAN_ID
		and t8.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TARIFF_PLAN_VALUES t9 on t7.TARIFF_PLAN_ID = t9.TARIFF_PLAN_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
