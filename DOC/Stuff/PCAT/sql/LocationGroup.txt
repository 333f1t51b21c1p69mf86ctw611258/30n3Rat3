select
	t1.LOCATION_GROUP_ID as locationGroupId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as locationHierarchyKey,
	t2.MAX_HIERARCHY_LEVEL_ID as maxHierarchyLevelId,
	t9.DISPLAY_VALUE as locationHierarchyKeyByMinHiera,
	t2.MIN_HIERARCHY_LEVEL_ID as minHierarchyLevelId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	LOCATION_GROUP_KEY t1
	inner join LOCATION_GROUP_REF t2 on t1.LOCATION_GROUP_ID = t2.LOCATION_GROUP_ID
	inner join LOCATION_GROUP_VALUES t3 on t2.LOCATION_GROUP_ID = t3.LOCATION_GROUP_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join LOCATION_HIERARCHY_KEY t4 on t2.MAX_HIERARCHY_LEVEL_ID = t4.HIERARCHY_LEVEL_ID
	inner join LOCATION_HIERARCHY_REF t5 on t4.HIERARCHY_LEVEL_ID = t5.HIERARCHY_LEVEL_ID
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join LOCATION_HIERARCHY_VALUES t6 on t4.HIERARCHY_LEVEL_ID = t6.HIERARCHY_LEVEL_ID
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join LOCATION_HIERARCHY_KEY t7 on t2.MIN_HIERARCHY_LEVEL_ID = t7.HIERARCHY_LEVEL_ID
	left outer join LOCATION_HIERARCHY_REF t8 on t7.HIERARCHY_LEVEL_ID = t8.HIERARCHY_LEVEL_ID
		and t8.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	left outer join LOCATION_HIERARCHY_VALUES t9 on t7.HIERARCHY_LEVEL_ID = t9.HIERARCHY_LEVEL_ID
		and t9.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
