select
	t1.CUG_ACCESS_CODE_ID as cugAccessCodeId,
	t1.ACCESS_CODE as accessCode,
	t6.DISPLAY_VALUE as cugAccessCodeKey,
	t2.ACCESS_CODE as accessCode,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	CUG_ACCESS_CODE_KEY t1
	inner join CUG_ACCESS_CODE_REF t2 on t1.CUG_ACCESS_CODE_ID = t2.CUG_ACCESS_CODE_ID
	inner join CUG_ACCESS_CODE_VALUES t3 on t2.CUG_ACCESS_CODE_ID = t3.CUG_ACCESS_CODE_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join CUG_ACCESS_CODE_KEY t4 on t2.ACCESS_CODE = t4.ACCESS_CODE
	inner join CUG_ACCESS_CODE_REF t5 on t4.ACCESS_CODE = t5.ACCESS_CODE
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join CUG_ACCESS_CODE_VALUES t6 on t4.ACCESS_CODE = t6.ACCESS_CODE
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
