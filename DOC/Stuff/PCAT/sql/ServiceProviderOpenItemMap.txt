select
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as serviceProvidersKey,
	t1.PROVIDER_ID as providerId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as openItemIdKey,
	t1.OPEN_ITEM_ID as openItemId
from
	SERVICE_PROVIDER_OPEN_ITEM_MAP t1
	inner join SERVICE_PROVIDERS_KEY t2 on t1.PROVIDER_ID = t2.PROVIDER_ID
	inner join SERVICE_PROVIDERS_REF t3 on t2.PROVIDER_ID = t3.PROVIDER_ID
		and t3.SERVICE_VERSION_ID = {1}
	inner join SERVICE_PROVIDERS_VALUES t4 on t2.PROVIDER_ID = t4.PROVIDER_ID
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join OPEN_ITEM_ID_KEY t5 on t1.OPEN_ITEM_ID = t5.OPEN_ITEM_ID
	inner join OPEN_ITEM_ID_REF t6 on t5.OPEN_ITEM_ID = t6.OPEN_ITEM_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join OPEN_ITEM_ID_VALUES t7 on t5.OPEN_ITEM_ID = t7.OPEN_ITEM_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
