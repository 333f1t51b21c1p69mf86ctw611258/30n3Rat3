select
	t1.EXTERNAL_ID_TYPE as externalIdType,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t5.DISPLAY_VALUE as uniquenessType,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t2.IS_VIEWABLE as isViewable,
	t2.IS_RANGE_MAP as isRangeMap,
	t7.DISPLAY_VALUE as isNotification,
	t2.IS_FOR_RATING as isForRating,
	t2.IS_FANDF_ELIGIBLE as isFandfEligible,
	t2.NETWORK_ADDRESS_PLAN as networkAddressPlan,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	EXTERNAL_ID_TYPE_KEY t1
	inner join EXTERNAL_ID_TYPE_REF t2 on t1.EXTERNAL_ID_TYPE = t2.EXTERNAL_ID_TYPE
	inner join EXTERNAL_ID_TYPE_VALUES t3 on t2.EXTERNAL_ID_TYPE = t3.EXTERNAL_ID_TYPE
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''UNIQUENESS_TYPE'')
		and t4.value = t2.UNIQUENESS_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t6 on t6.enumeration_key = lower(''IS_NOTIFICATION'')
		and t6.value = t2.IS_NOTIFICATION
	left outer join GENERIC_ENUMERATION_VALUES t7 on t7.enumeration_key = t6.enumeration_key
		and t7.value = t6.value
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
