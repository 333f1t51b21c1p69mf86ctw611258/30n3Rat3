select
	t1.MSC_INFO_ID as mscInfoId,
	t1.MSC_ADDRESS as mscAddress,
	t2.MSC_ADDRESS as mscAddress,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.IS_IVR_SUPPORTED as isIvrSupported,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	MSC_INFO_KEY t1
	inner join MSC_INFO_REF t2 on t1.MSC_INFO_ID = t2.MSC_INFO_ID
	inner join MSC_INFO_VALUES t3 on t2.MSC_INFO_ID = t3.MSC_INFO_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID 
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
