select
	t1.CONTENT_TYPE_ID as contentTypeId,
	t4.SERVICE_VERSION_ID as serviceVersionId,
	t5.DISPLAY_VALUE as serviceProvidersKey,
	t1.SERVICE_PROVIDER_ID as serviceProviderId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_CONTENT_TYPE_REF t1
	inner join CTC_CONTENT_TYPE_VALUES t2 on t1.CONTENT_TYPE_ID = t2.CONTENT_TYPE_ID
	left outer join SERVICE_PROVIDERS_KEY t3 on t1.SERVICE_PROVIDER_ID = t3.PROVIDER_ID
	left outer join SERVICE_PROVIDERS_REF t4 on t3.PROVIDER_ID = t4.PROVIDER_ID
		and t4.SERVICE_VERSION_ID = {1}
	left outer join SERVICE_PROVIDERS_VALUES t5 on t3.PROVIDER_ID = t5.PROVIDER_ID
		and t5.SERVICE_VERSION_ID = {1}
		and t5.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
