select
	t1.ITEM_ACTION_ID as itemActionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ORD_ITEM_ACTION_REF t1
	inner join ORD_ITEM_ACTION_VALUES t2 on t1.ITEM_ACTION_ID = t2.ITEM_ACTION_ID
where
	t2.LANGUAGE_CODE = {2}
