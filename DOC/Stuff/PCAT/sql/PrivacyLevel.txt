select
	t1.PRIVACY_LEVEL as privacyLevel,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	PRIVACY_LEVEL_REF t1
	inner join PRIVACY_LEVEL_VALUES t2 on t1.PRIVACY_LEVEL = t2.PRIVACY_LEVEL
where
	t2.LANGUAGE_CODE = {2}
