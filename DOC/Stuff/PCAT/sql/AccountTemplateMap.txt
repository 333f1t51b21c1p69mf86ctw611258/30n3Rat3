select
	t1.ACCOUNT_TEMPLATE_MAP_ID as accountTemplateMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as eventKey,
	t1.EVENT_ID as eventId,
	t6.SERVICE_VERSION_ID as serviceVersionId,
	t7.DISPLAY_VALUE as externalInterfaceKey,
	t1.EXT_INTERFACE_ID as extInterfaceId,
	t10.DISPLAY_VALUE as accountTemplateKey,
	t1.ACCOUNT_TEMPLATE_ID as accountTemplateId,
	t12.DISPLAY_VALUE as notifIvrLanguageRef,
	t1.NOTIFICATION_LANGUAGE_ID as notificationLanguageId,
	t1.LATCH as latch,
	t1.DISPLAY_POINTER as displayPointer,
	t1.MESSAGE as message,
	t1.UNICODE_FLAG as unicodeFlag
from
	ACCOUNT_TEMPLATE_MAP t1
	inner join EVENT_KEY t2 on t1.EVENT_ID = t2.EVENT_ID
	inner join EVENT_REF t3 on t2.EVENT_ID = t3.EVENT_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join EVENT_VALUES t4 on t2.EVENT_ID = t4.EVENT_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join EXTERNAL_INTERFACE_KEY t5 on t1.EXT_INTERFACE_ID = t5.INTERFACE_ID
	inner join EXTERNAL_INTERFACE_REF t6 on t5.INTERFACE_ID = t6.INTERFACE_ID
		and t6.SERVICE_VERSION_ID = {1}
	inner join EXTERNAL_INTERFACE_VALUES t7 on t5.INTERFACE_ID = t7.INTERFACE_ID
		and t7.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join ACCOUNT_TEMPLATE_KEY t8 on t1.ACCOUNT_TEMPLATE_ID = t8.ACCOUNT_TEMPLATE_ID
	inner join ACCOUNT_TEMPLATE_REF t9 on t8.ACCOUNT_TEMPLATE_ID = t9.ACCOUNT_TEMPLATE_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join ACCOUNT_TEMPLATE_VALUES t10 on t8.ACCOUNT_TEMPLATE_ID = t10.ACCOUNT_TEMPLATE_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join NOTIF_IVR_LANGUAGE_REF t11 on t1.NOTIFICATION_LANGUAGE_ID = t11.NOTIF_IVR_LANGUAGE_ID
	inner join NOTIF_IVR_LANGUAGE_VALUES t12 on t11.NOTIF_IVR_LANGUAGE_ID = t12.NOTIF_IVR_LANGUAGE_ID
		and t12.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
