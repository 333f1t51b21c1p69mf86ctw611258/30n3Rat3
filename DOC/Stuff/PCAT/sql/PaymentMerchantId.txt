select
	t1.PAYMENT_MERCHANT_ID as paymentMerchantId,
	t1.EPG_MERCHANT_ID as epgMerchantId,
	t1.STATUS as status,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	PAYMENT_MERCHANT_ID_REF t1
	inner join PAYMENT_MERCHANT_ID_VALUES t2 on t1.PAYMENT_MERCHANT_ID = t2.PAYMENT_MERCHANT_ID
where
	t2.LANGUAGE_CODE = {2}
