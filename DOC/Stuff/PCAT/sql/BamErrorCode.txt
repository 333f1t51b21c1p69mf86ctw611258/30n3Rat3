select
	t1.BAM_ERROR_CODE as bamErrorCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BAM_ERROR_CODE_REF t1
	inner join BAM_ERROR_CODE_VALUES t2 on t1.BAM_ERROR_CODE = t2.BAM_ERROR_CODE
where
	t2.LANGUAGE_CODE = {2}
