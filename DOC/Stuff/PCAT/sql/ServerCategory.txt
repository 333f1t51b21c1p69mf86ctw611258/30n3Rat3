select
	t1.SERVER_CATEGORY as serverCategory,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	SERVER_CATEGORY_REF t1
	inner join SERVER_CATEGORY_VALUES t2 on t1.SERVER_CATEGORY = t2.SERVER_CATEGORY
where
	t2.LANGUAGE_CODE = {2}
