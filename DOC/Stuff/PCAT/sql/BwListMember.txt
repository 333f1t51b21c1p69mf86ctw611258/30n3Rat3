select
	t1.BW_LIST_MEMBER_ID as bwListMemberId,
	t4.DISPLAY_VALUE as bwListKey,
	t1.BW_LIST_ID as bwListId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t1.SUBSCRIBER_PREFIX as subscriberPrefix,
	t1.IDENTIFIER_PREFIX as identifierPrefix,
	t1.IS_ALLOWED as isAllowed
from
	BW_LIST_MEMBER t1
	inner join BW_LIST_KEY t2 on t1.BW_LIST_ID = t2.BW_LIST_ID
	inner join BW_LIST_REF t3 on t2.BW_LIST_ID = t3.BW_LIST_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join BW_LIST_VALUES t4 on t2.BW_LIST_ID = t4.BW_LIST_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
