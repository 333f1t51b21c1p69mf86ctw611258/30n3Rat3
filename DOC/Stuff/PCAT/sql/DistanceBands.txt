select
	t1.SEQNUM as seqnum,
	t3.DISPLAY_VALUE as distanceBandsIdRef,
	t1.DISTANCE_BAND_ID as distanceBandId,
	t5.SERVICE_VERSION_ID as serviceVersionId,
	t6.DISPLAY_VALUE as autInitialKey,
	t1.INITIAL_AUT_ID as initialAutId,
	t1.JURISDICTION as jurisdiction,
	t8.DISPLAY_VALUE as accessRegionRef,
	t1.ACCESS_REGION_ORIGIN as accessRegionOrigin,
	t1.MIN_DISTANCE as minDistance,
	t1.MAX_DISTANCE as maxDistance
from
	DISTANCE_BANDS t1
	inner join DISTANCE_BANDS_ID_REF t2 on t1.DISTANCE_BAND_ID = t2.DISTANCE_BAND_ID
	inner join DISTANCE_BANDS_ID_VALUES t3 on t2.DISTANCE_BAND_ID = t3.DISTANCE_BAND_ID
		and t3.LANGUAGE_CODE = {2}
	inner join AUT_INITIAL_KEY t4 on t1.INITIAL_AUT_ID = t4.INITIAL_AUT_ID
	inner join AUT_INITIAL_REF t5 on t4.INITIAL_AUT_ID = t5.INITIAL_AUT_ID
		and t5.SERVICE_VERSION_ID = {1}
	inner join AUT_INITIAL_VALUES t6 on t4.INITIAL_AUT_ID = t6.INITIAL_AUT_ID
		and t6.SERVICE_VERSION_ID = t5.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ACCESS_REGION_REF t7 on t1.ACCESS_REGION_ORIGIN = t7.ACCESS_REGION
	left outer join ACCESS_REGION_VALUES t8 on t7.ACCESS_REGION = t8.ACCESS_REGION
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
