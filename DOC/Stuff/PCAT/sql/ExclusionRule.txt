select
	t1.EXCLUSION_RULE_ID as exclusionRuleId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	EXCLUSION_RULE_KEY t1
	inner join EXCLUSION_RULE_REF t2 on t1.EXCLUSION_RULE_ID = t2.EXCLUSION_RULE_ID
	inner join EXCLUSION_RULE_VALUES t3 on t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
		and t2.EXCLUSION_RULE_ID = t3.EXCLUSION_RULE_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
