select
	t1.ADDRESS_FORMAT_ID as addressFormatId,
	t1.TAG as tag,
	t1.IS_INTERNAL as isInternal,
	t1.IS_DEFAULT as isDefault,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	IGEN_ADDRESS_FORMAT_REF t1
	inner join IGEN_ADDRESS_FORMAT_VALUES t2 on t1.ADDRESS_FORMAT_ID = t2.ADDRESS_FORMAT_ID
where
	t2.LANGUAGE_CODE = {2}
