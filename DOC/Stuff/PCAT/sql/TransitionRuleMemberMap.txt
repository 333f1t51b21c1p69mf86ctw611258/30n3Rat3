select
	t1.TRANSITION_RULE_MAP_ID as transitionRuleMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as transitionRuleKey,
	t1.TRANSITION_RULE_ID as transitionRuleId,
	t1.TRANSITION_NUMBER as transitionNumber,
	t7.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId,
	t10.DISPLAY_VALUE as bundleKey,
	t1.BUNDLE_ID as bundleId
from
	TRANSITION_RULE_MEMBER_MAP t1
	inner join TRANSITION_RULE_KEY t2 on t1.TRANSITION_RULE_ID = t2.TRANSITION_RULE_ID
	inner join TRANSITION_RULE_REF t3 on t2.TRANSITION_RULE_ID = t3.TRANSITION_RULE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TRANSITION_RULE_VALUES t4 on t2.TRANSITION_RULE_ID = t4.TRANSITION_RULE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join OFFER_KEY t5 on t1.OFFER_ID = t5.OFFER_ID
	left outer join OFFER_REF t6 on t5.OFFER_ID = t6.OFFER_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join OFFER_VALUES t7 on t5.OFFER_ID = t7.OFFER_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join BUNDLE_KEY t8 on t1.BUNDLE_ID = t8.BUNDLE_ID
	left outer join BUNDLE_REF t9 on t8.BUNDLE_ID = t9.BUNDLE_ID
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join BUNDLE_VALUES t10 on t8.BUNDLE_ID = t10.BUNDLE_ID
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
