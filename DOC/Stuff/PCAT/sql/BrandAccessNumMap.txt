select
	t4.DISPLAY_VALUE as brandKey,
	t1.BRANDING_ID as brandingId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.ACCESS_NUMBER as accessNumber
from
	BRAND_ACCESS_NUM_MAP t1
	inner join BRAND_KEY t2 on t1.BRANDING_ID = t2.BRANDING_ID
	inner join BRAND_REF t3 on t2.BRANDING_ID = t3.BRANDING_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join BRAND_VALUES t4 on t2.BRANDING_ID = t4.BRANDING_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join ACCESS_NUMBERS_KEY t5 on t1.ACCESS_NUMBER = t5.ACCESS_NUMBER
where
	t1.SERVICE_VERSION_ID = {1}
