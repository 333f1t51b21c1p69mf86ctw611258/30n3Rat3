select
	t4.DISPLAY_VALUE as dealerKey,
	t1.DEALER_ID as dealerId,
	t7.DISPLAY_VALUE as dealerGroupKey,
	t1.DEALER_GROUP_ID as dealerGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	DEALER_GROUP_MAP t1
	inner join DEALER_KEY t2 on t1.DEALER_ID = t2.DEALER_ID
	inner join DEALER_REF t3 on t2.DEALER_ID = t3.DEALER_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join DEALER_VALUES t4 on t2.DEALER_ID = t4.DEALER_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join DEALER_GROUP_KEY t5 on t1.DEALER_GROUP_ID = t5.DEALER_GROUP_ID
	inner join DEALER_GROUP_REF t6 on t5.DEALER_GROUP_ID = t6.DEALER_GROUP_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join DEALER_GROUP_VALUES t7 on t5.DEALER_GROUP_ID = t7.DEALER_GROUP_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
