select
	t1.STATE_ID as stateId,
        t1.LEVEL_CODE as levelCode,
	t4.DISPLAY_VALUE as levelCodeString,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INITIAL_STATE as isInitialState,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	SUBSCRIBER_STATE_ID_REF t1
	inner join SUBSCRIBER_STATE_ID_VALUES t2 on t1.STATE_ID = t2.STATE_ID
		and t1.LEVEL_CODE = t2.LEVEL_CODE
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''LEVEL_CODE'')
		and t3.value = t1.LEVEL_CODE
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
