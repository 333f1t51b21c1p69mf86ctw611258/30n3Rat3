select
	t1.PARAM_ID as paramId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as diamFlexsleServiceKey,
	t1.SLE_SERVICE_ID as sleServiceId,
	t5.DISPLAY_VALUE as paramTagIdDisplayValue,
	t1.PARAM_VALUE as paramValue,
	t8.DISPLAY_VALUE as paramDataType
from
	DIAM_FLEXSLE_PARAMETER t1
	inner join DIAM_FLEXSLE_SERVICE_KEY t2 on t1.SLE_SERVICE_ID = t2.SLE_SERVICE_ID
	inner join DIAM_FLEXSLE_SERVICE_REF t3 on t2.SLE_SERVICE_ID = t3.SLE_SERVICE_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join DIAM_FLEXSLE_SERVICE_VALUES t4 on t2.SLE_SERVICE_ID = t4.SLE_SERVICE_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join DIAM_FLEXSLE_PARAMETER_LABEL t5 on t1.PARAM_TAG_ID = t5.PARAM_TAG_ID
		and t5.SLE_SERVICE_ID = t1.SLE_SERVICE_ID
		and t5.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t5.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t7 on t7.table_name = ''DIAM_FLEXSLE_PARAMETER''
		and t7.field_name = lower(''PARAM_DATA_TYPE'')
		and t7.integer_value = t1.PARAM_DATA_TYPE
	left outer join GUI_INDICATOR_VALUES t8 on t8.table_name = t7.table_name
		and t8.field_name = t7.field_name
		and t8.integer_value = t7.integer_value
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
