select
	t1.VANITY_CODE as vanityCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description,
        t2.SHORT_DISPLAY as shortDisplay
from
	INVS_VANITY_REF t1
	inner join INVS_VANITY_VALUES t2 on t1.VANITY_CODE = t2.VANITY_CODE
where
	t2.LANGUAGE_CODE = {2}
