select
	t3.DISPLAY_VALUE as invsStockPropConfigRef,
	t1.STOCK_PROPERTY_CONFIG_ID as stockPropertyConfigId,
	t5.DISPLAY_VALUE as invsSecurityTypeRef,
	t1.SECURITY_TYPE_ID as securityTypeId
from
	INVS_STOCK_PROP_SECURITY_LINK t1
	inner join INVS_STOCK_PROP_CONFIG_REF t2 on t1.STOCK_PROPERTY_CONFIG_ID = t2.STOCK_PROPERTY_CONFIG_ID
	inner join INVS_STOCK_PROP_CONFIG_VALUES t3 on t2.STOCK_PROPERTY_CONFIG_ID = t3.STOCK_PROPERTY_CONFIG_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_SECURITY_TYPE_REF t4 on t1.SECURITY_TYPE_ID = t4.SECURITY_TYPE_ID
	inner join INVS_SECURITY_TYPE_VALUES t5 on t4.SECURITY_TYPE_ID = t5.SECURITY_TYPE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
