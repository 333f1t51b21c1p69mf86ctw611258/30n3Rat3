select
	t1.DISCOUNT_THRESHOLD_TIER_ID as discountThresholdTierId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	DISCOUNT_THRESH_TIER_ID_KEY t1
	inner join DISCOUNT_THRESH_TIER_ID_REF t2 on t1.DISCOUNT_THRESHOLD_TIER_ID = t2.DISCOUNT_THRESHOLD_TIER_ID
	inner join DISCOUNT_THRESH_TIER_ID_VALUES t3 on t2.DISCOUNT_THRESHOLD_TIER_ID = t3.DISCOUNT_THRESHOLD_TIER_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
