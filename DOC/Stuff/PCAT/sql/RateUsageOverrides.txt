select
	t1.SEQNUM as seqnum,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as corridorPlanIdKey,
	t1.CORRIDOR_PLAN_ID as corridorPlanId,
	t7.DISPLAY_VALUE as autFinalKey,
	t1.AUT_ID as autId,
	t9.DISPLAY_VALUE as unitsIndicator,
	t1.USAGE_PLAN_ID as usagePlanId,
	t1.RATE_CLASS as rateClass,
	t11.DISPLAY_VALUE as equipTypeCodeRef,
	t1.EQUIP_TYPE_CODE as equipTypeCode,
	t13.DISPLAY_VALUE as equipClassCodeRef,
	t1.EQUIP_CLASS_CODE as equipClassCode,
	t15.DISPLAY_VALUE as classOfServiceCodeRef,
	t1.CLASS_OF_SERVICE_CODE as classOfServiceCode,
	t1.PROVIDER_CLASS as providerClass,
	t1.BILL_CLASS as billClass,
	t1.JURISDICTION as jurisdiction,
	t1.CURRENCY_CODE as currencyCode,
	t1.TIME_TYPE_ID as timeTypeId,
	t1.INCREMENTAL_BANDS as incrementalBands,
	t1.FIXED_CHARGE_AMT as fixedChargeAmt,
	t1.ADD_FIXED_AMT as addFixedAmt,
	t1.ADD_UNIT_RATE as addUnitRate,
	t1.CREATE_DT as createDt,
	t1.CHG_DT as chgDt,
	t1.CHG_WHO as chgWho,
	t18.SERVICE_VERSION_ID as serviceVersionId,
	t19.DISPLAY_VALUE as unitsTypeKey,
	t1.RATE_UNIT_TYPE as rateUnitType,
	t1.DISTANCE_BAND_ID as distanceBandId,
	t1.ADD_IMPLIED_DECIMAL as addImpliedDecimal,
	t1.OFFER_ID as offerId,
	t23.DISPLAY_VALUE as unitsTypeKeyByBillingUnitType,
	t1.BILLING_UNIT_TYPE as billingUnitType
from
	RATE_USAGE_OVERRIDES t1
	inner join CORRIDOR_PLAN_ID_KEY t2 on t1.CORRIDOR_PLAN_ID = t2.CORRIDOR_PLAN_ID
	inner join CORRIDOR_PLAN_ID_REF t3 on t2.CORRIDOR_PLAN_ID = t3.CORRIDOR_PLAN_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join CORRIDOR_PLAN_ID_VALUES t4 on t2.CORRIDOR_PLAN_ID = t4.CORRIDOR_PLAN_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join AUT_FINAL_KEY t5 on t1.AUT_ID = t5.AUT_ID
	inner join AUT_FINAL_REF t6 on t5.AUT_ID = t6.AUT_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join AUT_FINAL_VALUES t7 on t5.AUT_ID = t7.AUT_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t8 on t8.enumeration_key = lower(''UNITS_INDICATOR'')
		and t8.value = t1.UNITS_INDICATOR
	left outer join GENERIC_ENUMERATION_VALUES t9 on t9.enumeration_key = t8.enumeration_key
		and t9.value = t8.value
		and t9.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join EQUIP_TYPE_CODE_REF t10 on t1.EQUIP_TYPE_CODE = t10.EQUIP_TYPE_CODE
	inner join EQUIP_TYPE_CODE_VALUES t11 on t10.EQUIP_TYPE_CODE = t11.EQUIP_TYPE_CODE
		and t11.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join EQUIP_CLASS_CODE_REF t12 on t1.EQUIP_CLASS_CODE = t12.EQUIP_CLASS_CODE
	inner join EQUIP_CLASS_CODE_VALUES t13 on t12.EQUIP_CLASS_CODE = t13.EQUIP_CLASS_CODE
		and t13.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join CLASS_OF_SERVICE_CODE_REF t14 on t1.CLASS_OF_SERVICE_CODE = t14.CLASS_OF_SERVICE_CODE
	inner join CLASS_OF_SERVICE_CODE_VALUES t15 on t14.CLASS_OF_SERVICE_CODE = t15.CLASS_OF_SERVICE_CODE
		and t15.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join RATE_USAGE_IMPLIED_DECIMALS t16 on t1.RESELLER_VERSION_ID = t16.RESELLER_VERSION_ID
		and t1.CURRENCY_CODE = t16.CURRENCY_CODE
		and t1.ADD_IMPLIED_DECIMAL = t16.ADD_IMPLIED_DECIMAL
	inner join UNITS_TYPE_KEY t17 on t1.RATE_UNIT_TYPE = t17.UNIT_TYPE
	inner join UNITS_TYPE_REF t18 on t17.UNIT_TYPE = t18.UNIT_TYPE
		and t18.SERVICE_VERSION_ID = {1}
	inner join UNITS_TYPE_VALUES t19 on t17.UNIT_TYPE = t19.UNIT_TYPE
		and t19.SERVICE_VERSION_ID = t18.SERVICE_VERSION_ID
		and t19.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join RATE_USAGE_IMPLIED_DECIMALS t20 on t1.RESELLER_VERSION_ID = t20.RESELLER_VERSION_ID
		and t1.CURRENCY_CODE = t20.CURRENCY_CODE
		and t1.ADD_IMPLIED_DECIMAL = t20.ADD_IMPLIED_DECIMAL
	left outer join UNITS_TYPE_KEY t21 on t1.BILLING_UNIT_TYPE = t21.UNIT_TYPE
	left outer join UNITS_TYPE_REF t22 on t21.UNIT_TYPE = t22.UNIT_TYPE
		and t22.SERVICE_VERSION_ID = t18.SERVICE_VERSION_ID
	left outer join UNITS_TYPE_VALUES t23 on t21.UNIT_TYPE = t23.UNIT_TYPE
		and t23.SERVICE_VERSION_ID = t18.SERVICE_VERSION_ID
		and t23.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
