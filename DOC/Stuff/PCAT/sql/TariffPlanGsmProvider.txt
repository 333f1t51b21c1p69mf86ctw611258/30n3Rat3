select
	t1.TARIFF_PLAN_GSM_PROVIDER_ID as tariffPlanGsmProviderId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as tariffPlanKey,
	t1.TARIFF_PLAN_ID as tariffPlanId,
	t6.DISPLAY_VALUE as tapGsmProviderIdRef,
	t1.TAP_GSM_PROVIDER_ID as tapGsmProviderId,
	t1.MARKUP_PERCENT as markupPercent,
	t8.DISPLAY_VALUE as chargeCodeRef,
	t1.CHARGE_CODE as chargeCode
from
	TARIFF_PLAN_GSM_PROVIDER t1
	inner join TARIFF_PLAN_KEY t2 on t1.TARIFF_PLAN_ID = t2.TARIFF_PLAN_ID
	inner join TARIFF_PLAN_REF t3 on t2.TARIFF_PLAN_ID = t3.TARIFF_PLAN_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TARIFF_PLAN_VALUES t4 on t2.TARIFF_PLAN_ID = t4.TARIFF_PLAN_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join TAP_GSM_PROVIDER_ID_REF t5 on t1.TAP_GSM_PROVIDER_ID = t5.TAP_GSM_PROVIDER_ID
	inner join TAP_GSM_PROVIDER_ID_VALUES t6 on t5.TAP_GSM_PROV_ID = t6.TAP_GSM_PROV_ID
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join CHARGE_CODE_REF t7 on t1.CHARGE_CODE = t7.CHARGE_CODE
	left outer join CHARGE_CODE_VALUES t8 on t7.CHARGE_CODE = t8.CHARGE_CODE
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
