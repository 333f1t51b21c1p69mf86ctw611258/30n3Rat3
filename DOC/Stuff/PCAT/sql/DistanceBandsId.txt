select
	t1.DISTANCE_BAND_ID as distanceBandId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	DISTANCE_BANDS_ID_REF t1
	inner join DISTANCE_BANDS_ID_VALUES t2 on t1.DISTANCE_BAND_ID = t2.DISTANCE_BAND_ID
where
	t2.LANGUAGE_CODE = {2}
