select
	t1.CUG_CODE as cugCode,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.MIN_SHORT_DIAL_LENGTH as minShortDialLength,
	t2.MAX_SHORT_DIAL_LENGTH as maxShortDialLength,
	t2.DEFAULT_NDC as defaultNdc,
	t2.CUG_CALL_RESTRICTION as cugCallRestriction,
	t2.MAX_ONNET_DURATION as maxOnnetDuration,
	t2.MAX_OFFNET_DURATION as maxOffnetDuration,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	CUG_DEFINITION_KEY t1
	inner join CUG_DEFINITION_REF t2 on t1.CUG_CODE = t2.CUG_CODE
	inner join CUG_DEFINITION_VALUES t3 on t2.CUG_CODE = t3.CUG_CODE
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
