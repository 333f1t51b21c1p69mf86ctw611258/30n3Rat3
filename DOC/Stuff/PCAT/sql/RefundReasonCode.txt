select
	t1.REFUND_REASON_CODE as refundReasonCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	REFUND_REASON_CODE_REF t1
	inner join REFUND_REASON_CODE_VALUES t2 on t1.REFUND_REASON_CODE = t2.REFUND_REASON_CODE
where
	t2.LANGUAGE_CODE = {2}
