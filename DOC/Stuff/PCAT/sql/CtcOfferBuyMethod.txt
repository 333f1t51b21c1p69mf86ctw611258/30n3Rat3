select
	t1.OFFER_BUY_ID as offerBuyId,
	t4.DISPLAY_VALUE as ctcOrderMethodRef,
	t1.ORDER_METHOD_ID as orderMethodId
from
	CTC_OFFER_BUY_METHOD t1
	inner join CTC_OFFER_BUY t2 on t1.OFFER_BUY_ID = t2.OFFER_BUY_ID
	inner join CTC_ORDER_METHOD_REF t3 on t1.ORDER_METHOD_ID = t3.ORDER_METHOD_ID
	inner join CTC_ORDER_METHOD_VALUES t4 on t3.ORDER_METHOD_ID = t4.ORDER_METHOD_ID
		and t4.LANGUAGE_CODE = {2}
