select
	t1.NOTE_REASON_ID as noteReasonId,
	t1.IS_SELECTABLE as isSelectable,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	NOTE_REASON_REF t1
	inner join NOTE_REASON_VALUES t2 on t1.NOTE_REASON_ID = t2.NOTE_REASON_ID
where
	t2.LANGUAGE_CODE = {2}
