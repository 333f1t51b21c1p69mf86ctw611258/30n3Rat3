select
	t1.CELL_ID as cellId,
	t1.CELL_NAME as cellName,
	t1.CIBER_SID_BID as ciberSidBid,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CELL_ID_REF t1
	inner join CELL_ID_VALUES t2 on t1.CELL_ID = t2.CELL_ID
where
	t2.LANGUAGE_CODE = {2}
