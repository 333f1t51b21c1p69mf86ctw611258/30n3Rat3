select
	t1.RT_PAYMENT_RESP_ID as rtPaymentRespId,
	t1.EPG_TYPE as epgType,
	t1.REPLY_FLAG as replyFlag,
	t6.DISPLAY_VALUE as rtPaymentRespKey,
	t2.EPG_TYPE as epgType,
	t9.DISPLAY_VALUE as rtPaymentRespKeyByEpgType,
	t2.REPLY_FLAG as replyFlag,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.IS_DEFAULT as isDefault,
	t2.IS_INTERNAL as isInternal,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	RT_PAYMENT_RESP_KEY t1
	inner join RT_PAYMENT_RESP_REF t2 on t1.RT_PAYMENT_RESP_ID = t2.RT_PAYMENT_RESP_ID
	inner join RT_PAYMENT_RESP_VALUES t3 on t2.RT_PAYMENT_RESP_ID = t3.RT_PAYMENT_RESP_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	inner join RT_PAYMENT_RESP_KEY t4 on t2.EPG_TYPE = t4.EPG_TYPE
		and t2.REPLY_FLAG = t4.REPLY_FLAG
	inner join RT_PAYMENT_RESP_REF t5 on t4.EPG_TYPE = t5.EPG_TYPE
		and t4.REPLY_FLAG = t5.REPLY_FLAG
		and t5.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join RT_PAYMENT_RESP_VALUES t6 on t4.EPG_TYPE = t6.EPG_TYPE
		and t4.REPLY_FLAG = t6.REPLY_FLAG
		and t6.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join RT_PAYMENT_RESP_KEY t7 on t2.EPG_TYPE = t7.EPG_TYPE
		and t2.REPLY_FLAG = t7.REPLY_FLAG
	inner join RT_PAYMENT_RESP_REF t8 on t7.EPG_TYPE = t8.EPG_TYPE
		and t7.REPLY_FLAG = t8.REPLY_FLAG
		and t8.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
	inner join RT_PAYMENT_RESP_VALUES t9 on t7.EPG_TYPE = t9.EPG_TYPE
		and t7.REPLY_FLAG = t9.REPLY_FLAG
		and t9.SERVICE_VERSION_ID = t2.SERVICE_VERSION_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
