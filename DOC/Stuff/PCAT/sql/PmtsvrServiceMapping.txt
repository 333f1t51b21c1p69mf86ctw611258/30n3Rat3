select
	t1.PMTSVR_SERVICE_MAPPING_ID as pmtsvrServiceMappingId,
	t1.BEARER_CAPABILITY as bearerCapability,
	t1.DISCOUNT as discount,
	t1.EXTERNAL_UNIT_TYPE_ID as externalUnitTypeId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as autInitialKey,
    t5.EXTERNAL_UNIT_TYPE_DESC as externalUnitType,
	t1.AUT_ID as autId
from
	PMTSVR_SERVICE_MAPPING t1
	inner join AUT_INITIAL_KEY t2 on t1.AUT_ID = t2.INITIAL_AUT_ID
	inner join AUT_INITIAL_REF t3 on t2.INITIAL_AUT_ID = t3.INITIAL_AUT_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join AUT_INITIAL_VALUES t4 on t2.INITIAL_AUT_ID = t4.INITIAL_AUT_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
    inner join PS_EXTERNAL_UNIT_MAPPING t5 on t1.EXTERNAL_UNIT_TYPE_ID = t5.EXTERNAL_UNIT_TYPE_ID
                and t5.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
where
	t1.SERVICE_VERSION_ID = {1}
