select
	t1.MARKETING_CATEGORY_MAP_ID as marketingCategoryMapId,
	t1.PC_ENTITY_TYPE as pcEntityType,
	t1.PC_ENTITY_ID as pcEntityId,
	t4.DISPLAY_VALUE as marketingCategoryKey,
	t1.MARKETING_CATEGORY_ID as marketingCategoryId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	MARKETING_CATEGORY_MAP t1
	inner join MARKETING_CATEGORY_KEY t2 on t1.MARKETING_CATEGORY_ID = t2.MARKETING_CATEGORY_ID
	inner join MARKETING_CATEGORY_REF t3 on t2.MARKETING_CATEGORY_ID = t3.MARKETING_CATEGORY_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join MARKETING_CATEGORY_VALUES t4 on t2.MARKETING_CATEGORY_ID = t4.MARKETING_CATEGORY_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
