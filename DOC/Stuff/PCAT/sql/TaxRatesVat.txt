select
	t1.TAX_RATES_VAT_ID as taxRatesVatId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as taxPkgInstIdKey,
	t1.TAX_PKG_INST_ID as taxPkgInstId,
	t6.SERVICE_VERSION_ID as serviceVersionId,
	t6.ISO_CODE as currencyIsoCode,
	t6.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t7.DISPLAY_VALUE as rateCurrencyKey,
	t1.CURRENCY_CODE as currencyCode,
	t1.FIXED_AMT as fixedAmt,
	t10.DISPLAY_VALUE as taxClassKey,
	t1.TAX_CLASS as taxClass,
	t1.RATE as rate,
	t13.DISPLAY_VALUE as taxTypeCodeKey,
	t1.TAX_TYPE_CODE as taxTypeCode,
	t15.DISPLAY_VALUE as billingFrequencyRef,
	t1.BILLING_FREQUENCY as billingFrequency,
	t1.RANGE_TERMINUS as rangeTerminus
from
	TAX_RATES_VAT t1
	inner join TAX_PKG_INST_ID_KEY t2 on t1.TAX_PKG_INST_ID = t2.TAX_PKG_INST_ID
	inner join TAX_PKG_INST_ID_REF t3 on t2.TAX_PKG_INST_ID = t3.TAX_PKG_INST_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TAX_PKG_INST_ID_VALUES t4 on t2.TAX_PKG_INST_ID = t4.TAX_PKG_INST_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join RATE_CURRENCY_KEY t5 on t1.CURRENCY_CODE = t5.CURRENCY_CODE
	left outer join RATE_CURRENCY_REF t6 on t5.CURRENCY_CODE = t6.CURRENCY_CODE
		and t6.SERVICE_VERSION_ID = {1}
	left outer join RATE_CURRENCY_VALUES t7 on t5.CURRENCY_CODE = t7.CURRENCY_CODE
		and t7.SERVICE_VERSION_ID = {1}
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join TAX_CLASS_KEY t8 on t1.TAX_CLASS = t8.TAX_CLASS
	inner join TAX_CLASS_REF t9 on t8.TAX_CLASS = t9.TAX_CLASS
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TAX_CLASS_VALUES t10 on t8.TAX_CLASS = t10.TAX_CLASS
		and t10.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join TAX_TYPE_CODE_KEY t11 on t1.TAX_TYPE_CODE = t11.TAX_TYPE_CODE
	inner join TAX_TYPE_CODE_REF t12 on t11.TAX_TYPE_CODE = t12.TAX_TYPE_CODE
		and t12.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join TAX_TYPE_CODE_VALUES t13 on t11.TAX_TYPE_CODE = t13.TAX_TYPE_CODE
		and t13.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t13.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join BILLING_FREQUENCY_REF t14 on t1.BILLING_FREQUENCY = t14.BILLING_FREQUENCY
	left outer join BILLING_FREQUENCY_VALUES t15 on t14.BILLING_FREQUENCY = t15.BILLING_FREQUENCY
		and t15.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
