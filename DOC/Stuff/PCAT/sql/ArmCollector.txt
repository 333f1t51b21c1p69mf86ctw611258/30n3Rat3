select
	t1.COLLECTOR_ID as collectorId,
	t1.GROUP_ID as groupId,
	t1.FIRST_NAME as firstName,
	t1.LAST_NAME as lastName,
	t1.COMPANY as company,
	t1.ADDRESS1 as address1,
	t1.ADDRESS2 as address2,
	t1.ADDRESS3 as address3,
	t1.CITY as city,
	t1.STATE as state,
	t1.ZIP as zip,
	t1.PHONE as phone,
	t1.ACTIVE_FLAG as activeFlag,
	t1.USERNAME as username
from
	ARM_COLLECTOR t1
