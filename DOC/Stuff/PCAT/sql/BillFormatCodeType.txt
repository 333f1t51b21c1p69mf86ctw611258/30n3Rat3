select
	t1.CODE_TYPE as codeType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_FORMAT_CODE_TYPE_REF t1
	inner join BILL_FORMAT_CODE_TYPE_VALUES t2 on t1.CODE_TYPE = t2.CODE_TYPE
where
	t2.LANGUAGE_CODE = {2}
