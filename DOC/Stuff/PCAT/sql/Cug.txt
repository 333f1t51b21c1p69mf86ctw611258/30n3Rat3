select
	t1.CUG_CODE as cugCode,
	t4.SERVICE_VERSION_ID as serviceVersionId,
	t5.DISPLAY_VALUE as cugAccessCodeKey,
	t1.ACCESS_CODE as accessCode,
	t1.MIN_SHORT_DIAL_LENGTH as minShortDialLength,
	t1.MAX_SHORT_DIAL_LENGTH as maxShortDialLength,
	t1.DEFAULT_NDC as defaultNdc,
	t1.PASSWORD as password,
	t1.CALL_RESTRICTION as callRestriction,
	t1.MAX_ONNET_DURATION as maxOnnetDuration,
	t1.MAX_OFFNET_DURATION as maxOffnetDuration,
	t7.DISPLAY_VALUE as cugRef,
	t1.SUPER_CUG_CODE as superCugCode,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	CUG_REF t1
	inner join CUG_VALUES t2 on t1.CUG_CODE = t2.CUG_CODE
	inner join CUG_ACCESS_CODE_KEY t3 on t1.ACCESS_CODE = t3.ACCESS_CODE
	inner join CUG_ACCESS_CODE_REF t4 on t3.ACCESS_CODE = t4.ACCESS_CODE
		and t4.SERVICE_VERSION_ID = {1}
	inner join CUG_ACCESS_CODE_VALUES t5 on t3.ACCESS_CODE = t5.ACCESS_CODE
		and t5.SERVICE_VERSION_ID = t4.SERVICE_VERSION_ID
		and t5.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join CUG_REF t6 on t1.SUPER_CUG_CODE = t6.CUG_CODE
	left outer join CUG_VALUES t7 on t6.CUG_CODE = t7.CUG_CODE
		and t7.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
