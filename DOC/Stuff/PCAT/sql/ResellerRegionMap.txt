select
	t4.DISPLAY_VALUE as resellerKey,
	t1.RESELLER_ID as resellerId,
	t7.DISPLAY_VALUE as regionKey,
	t1.REGION_ID as regionId,
	t1.SERVICE_VERSION_ID as serviceVersionId
from
	RESELLER_REGION_MAP t1
	inner join RESELLER_KEY t2 on t1.RESELLER_ID = t2.RESELLER_ID
	inner join RESELLER_REF t3 on t2.RESELLER_ID = t3.RESELLER_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join RESELLER_VALUES t4 on t2.RESELLER_ID = t4.RESELLER_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join REGION_KEY t5 on t1.REGION_ID = t5.REGION_ID
	inner join REGION_REF t6 on t5.REGION_ID = t6.REGION_ID
		and t6.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join REGION_VALUES t7 on t5.REGION_ID = t7.REGION_ID
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.SERVICE_VERSION_ID = {1}
