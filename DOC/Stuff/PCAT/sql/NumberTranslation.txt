select
	t1.NUM_TRANS_ID as numTransId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t2.PREFIX as prefix,
	t2.NORMALIZATION_TYPE as normalizationType,
	t2.PRIORITY_CODE as priorityCode,
	t5.DISPLAY_VALUE as actionCode,
	t2.PATTERN as pattern,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	NUMBER_TRANSLATION_KEY t1
	inner join NUMBER_TRANSLATION_REF t2 on t1.NUM_TRANS_ID = t2.NUM_TRANS_ID
	inner join NUMBER_TRANSLATION_VALUES t3 on t2.NUM_TRANS_ID = t3.NUM_TRANS_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join GUI_INDICATOR_REF t4 on t4.table_name = ''NUMBER_TRANSLATION_REF''
		and t4.field_name = lower(''ACTION_CODE'')
		and t4.integer_value = t2.ACTION_CODE
	left outer join GUI_INDICATOR_VALUES t5 on t5.table_name = t4.table_name
		and t5.field_name = t4.field_name
		and t5.integer_value = t4.integer_value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
