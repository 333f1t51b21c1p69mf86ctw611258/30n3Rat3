select
	t1.ARM_PLUGIN_ID as armPluginId,
	t1.PROCESS_ID as processId,
	t1.PROCESS_USER_DATA_XSD as processUserDataXsd,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARM_PLUGIN_PROCESS_REF t1
	inner join ARM_PLUGIN_PROCESS_VALUES t2 on t1.ARM_PLUGIN_ID = t2.ARM_PLUGIN_ID
		and t1.PROCESS_ID = t2.PROCESS_ID
where
	t2.LANGUAGE_CODE = {2}
