select
	t4.DISPLAY_VALUE as marketSegmentIdKey,
	t1.MARKET_SEG_ID as marketSegId,
	t7.DISPLAY_VALUE as offerKey,
	t1.OFFER_ID as offerId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	OFFER_MARKET_SEGMENT_MAP t1
	inner join MARKET_SEGMENT_ID_KEY t2 on t1.MARKET_SEG_ID = t2.MARKET_SEG_ID
	inner join MARKET_SEGMENT_ID_REF t3 on t2.MARKET_SEG_ID = t3.MARKET_SEG_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join MARKET_SEGMENT_ID_VALUES t4 on t2.MARKET_SEG_ID = t4.MARKET_SEG_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join OFFER_KEY t5 on t1.OFFER_ID = t5.OFFER_ID
	inner join OFFER_REF t6 on t5.OFFER_ID = t6.OFFER_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join OFFER_VALUES t7 on t5.OFFER_ID = t7.OFFER_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
