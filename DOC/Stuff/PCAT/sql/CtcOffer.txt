select
	t1.OFFER_ID as offerId,
	t1.P_OFFER_ID as pOfferId,
	t1.VERSION as version,
	t1.EXTERNAL_ID as externalId,
	t3.DISPLAY_VALUE as ctcScheduleRef,
	t1.SCHEDULE_ID as scheduleId,
	t5.DISPLAY_VALUE as ctcOfferCategoryRef,
	t1.PRIMARY_CATEGORY_ID as primaryCategoryId,
	t7.DISPLAY_VALUE as ctcOfferCategoryRefBySecondary,
	t1.SECONDARY_CATEGORY_ID as secondaryCategoryId,
	t9.DISPLAY_VALUE as ctcContentTypeRef,
	t1.CONTENT_TYPE_ID as contentTypeId,
	t1.FF_ID as ffId,
	t1.START_DT as startDt,
	t1.END_DT as endDt,
	t1.DISPLAY_START_DT as displayStartDt,
	t1.LAST_CANCEL_DT as lastCancelDt,
	t1.ADULT_FG as adultFg,
	t1.ITEM_COUNT as itemCount,
	t1.EVENT_COUNT as eventCount,
	t1.IS_DELETED as isDeleted,
	t1.TAPING_FG as tapingFg,
	t1.CHG_CODE as chgCode,
	t1.CHG_DT as chgDt
from
	CTC_OFFER t1
	inner join CTC_SCHEDULE_REF t2 on t1.SCHEDULE_ID = t2.SCHEDULE_ID
	inner join CTC_SCHEDULE_VALUES t3 on t2.SCHEDULE_ID = t3.SCHEDULE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join CTC_OFFER_CATEGORY_REF t4 on t1.PRIMARY_CATEGORY_ID = t4.OFFER_CATEGORY_ID
	inner join CTC_OFFER_CATEGORY_VALUES t5 on t4.OFFER_CATEGORY_ID = t5.OFFER_CATEGORY_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join CTC_OFFER_CATEGORY_REF t6 on t1.SECONDARY_CATEGORY_ID = t6.OFFER_CATEGORY_ID
	left outer join CTC_OFFER_CATEGORY_VALUES t7 on t6.OFFER_CATEGORY_ID = t7.OFFER_CATEGORY_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join CTC_CONTENT_TYPE_REF t8 on t1.CONTENT_TYPE_ID = t8.CONTENT_TYPE_ID
	inner join CTC_CONTENT_TYPE_VALUES t9 on t8.CONTENT_TYPE_ID = t9.CONTENT_TYPE_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join CTC_FF t10 on t1.FF_ID = t10.FF_ID
