select
	t1.EXT_EVENT_ID as extEventId,
	t1.EXT_EVENT_TYPE as extEventType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARM_EXT_EVENT_REF t1
	inner join ARM_EXT_EVENT_VALUES t2 on t1.EXT_EVENT_ID = t2.EXT_EVENT_ID
		and t1.EXT_EVENT_TYPE = t2.EXT_EVENT_TYPE
where
	t2.LANGUAGE_CODE = {2}
