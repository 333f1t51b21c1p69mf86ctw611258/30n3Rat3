select
	t1.RATE_CENTER_ID as rateCenterId,
	t4.DISPLAY_VALUE as invsGeographicRegionRef,
	t1.GEOGRAPHIC_REGION_ID as geographicRegionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_RATE_CENTER_REF t1
	inner join INVS_RATE_CENTER_VALUES t2 on t1.RATE_CENTER_ID = t2.RATE_CENTER_ID
	inner join INVS_GEOGRAPHIC_REGION_REF t3 on t1.GEOGRAPHIC_REGION_ID = t3.GEOGRAPHIC_REGION_ID
	inner join INVS_GEOGRAPHIC_REGION_VALUES t4 on t3.GEOGRAPHIC_REGION_ID = t4.GEOGRAPHIC_REGION_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
