select
	t1.PAPER_SIZE_ID as paperSizeId,
	t1.TEST_FLAG as testFlag,
	t1.TAG as tag,
	t1.HEIGHT as height,
	t1.WIDTH as width,
	t1.LINES as lines,
	t1.COLS as cols,
	t1.LINES_LAND as linesLand,
	t1.COLS_LAND as colsLand,
	t4.DISPLAY_VALUE as hpControl,
	t6.DISPLAY_VALUE as hpPaperBin,
	t8.DISPLAY_VALUE as afpPaperBin,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	PAPER_SIZE_REF t1
	inner join PAPER_SIZE_VALUES t2 on t1.PAPER_SIZE_ID = t2.PAPER_SIZE_ID
		and t1.TEST_FLAG = t2.TEST_FLAG
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''HP_CONTROL'')
		and t3.value = t1.HP_CONTROL
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t5 on t5.enumeration_key = lower(''HP_PAPER_BIN'')
		and t5.value = t1.HP_PAPER_BIN
	left outer join GENERIC_ENUMERATION_VALUES t6 on t6.enumeration_key = t5.enumeration_key
		and t6.value = t5.value
		and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t7 on t7.enumeration_key = lower(''AFP_PAPER_BIN'')
		and t7.value = t1.AFP_PAPER_BIN
	left outer join GENERIC_ENUMERATION_VALUES t8 on t8.enumeration_key = t7.enumeration_key
		and t8.value = t7.value
		and t8.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
