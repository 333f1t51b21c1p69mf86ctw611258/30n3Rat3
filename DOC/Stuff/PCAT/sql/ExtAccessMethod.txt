select
	t1.ACCESS_METHOD as accessMethod,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	EXT_ACCESS_METHOD_REF t1
	inner join EXT_ACCESS_METHOD_VALUES t2 on t1.ACCESS_METHOD = t2.ACCESS_METHOD
where
	t2.LANGUAGE_CODE = {2}
