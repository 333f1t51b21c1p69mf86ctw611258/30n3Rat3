select
	t1.ADDRESS_TYPE_ID as addressTypeId,
	t1.IS_FREEFORM as isFreeform,
	t1.ADDRESS_LINE1_FORMAT_STRING as addressLine1FormatString,
	t1.ADDRESS_LINE2_FORMAT_STRING as addressLine2FormatString,
	t1.ADDRESS_LINE3_FORMAT_STRING as addressLine3FormatString,
	t1.ADDRESS_LINE4_FORMAT_STRING as addressLine4FormatString,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	ADDRESS_TYPE_REF t1
	inner join ADDRESS_TYPE_VALUES t2 on t1.ADDRESS_TYPE_ID = t2.ADDRESS_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
