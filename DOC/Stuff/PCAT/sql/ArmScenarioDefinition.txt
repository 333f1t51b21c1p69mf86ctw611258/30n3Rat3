select
	t1.SCENARIO_ID as scenarioId,
	t1.EVENT_NO as eventNo,
	t3.DISPLAY_VALUE as eventCategory,
	t1.ACTION as action,
	t1.DAYS_FROM_PREV_EVENT as daysFromPrevEvent,
	t5.DISPLAY_VALUE as armStatusRef,
	t1.COLLECTION_STATUS as collectionStatus,
	t1.SCENARIO_CLASS_MEMBERSHIP as scenarioClassMembership,
	t1.ACTION_TYPE as actionType,
	t1.EXCLUSIVE_ACTION_GROUP as exclusiveActionGroup,
	t1.REVIEW_REQUIRED as reviewRequired
from
	ARM_SCENARIO_DEFINITION t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''EVENT_CATEGORY'')
		and t2.value = t1.EVENT_CATEGORY
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
	inner join ARM_STATUS_REF t4 on t1.COLLECTION_STATUS = t4.COLLECTION_STATUS
	inner join ARM_STATUS_VALUES t5 on t4.COLLECTION_STATUS = t5.COLLECTION_STATUS
		and t5.LANGUAGE_CODE = {2}
