select
	t1.RELEASE_CODE as releaseCode,
	t1.RELEASE_REASON as releaseReason,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	RELEASE_CODE_REF t1
	inner join RELEASE_CODE_VALUES t2 on t1.RELEASE_CODE = t2.RELEASE_CODE
where
	t2.LANGUAGE_CODE = {2}
