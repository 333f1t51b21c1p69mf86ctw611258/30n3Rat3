select
	t1.OFFER_NETWORK_NOTIF_MAP_ID as offerNetworkNotifMapId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as offerNotificationKey,
	t1.NOTIFICATION_ID as notificationId,
	t1.OFFER_ID as offerId,
	t1.PROVISIONING_ITEM_ID as provisioningItemId,
	t1.SERVICE_ITEM_ID as serviceItemId,
	t1.VALUE as value,
	t6.DISPLAY_VALUE as suppOfferState
from
	OFFER_NETWORK_NOTIF_MAP t1
	inner join OFFER_NOTIFICATION_KEY t2 on t1.NOTIFICATION_ID = t2.NOTIFICATION_ID
	inner join OFFER_NOTIFICATION_REF t3 on t2.NOTIFICATION_ID = t3.NOTIFICATION_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join OFFER_NOTIFICATION_VALUES t4 on t2.NOTIFICATION_ID = t4.NOTIFICATION_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join GUI_INDICATOR_REF t5 on t5.table_name = ''OFFER_NETWORK_NOTIF_MAP''
		and t5.field_name = lower(''SUPP_OFFER_STATE'')
		and t5.integer_value = t1.SUPP_OFFER_STATE
	left outer join GUI_INDICATOR_VALUES t6 on t6.table_name = t5.table_name
		and t6.field_name = t5.field_name
		and t6.integer_value = t5.integer_value
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
