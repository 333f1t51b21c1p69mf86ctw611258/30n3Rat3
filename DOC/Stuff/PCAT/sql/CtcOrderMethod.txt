select
	t1.ORDER_METHOD_ID as orderMethodId,
	t1.P_ORDER_METHOD_ID as pOrderMethodId,
	t1.RATING_KEY as ratingKey,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_ORDER_METHOD_REF t1
	inner join CTC_ORDER_METHOD_VALUES t2 on t1.ORDER_METHOD_ID = t2.ORDER_METHOD_ID
where
	t2.LANGUAGE_CODE = {2}
