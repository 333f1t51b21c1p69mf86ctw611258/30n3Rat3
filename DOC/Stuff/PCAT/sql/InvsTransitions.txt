select
	t1.TRANSITION_ID as transitionId,
	t1.TRANSITION_TYPE as transitionType,
	t1.ACTION_ID as actionId,
	t3.DISPLAY_VALUE as invsStatusRef,
	t1.ORIGIN_STATUS_ID as originStatusId,
	t5.DISPLAY_VALUE as invsStatusRefByDestStatusId,
	t1.DEST_STATUS_ID as destStatusId,
	t7.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t1.REASON_ID as reasonId,
	t1.TIME_THRESHOLD as timeThreshold,
	t1.SET_BLACKLIST_ID as setBlacklistId,
	t1.ACCOUNT_COMP as accountComp,
	t1.COPY_ACCOUNT_ID as copyAccountId,
	t1.CUST_OWNED as custOwned,
	t1.ALLOW_INDEPENDENT_TRANSITIONS as allowIndependentTransitions,
	t11.DISPLAY_VALUE as invsContainertypeRef,
	t1.TOP_LEVEL_CONTAINERTYPE_ID as topLevelContainertypeId
from
	INVS_TRANSITIONS t1
	inner join INVS_STATUS_REF t2 on t1.ORIGIN_STATUS_ID = t2.STATUS_ID
	inner join INVS_STATUS_VALUES t3 on t2.STATUS_ID = t3.STATUS_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_STATUS_REF t4 on t1.DEST_STATUS_ID = t4.STATUS_ID
	inner join INVS_STATUS_VALUES t5 on t4.STATUS_ID = t5.STATUS_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join INVS_TYPE_REF t6 on t1.INVENTORY_TYPE_ID = t6.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t7 on t6.INVENTORY_TYPE_ID = t7.INVENTORY_TYPE_ID
		and t7.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join INVS_REASON_REF t8 on t1.REASON_ID = t8.REASON_ID
	left outer join INVS_REASON_VALUES t9 on t8.REASON_ID = t9.REASON_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join INVS_CONTAINERTYPE_REF t10 on t1.TOP_LEVEL_CONTAINERTYPE_ID = t10.CONTAINERTYPE_ID
	left outer join INVS_CONTAINERTYPE_VALUES t11 on t10.CONTAINERTYPE_ID = t11.CONTAINERTYPE_ID
		and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
