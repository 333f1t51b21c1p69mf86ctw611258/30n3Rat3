select
	t3.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as autFinalKey,
	t1.AUT_ID as autId,
	t1.JURISDICTION as jurisdiction,
	t1.RATE_CLASS as rateClass,
	t1.EQUIP_TYPE_CODE as equipTypeCode,
	t1.EQUIP_CLASS_CODE as equipClassCode,
	t1.CLASS_OF_SERVICE_CODE as classOfServiceCode,
	t6.SERVICE_VERSION_ID as serviceVersionId,
	t7.DISPLAY_VALUE as unitsTypeKey,
	t1.BILLING_UNIT_TYPE as billingUnitType,
	t1.NO_BILL as noBill,
	t1.PRIORITY as priority,
	t1.INV_RATE_DEF_UNITS as invRateDefUnits,
	t1.INV_RATE_MAX_UNITS as invRateMaxUnits
from
	USAGE_TYPES_UNITS t1
	inner join AUT_FINAL_KEY t2 on t1.AUT_ID = t2.AUT_ID
	inner join AUT_FINAL_REF t3 on t2.AUT_ID = t3.AUT_ID
		and t3.RESELLER_VERSION_ID = {0}
	inner join AUT_FINAL_VALUES t4 on t2.AUT_ID = t4.AUT_ID
		and t4.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join UNITS_TYPE_KEY t5 on t1.BILLING_UNIT_TYPE = t5.UNIT_TYPE
	inner join UNITS_TYPE_REF t6 on t5.UNIT_TYPE = t6.UNIT_TYPE
		and t6.SERVICE_VERSION_ID = {1}
	inner join UNITS_TYPE_VALUES t7 on t5.UNIT_TYPE = t7.UNIT_TYPE
		and t7.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
