select
	t1.SERVICE_ID as serviceId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.ORIG_ROUTING_PREF as origRoutingPref,
	t2.TERM_ROUTING_PREF as termRoutingPref,
	t2.TRUNK_GROUP as trunkGroup,
	t2.PERIOD_OF_CDR as periodOfCdr,
	t2.COUNTER_RESET_PER as counterResetPer,
	t2.SAMPLE_COLLECT_PER as sampleCollectPer,
	t2.COUNTER_PURGE_PER as counterPurgePer,
	t2.SAMPLE_PURGE_PER as samplePurgePer,
	t2.EVENT_PURGE_PER as eventPurgePer,
	t2.AWAIT_ACTIVATION as awaitActivation,
	t2.AWAIT_RECHARGE as awaitRecharge,
	t2.INTER_CSR_NO_ACCT as interCsrNoAcct,
	t2.DEFAULT_CSR_NUMBER as defaultCsrNumber,
	t5.DISPLAY_VALUE as callDisc,
	t2.SN_LOCATION as snLocation,
	t2.OBSOLETE_CHARGE as obsoleteCharge,
	t2.ALLOW_SUPER_PLUS as allowSuperPlus,
	t2.ALLOW_CONTRACT_SERVICE as allowContractService,
	t2.IS_HASHING_IN_USE as isHashingInUse,
	t2.IS_HASHING_DONE as isHashingDone,
	t2.SERVICE_KEY as serviceKey,
	t2.FRC_IND as frcInd,
	t2.FRC_ERR_RESP as frcErrResp,
	t2.FRC_POLL_CYCLE as frcPollCycle,
	t2.FRC_RESP_TIMEOUT as frcRespTimeout,
	t2.FRC_SRV_AVAIL as frcSrvAvail,
	t2.UCARD_IN_USE as ucardInUse,
	t2.VM_ACCESS_PREFIX_FLAG as vmAccessPrefixFlag,
	t2.VM_ACCESS_PREFIX as vmAccessPrefix,
	t2.BILL_TPPS_CF_FLAG as billTppsCfFlag,
	t2.CALL_IF_SDP_DOWN as callIfSdpDown,
	t2.OFFER_IF_SDP_DOWN as offerIfSdpDown,
	t2.IS_VPN_ENABLED as isVpnEnabled,
	t2.VPN_COUNTRY_CODE as vpnCountryCode,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	RT_SERVICE_KEY t1
	inner join RT_SERVICE_REF t2 on t1.SERVICE_ID = t2.SERVICE_ID
	inner join RT_SERVICE_VALUES t3 on t2.SERVICE_ID = t3.SERVICE_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''CALL_DISC'')
		and t4.value = t2.CALL_DISC
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
