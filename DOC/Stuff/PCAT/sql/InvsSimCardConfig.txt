select
	t1.SIM_CARD_CONFIG_ID as simCardConfigId,
	t4.DISPLAY_VALUE as invsContainertypeRef,
	t1.CONTAINERTYPE_ID as containertypeId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_SIM_CARD_CONFIG_REF t1
	inner join INVS_SIM_CARD_CONFIG_VALUES t2 on t1.SIM_CARD_CONFIG_ID = t2.SIM_CARD_CONFIG_ID
	inner join INVS_CONTAINERTYPE_REF t3 on t1.CONTAINERTYPE_ID = t3.CONTAINERTYPE_ID
	inner join INVS_CONTAINERTYPE_VALUES t4 on t3.CONTAINERTYPE_ID = t4.CONTAINERTYPE_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
