select
	t1.FA_LIST_TEMPLATE_ID as faListTemplateId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t5.DISPLAY_VALUE as listType,
	t7.SERVICE_VERSION_ID as serviceVersionId,
	t8.DISPLAY_VALUE as locationHierarchyKey,
	t2.MAX_HIERARCHY_LEVEL_ID as maxHierarchyLevelId,
	t11.DISPLAY_VALUE as locationHierarchyKeyByMinHiera,
	t2.MIN_HIERARCHY_LEVEL_ID as minHierarchyLevelId,
	t2.USE_LOCATION_GROUP as useLocationGroup,
	t2.MAX_LIST_SIZE as maxListSize,
	t14.DISPLAY_VALUE as rcTermKey,
	t2.RC_TERM_ID as rcTermId,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	FA_LIST_TEMPLATE_KEY t1
	inner join FA_LIST_TEMPLATE_REF t2 on t1.FA_LIST_TEMPLATE_ID = t2.FA_LIST_TEMPLATE_ID
	inner join FA_LIST_TEMPLATE_VALUES t3 on t2.FA_LIST_TEMPLATE_ID = t3.FA_LIST_TEMPLATE_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join GUI_INDICATOR_REF t4 on t4.table_name = ''FA_LIST_TEMPLATE_REF''
		and t4.field_name = lower(''LIST_TYPE'')
		and t4.integer_value = t2.LIST_TYPE
	left outer join GUI_INDICATOR_VALUES t5 on t5.table_name = t4.table_name
		and t5.field_name = t4.field_name
		and t5.integer_value = t4.integer_value
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join LOCATION_HIERARCHY_KEY t6 on t2.MAX_HIERARCHY_LEVEL_ID = t6.HIERARCHY_LEVEL_ID
	inner join LOCATION_HIERARCHY_REF t7 on t6.HIERARCHY_LEVEL_ID = t7.HIERARCHY_LEVEL_ID
		and t7.SERVICE_VERSION_ID = {1}
	inner join LOCATION_HIERARCHY_VALUES t8 on t6.HIERARCHY_LEVEL_ID = t8.HIERARCHY_LEVEL_ID
		and t8.SERVICE_VERSION_ID = t7.SERVICE_VERSION_ID
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join LOCATION_HIERARCHY_KEY t9 on t2.MIN_HIERARCHY_LEVEL_ID = t9.HIERARCHY_LEVEL_ID
	left outer join LOCATION_HIERARCHY_REF t10 on t9.HIERARCHY_LEVEL_ID = t10.HIERARCHY_LEVEL_ID
		and t10.SERVICE_VERSION_ID = t7.SERVICE_VERSION_ID
	left outer join LOCATION_HIERARCHY_VALUES t11 on t9.HIERARCHY_LEVEL_ID = t11.HIERARCHY_LEVEL_ID
		and t11.SERVICE_VERSION_ID = t7.SERVICE_VERSION_ID
		and t11.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join RC_TERM_KEY t12 on t2.RC_TERM_ID = t12.RC_TERM_ID
	left outer join RC_TERM_REF t13 on t12.RC_TERM_ID = t13.RC_TERM_ID
		and t13.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join RC_TERM_VALUES t14 on t12.RC_TERM_ID = t14.RC_TERM_ID
		and t14.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t14.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
