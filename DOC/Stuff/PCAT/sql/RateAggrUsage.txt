select
	t3.DISPLAY_VALUE as aggrUsageDefinitionsRef,
	t1.AGGR_USAGE_ID as aggrUsageId,
	t5.DISPLAY_VALUE as billingFrequencyRef,
	t1.BILLING_FREQUENCY as billingFrequency,
	t1.LOWER_LIMIT as lowerLimit,
	t1.UPPER_LIMIT as upperLimit,
	t7.SERVICE_VERSION_ID as serviceVersionId,
	t7.ISO_CODE as currencyIsoCode,
	t7.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t8.DISPLAY_VALUE as rateCurrencyKey,
	t1.CURRENCY_CODE as currencyCode,
	t1.FIXED_RATE as fixedRate,
	t1.VARIABLE_RATE as variableRate,
	t1.VARIABLE_RATE_QUANTUM as variableRateQuantum,
	t1.ADD_IMPLIED_DECIMAL as addImpliedDecimal,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.CREATE_DT as createDt,
	t1.CHANGE_DT as changeDt,
	t1.CHANGE_WHO as changeWho
from
	RATE_AGGR_USAGE t1
	inner join AGGR_USAGE_DEFINITIONS_REF t2 on t1.AGGR_USAGE_ID = t2.AGGR_USAGE_ID
	inner join AGGR_USAGE_DEFINITIONS_VALUES t3 on t2.AGGR_USAGE_ID = t3.AGGR_USAGE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join BILLING_FREQUENCY_REF t4 on t1.BILLING_FREQUENCY = t4.BILLING_FREQUENCY
	inner join BILLING_FREQUENCY_VALUES t5 on t4.BILLING_FREQUENCY = t5.BILLING_FREQUENCY
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join RATE_CURRENCY_KEY t6 on t1.CURRENCY_CODE = t6.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t7 on t6.CURRENCY_CODE = t7.CURRENCY_CODE
		and t7.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t8 on t6.CURRENCY_CODE = t8.CURRENCY_CODE
		and t8.SERVICE_VERSION_ID = t7.SERVICE_VERSION_ID
		and t8.LANGUAGE_CODE = t3.LANGUAGE_CODE
