select
	t1.ITEM_ID as itemId,
	t3.DISPLAY_VALUE as ctcRatingTypeRef,
	t1.RATING_TYPE_ID as ratingTypeId,
	t5.DISPLAY_VALUE as ctcRatingRef,
	t1.RATING_ID as ratingId
from
	CTC_ITEM_RATING t1
	inner join CTC_RATING_TYPE_REF t2 on t1.RATING_TYPE_ID = t2.RATING_TYPE_ID
	inner join CTC_RATING_TYPE_VALUES t3 on t2.RATING_TYPE_ID = t3.RATING_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join CTC_RATING_REF t4 on t1.RATING_ID = t4.RATING_ID
	inner join CTC_RATING_VALUES t5 on t4.RATING_ID = t5.RATING_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
