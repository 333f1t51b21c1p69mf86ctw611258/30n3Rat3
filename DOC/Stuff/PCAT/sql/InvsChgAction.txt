select
	t1.CHG_ACTION_ID as chgActionId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_CHG_ACTION_REF t1
	inner join INVS_CHG_ACTION_VALUES t2 on t1.CHG_ACTION_ID = t2.CHG_ACTION_ID
where
	t2.LANGUAGE_CODE = {2}
