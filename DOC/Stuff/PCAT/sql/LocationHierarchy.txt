select
	t1.HIERARCHY_LEVEL_ID as hierarchyLevelId,
	t2.SERVICE_VERSION_ID as serviceVersionId,
	t2.INITIAL_VALUE as initialValue,
	t2.LENGTH as length,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	LOCATION_HIERARCHY_KEY t1
	inner join LOCATION_HIERARCHY_REF t2 on t1.HIERARCHY_LEVEL_ID = t2.HIERARCHY_LEVEL_ID
	inner join LOCATION_HIERARCHY_VALUES t3 on t2.HIERARCHY_LEVEL_ID = t3.HIERARCHY_LEVEL_ID
		and t2.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
where
	t2.SERVICE_VERSION_ID = {1}
	and t3.LANGUAGE_CODE = {2}
