select
	t1.DISCOUNT_EXCLUSION_ID as discountExclusionId,
	t4.DISPLAY_VALUE as rtDiscountKey,
	t1.RT_DISCOUNT_ID as rtDiscountId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t7.DISPLAY_VALUE as timeTypeKey,
	t1.TIME_TYPE_ID as timeTypeId
from
	DISCOUNT_EXCLUSION t1
	inner join RT_DISCOUNT_KEY t2 on t1.RT_DISCOUNT_ID = t2.RT_DISCOUNT_ID
	inner join RT_DISCOUNT_REF t3 on t2.RT_DISCOUNT_ID = t3.RT_DISCOUNT_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join RT_DISCOUNT_VALUES t4 on t2.RT_DISCOUNT_ID = t4.RT_DISCOUNT_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join TIME_TYPE_KEY t5 on t1.TIME_TYPE_ID = t5.TIME_TYPE_ID
	left outer join TIME_TYPE_REF t6 on t5.TIME_TYPE_ID = t6.TIME_TYPE_ID
		and t6.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join TIME_TYPE_VALUES t7 on t5.TIME_TYPE_ID = t7.TIME_TYPE_ID
		and t7.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
