select
	t3.DISPLAY_VALUE as position,
	t1.LR_CHRSTICS_NAME as lrChrsticsName,
	t1.SERVICE_VERSION_ID as serviceVersionId
from
	LOCATION_RELATION_CHRSTICS t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''POSITION'')
		and t2.value = t1.POSITION
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
