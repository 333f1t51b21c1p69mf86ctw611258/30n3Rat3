select
	t3.DISPLAY_VALUE as ttSubTypeRef,
	t1.TT_SUB_TYPE_ID as ttSubTypeId,
	t1.WP_PROCESS_ID as wpProcessId,
	t1.DURATION_UNITS as durationUnits,
	t1.PRE_PIVOT_DURATION as prePivotDuration,
	t1.POST_PIVOT_DURATION as postPivotDuration,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt
from
	TT_SUB_TYPE_WP_PROCESS_ID_MAP t1
	inner join TT_SUB_TYPE_REF t2 on t1.TT_SUB_TYPE_ID = t2.TT_SUB_TYPE_ID
	inner join TT_SUB_TYPE_VALUES t3 on t2.TT_SUB_TYPE_ID = t3.TT_SUB_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
