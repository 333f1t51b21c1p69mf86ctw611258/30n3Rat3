select
	t4.DISPLAY_VALUE as faListTemplateKey,
	t1.FA_LIST_TEMPLATE_ID as faListTemplateId,
	t6.SERVICE_VERSION_ID as serviceVersionId,
	t7.DISPLAY_VALUE as locationGroupKey,
	t1.LOCATION_GROUP_ID as locationGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	FA_TEMPLATE_LOCATION_GROUP_MAP t1
	inner join FA_LIST_TEMPLATE_KEY t2 on t1.FA_LIST_TEMPLATE_ID = t2.FA_LIST_TEMPLATE_ID
	inner join FA_LIST_TEMPLATE_REF t3 on t2.FA_LIST_TEMPLATE_ID = t3.FA_LIST_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join FA_LIST_TEMPLATE_VALUES t4 on t2.FA_LIST_TEMPLATE_ID = t4.FA_LIST_TEMPLATE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join LOCATION_GROUP_KEY t5 on t1.LOCATION_GROUP_ID = t5.LOCATION_GROUP_ID
	inner join LOCATION_GROUP_REF t6 on t5.LOCATION_GROUP_ID = t6.LOCATION_GROUP_ID
		and t6.SERVICE_VERSION_ID = {1}
	inner join LOCATION_GROUP_VALUES t7 on t5.LOCATION_GROUP_ID = t7.LOCATION_GROUP_ID
		and t7.SERVICE_VERSION_ID = t6.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
