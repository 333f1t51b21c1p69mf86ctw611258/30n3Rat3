select
	t1.BASE_ENTITY as baseEntity,
        t3.DISPLAY_VALUE as baseEntityDisplay,
	t1.ENTITY_TYPE as entityType,
	t5.DISPLAY_VALUE as paramDef,
	t1.PARAM_ID as paramId,
	t1.GROUPING_ID as groupingId,
	t1.ATTRIBUTE_DISPLAY_ORDER as attributeDisplayOrder
from
	EXT_PARAM_TYPE_ASSOC t1
	left outer join GENERIC_ENUMERATION_REF t2 on t2.enumeration_key = lower(''BASE_ENTITY'')
		and t2.value = t1.BASE_ENTITY
	left outer join GENERIC_ENUMERATION_VALUES t3 on t3.enumeration_key = t2.enumeration_key
		and t3.value = t2.value
		and t3.LANGUAGE_CODE = {2}
	inner join PARAM_DEF t4 on t1.PARAM_ID = t4.PARAM_ID
	inner join PARAM_VALUES t5 on t4.PARAM_ID = t5.PARAM_ID
		and t5.LANGUAGE_CODE = {2}
