select
	t3.DISPLAY_VALUE as invsTypeRef,
	t1.INVENTORY_TYPE_ID as inventoryTypeId,
	t5.DISPLAY_VALUE as invsEquipmentTxnRef,
	t1.EQUIPMENT_TRANSACTION_ID as equipmentTransactionId
from
	INVS_TYPE_EQUIP_TXN_ASSOC t1
	inner join INVS_TYPE_REF t2 on t1.INVENTORY_TYPE_ID = t2.INVENTORY_TYPE_ID
	inner join INVS_TYPE_VALUES t3 on t2.INVENTORY_TYPE_ID = t3.INVENTORY_TYPE_ID
		and t3.LANGUAGE_CODE = {2}
	inner join INVS_EQUIPMENT_TXN_REF t4 on t1.EQUIPMENT_TRANSACTION_ID = t4.EQUIPMENT_TRANSACTION_ID
	inner join INVS_EQUIPMENT_TXN_VALUES t5 on t4.EQUIPMENT_TRANSACTION_ID = t5.EQUIPMENT_TRANSACTION_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
