select
	t1.WORKORDER_TYPE as workorderType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	WO_WORKORDER_TYPE_REF t1
	inner join WO_WORKORDER_TYPE_VALUES t2 on t1.WORKORDER_TYPE = t2.WORKORDER_TYPE
where
	t2.LANGUAGE_CODE = {2}
