select
	t1.REGULATORY_ID as regulatoryId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	REGULATORY_ID_REF t1
	inner join REGULATORY_ID_VALUES t2 on t1.REGULATORY_ID = t2.REGULATORY_ID
where
	t2.LANGUAGE_CODE = {2}
