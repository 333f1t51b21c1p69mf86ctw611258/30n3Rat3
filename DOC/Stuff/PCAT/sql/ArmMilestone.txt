select
	t1.MILESTONE as milestone,
	t1.LOG as log,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	ARM_MILESTONE_REF t1
	inner join ARM_MILESTONE_VALUES t2 on t1.MILESTONE = t2.MILESTONE
where
	t2.LANGUAGE_CODE = {2}
