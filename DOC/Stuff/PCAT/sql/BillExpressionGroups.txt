select
	t3.DISPLAY_VALUE as billExpressionRef,
	t1.EXPRESSION_ID as expressionId,
	t5.DISPLAY_VALUE as billExpressionGroupRef,
	t1.EXPRESSION_GROUP_ID as expressionGroupId,
	t1.IS_INTERNAL as isInternal,
	t1.IS_HIDDEN as isHidden
from
	BILL_EXPRESSION_GROUPS t1
	inner join BILL_EXPRESSION_REF t2 on t1.EXPRESSION_ID = t2.EXPRESSION_ID
	inner join BILL_EXPRESSION_VALUES t3 on t2.EXPRESSION_ID = t3.EXPRESSION_ID
		and t3.LANGUAGE_CODE = {2}
	inner join BILL_EXPRESSION_GROUP_REF t4 on t1.EXPRESSION_GROUP_ID = t4.EXPRESSION_GROUP_ID
	inner join BILL_EXPRESSION_GROUP_VALUES t5 on t4.EXPRESSION_GROUP_ID = t5.EXPRESSION_GROUP_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
