select
	t1.MSC_ADDRESS_IVR_MAP_ID as mscAddressIvrMapId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.ACCESS_NUMBER as accessNumber,
	t1.MSC_ADDRESS as mscAddress,
	t1.IVR_DEST_NUMBER as ivrDestNumber,
	t1.IVR_NOA as ivrNoa
from
	MSC_ADDRESS_IVR_MAP t1
	inner join ACCESS_NUMBERS_KEY t2 on t1.ACCESS_NUMBER = t2.ACCESS_NUMBER
where
	t1.SERVICE_VERSION_ID = {1}
