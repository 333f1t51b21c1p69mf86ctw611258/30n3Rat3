select
	t1.PARAM_ID as paramId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t3.DISPLAY_VALUE as paramTag,
	t1.PARAM_VALUE as paramValue,
	t5.DISPLAY_VALUE as paramDatatype
from
	OSA_PARAMETER t1
	left outer join GUI_INDICATOR_REF t2 on t2.table_name = ''OSA_PARAMETER''
		and t2.field_name = lower(''PARAM_TAG_ID'')
		and t2.integer_value = t1.PARAM_TAG_ID
	left outer join GUI_INDICATOR_VALUES t3 on t3.table_name = t2.table_name
		and t3.field_name = t2.field_name
		and t3.integer_value = t2.integer_value
		and t3.LANGUAGE_CODE = {2}
	left outer join GENERIC_ENUMERATION_REF t4 on t4.enumeration_key = lower(''PARAM_DATATYPE'')
		and t4.value = t1.PARAM_DATATYPE
	left outer join GENERIC_ENUMERATION_VALUES t5 on t5.enumeration_key = t4.enumeration_key
		and t5.value = t4.value
		and t5.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
