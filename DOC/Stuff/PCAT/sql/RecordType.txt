select
	t1.RECORD_ID as recordId,
	t1.RECORD_TYPE as recordType,
	t1.SWITCH_ID as switchId,
	t1.ACTIVE_DT as activeDt,
	t1.INACTIVE_DT as inactiveDt,
	t1.TASK_NAME as taskName,
	t4.DISPLAY_VALUE as sourceType,
	t1.SERVER_TYPE as serverType,
	t1.RECORD_SIZE as recordSize,
	t1.IS_IGNORED as isIgnored,
	t1.IO_TYPE as ioType,
	t1.RECORD_CATEGORY as recordCategory,
	t1.OUTPUT_BY as outputBy,
	t1.NULL_TERMINATED as nullTerminated,
	t1.WHERE_CLAUSE as whereClause,
	t1.ORDER_CLAUSE as orderClause,
	t1.SET_QUERY as setQuery,
	t1.PROCESS_STATUS as processStatus,
	t1.PROCESS_NAME as processName,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.IS_VARIABLE_LENGTH as isVariableLength,
	t1.FIELD_DELIMITER_CHAR as fieldDelimiterChar,
	t1.FIELD_DELIMITER_CODE as fieldDelimiterCode,
	t1.RECORD_DELIMITER_CHAR as recordDelimiterChar,
	t1.RECORD_DELIMITER_CODE as recordDelimiterCode,
	t1.RECORD_DELIMITER_COUNT as recordDelimiterCount,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	RECORD_TYPE_REF t1
	inner join RECORD_TYPE_VALUES t2 on t1.RECORD_ID = t2.RECORD_ID
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''SOURCE_TYPE'')
		and t3.value = t1.SOURCE_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
