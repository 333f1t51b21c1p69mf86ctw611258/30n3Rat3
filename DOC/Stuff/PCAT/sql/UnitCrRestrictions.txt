select
	t1.UNIT_CR_RESTRICTION_ID as unitCrRestrictionId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t4.DISPLAY_VALUE as unitCrItemKey,
	t1.UNIT_CR_ID as unitCrId,
	t6.DISPLAY_VALUE as restrictionType,
	t1.RESTRICTED_ID as restrictedId,
	t1.IS_EXCLUSION as isExclusion,
	t8.DISPLAY_VALUE as restrictedDomain,
	t1.RESTRICTED_ID2 as restrictedId2
from
	UNIT_CR_RESTRICTIONS t1
	inner join UNIT_CR_ITEM_KEY t2 on t1.UNIT_CR_ID = t2.UNIT_CR_ID
	inner join UNIT_CR_ITEM_REF t3 on t2.UNIT_CR_ID = t3.UNIT_CR_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join UNIT_CR_ITEM_VALUES t4 on t2.UNIT_CR_ID = t4.UNIT_CR_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join GUI_INDICATOR_REF t5 on t5.table_name = ''UNIT_CR_RESTRICTIONS''
		and t5.field_name = lower(''RESTRICTION_TYPE'')
		and t5.integer_value = t1.RESTRICTION_TYPE
	left outer join GUI_INDICATOR_VALUES t6 on t6.table_name = t5.table_name
		and t6.field_name = t5.field_name
		and t6.integer_value = t5.integer_value
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GUI_INDICATOR_REF t7 on t7.table_name = ''UNIT_CR_RESTRICTIONS''
		and t7.field_name = lower(''RESTRICTED_DOMAIN'')
		and t7.integer_value = t1.RESTRICTED_DOMAIN
	left outer join GUI_INDICATOR_VALUES t8 on t8.table_name = t7.table_name
		and t8.field_name = t7.field_name
		and t8.integer_value = t7.integer_value
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
where
	t1.RESELLER_VERSION_ID = {0}
