select
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t3.ISO_CODE as currencyIsoCode,
	t3.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t4.DISPLAY_VALUE as rateCurrencyKey,
	t1.CURRENCY_CODE as currencyCode,
	t1.FORMAT_COUNT as formatCount,
	t1.SYMBOL as symbol,
	t1.DECIMAL_COUNT as decimalCount,
	t1.DECIMAL_STRING as decimalString,
	t1.SEPARATOR_COUNT as separatorCount,
	t1.SEPARATOR_STRING as separatorString,
	t6.DISPLAY_VALUE as position,
        t6.DISPLAY_VALUE as positionDisplay,
	t8.DISPLAY_VALUE as roundingMethod,
	t1.IS_DEFAULT as isDefault
from
	CURRENCY_LOCALE t1
	inner join RATE_CURRENCY_KEY t2 on t1.CURRENCY_CODE = t2.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t3 on t2.CURRENCY_CODE = t3.CURRENCY_CODE
		and t3.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t4 on t2.CURRENCY_CODE = t4.CURRENCY_CODE
		and t4.SERVICE_VERSION_ID = t3.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	left outer join GENERIC_ENUMERATION_REF t5 on t5.enumeration_key = lower(''POSITION'')
		and t5.value = t1.POSITION
	left outer join GENERIC_ENUMERATION_VALUES t6 on t6.enumeration_key = t5.enumeration_key
		and t6.value = t5.value
		and t6.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t7 on t7.enumeration_key = lower(''ROUNDING_METHOD'')
		and t7.value = t1.ROUNDING_METHOD
	left outer join GENERIC_ENUMERATION_VALUES t8 on t8.enumeration_key = t7.enumeration_key
		and t8.value = t7.value
		and t8.LANGUAGE_CODE = t4.LANGUAGE_CODE
