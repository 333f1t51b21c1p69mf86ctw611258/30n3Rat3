select
	t1.AUT_TRANS_SET_ID as autTransSetId,
	t5.DISPLAY_VALUE as expressionTemplateIdKey,
	t1.EXPRESSION_TEMPLATE_ID as expressionTemplateId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	AUT_TRANSLATION_SET_MAP t1
	inner join AUT_TRANSLATION_SET_KEY t2 on t1.AUT_TRANS_SET_ID = t2.AUT_TRANS_SET_ID
	inner join EXPRESSION_TEMPLATE_ID_KEY t3 on t1.EXPRESSION_TEMPLATE_ID = t3.EXPRESSION_TEMPLATE_ID
	inner join EXPRESSION_TEMPLATE_ID_REF t4 on t3.EXPRESSION_TEMPLATE_ID = t4.EXPRESSION_TEMPLATE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join EXPRESSION_TEMPLATE_ID_VALUES t5 on t3.EXPRESSION_TEMPLATE_ID = t5.EXPRESSION_TEMPLATE_ID
		and t5.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t5.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
