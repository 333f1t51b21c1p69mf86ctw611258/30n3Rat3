select
	t1.RATING_REASON_ID as ratingReasonId,
	t1.P_RATING_REASON_ID as pRatingReasonId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	CTC_RATING_REASON_REF t1
	inner join CTC_RATING_REASON_VALUES t2 on t1.RATING_REASON_ID = t2.RATING_REASON_ID
where
	t2.LANGUAGE_CODE = {2}
