select
	t1.FONT_ID as fontId,
	t1.TEST_FLAG as testFlag,
	t1.TAG as tag,
	t1.HP_SYMBOL_SET as hpSymbolSet,
	t4.DISPLAY_VALUE as hpSpacing,
	t1.HP_PITCH as hpPitch,
	t1.HP_HEIGHT as hpHeight,
	t6.DISPLAY_VALUE as hpStyle,
	t8.DISPLAY_VALUE as hpStrokeWeight,
	t1.HP_TYPEFACE as hpTypeface,
	t1.HP_UNDERLINE as hpUnderline,
	t10.DISPLAY_VALUE as hpDirection,
	t1.HP_OUTLINE as hpOutline,
	t1.AFP_TRCREF as afpTrcref,
	t1.AFP_PITCH as afpPitch,
	t1.AFP_CODE_PAGE as afpCodePage,
	t1.AFP_CHAR_SET as afpCharSet,
	t1.HTML_START_TAG as htmlStartTag,
	t1.HTML_END_TAG as htmlEndTag,
	t1.WIN_FACENAME as winFacename,
	t1.WIN_POINTSIZE as winPointsize,
	t1.WIN_HEIGHT as winHeight,
	t1.WIN_WEIGHT as winWeight,
	t1.WIN_ITALIC as winItalic,
	t1.WIN_UNDERLINE as winUnderline,
	t1.WIN_STRIKEOUT as winStrikeout,
	t12.DISPLAY_VALUE as winColor,
	t14.DISPLAY_VALUE as winDirection,
	t1.WIN_OUTLINE as winOutline,
	t1.FONT_FILE_ID as fontFileId,
	t1.PDF_TAG as pdfTag,
	t1.PDF_ENCODER as pdfEncoder,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	BILL_FONT_REF t1
	inner join BILL_FONT_VALUES t2 on t1.FONT_ID = t2.FONT_ID
		and t1.TEST_FLAG = t2.TEST_FLAG
	left outer join GENERIC_ENUMERATION_REF t3 on t3.enumeration_key = lower(''HP_SPACING'')
		and t3.value = t1.HP_SPACING
	left outer join GENERIC_ENUMERATION_VALUES t4 on t4.enumeration_key = t3.enumeration_key
		and t4.value = t3.value
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t5 on t5.enumeration_key = lower(''HP_STYLE'')
		and t5.value = t1.HP_STYLE
	left outer join GENERIC_ENUMERATION_VALUES t6 on t6.enumeration_key = t5.enumeration_key
		and t6.value = t5.value
		and t6.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t7 on t7.enumeration_key = lower(''HP_STROKE_WEIGHT'')
		and t7.value = t1.HP_STROKE_WEIGHT
	left outer join GENERIC_ENUMERATION_VALUES t8 on t8.enumeration_key = t7.enumeration_key
		and t8.value = t7.value
		and t8.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t9 on t9.enumeration_key = lower(''HP_DIRECTION'')
		and t9.value = t1.HP_DIRECTION
	left outer join GENERIC_ENUMERATION_VALUES t10 on t10.enumeration_key = t9.enumeration_key
		and t10.value = t9.value
		and t10.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t11 on t11.enumeration_key = lower(''WIN_COLOR'')
		and t11.value = t1.WIN_COLOR
	left outer join GENERIC_ENUMERATION_VALUES t12 on t12.enumeration_key = t11.enumeration_key
		and t12.value = t11.value
		and t12.LANGUAGE_CODE = t2.LANGUAGE_CODE
	left outer join GENERIC_ENUMERATION_REF t13 on t13.enumeration_key = lower(''WIN_DIRECTION'')
		and t13.value = t1.WIN_DIRECTION
	left outer join GENERIC_ENUMERATION_VALUES t14 on t14.enumeration_key = t13.enumeration_key
		and t14.value = t13.value
		and t14.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
