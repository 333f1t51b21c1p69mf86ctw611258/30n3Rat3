select
	t3.DISPLAY_VALUE as noteReasonRef,
	t1.NOTE_REASON_ID as noteReasonId,
	t5.DISPLAY_VALUE as noteTypeRef,
	t1.NOTE_TYPE_ID as noteTypeId
from
	NOTE_TYPE_REASON_MAP t1
	inner join NOTE_REASON_REF t2 on t1.NOTE_REASON_ID = t2.NOTE_REASON_ID
	inner join NOTE_REASON_VALUES t3 on t2.NOTE_REASON_ID = t3.NOTE_REASON_ID
		and t3.LANGUAGE_CODE = {2}
	inner join NOTE_TYPE_REF t4 on t1.NOTE_TYPE_ID = t4.NOTE_TYPE_ID
	inner join NOTE_TYPE_VALUES t5 on t4.NOTE_TYPE_ID = t5.NOTE_TYPE_ID
		and t5.LANGUAGE_CODE = t3.LANGUAGE_CODE
