select
	t1.CURRENCY_TYPE as currencyType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	CURRENCY_TYPE_REF t1
	inner join CURRENCY_TYPE_VALUES t2 on t1.CURRENCY_TYPE = t2.CURRENCY_TYPE
where
	t2.LANGUAGE_CODE = {2}
