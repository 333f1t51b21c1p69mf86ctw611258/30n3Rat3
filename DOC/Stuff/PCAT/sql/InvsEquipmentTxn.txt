select
	t1.EQUIPMENT_TRANSACTION_ID as equipmentTransactionId,
	t1.SUPPORTS_MULTIPLE_ITEMS as supportsMultipleItems,
	t1.SUPPORTS_SINGLE_ITEMS as supportsSingleItems,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_EQUIPMENT_TXN_REF t1
	inner join INVS_EQUIPMENT_TXN_VALUES t2 on t1.EQUIPMENT_TRANSACTION_ID = t2.EQUIPMENT_TRANSACTION_ID
where
	t2.LANGUAGE_CODE = {2}
