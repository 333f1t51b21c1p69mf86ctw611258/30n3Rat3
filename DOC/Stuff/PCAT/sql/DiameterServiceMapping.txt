select
    t1.DIAMETER_SERVICE_MAPPING_ID as diameterServiceMappingId,
	t4.DISPLAY_VALUE as applicationIdKey,
    t1.APPLICATION_ID as applicationId,
	t7.DISPLAY_VALUE as subTypeIdKey,
    t1.SUB_TYPE_ID as subTypeId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
    t8.PARAM_VALUE as serviceContextId,
    t9.MERCHANT_NAME as diameterAcctId,
    t10.PARAM_VALUE as serviceType,
    t11.PARAM_VALUE as serviceSubType,
    t12.PARAM_VALUE as qos,
    t13.PARAM_VALUE as custom1,
    t14.PARAM_VALUE as custom2,
    t15.PARAM_VALUE as custom3,
    t16.PARAM_VALUE as custom4,
    t17.PARAM_VALUE as custom5
from
    DIAMETER_SERVICE_MAPPING t1
	inner join APPLICATION_ID_KEY t2 on t1.APPLICATION_ID = t2.APPLICATION_ID
	inner join APPLICATION_ID_REF t3 on t2.APPLICATION_ID = t3.APPLICATION_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join APPLICATION_ID_VALUES t4 on t2.APPLICATION_ID = t4.APPLICATION_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join SUB_TYPE_ID_KEY t5 on t1.SUB_TYPE_ID = t5.SUB_TYPE_ID
	inner join SUB_TYPE_ID_REF t6 on t5.SUB_TYPE_ID = t6.SUB_TYPE_ID
		and t6.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join SUB_TYPE_ID_VALUES t7 on t5.SUB_TYPE_ID = t7.SUB_TYPE_ID
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
    left outer join DIAMETER_PARAMETER t8 on t1.SERVICE_CONTEXT_ID = t8.param_id
                and t1.SERVICE_VERSION_ID =t8.SERVICE_VERSION_ID
    left outer join DIAMETER_MERCHANT_ACCT_ID t9 on t1.DIAMETER_ACCT_ID = t9.DIAMETER_ACCT_ID
                and t1.SERVICE_VERSION_ID =t9.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t10 on t1.SERVICE_TYPE = t10.param_id
                and t1.SERVICE_VERSION_ID =t10.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t11 on t1.SERVICE_SUB_TYPE = t11.param_id
                and t1.SERVICE_VERSION_ID =t11.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t12 on t1.qos = t12.param_id
                and t1.SERVICE_VERSION_ID =t12.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t13 on t1.CUSTOM1 = t13.param_id
                and t1.SERVICE_VERSION_ID =t13.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t14 on t1.CUSTOM2 = t14.param_id
                and t1.SERVICE_VERSION_ID =t14.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t15 on t1.CUSTOM3 = t15.param_id
                and t1.SERVICE_VERSION_ID =t15.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t16 on t1.CUSTOM4 = t16.param_id
                and t1.SERVICE_VERSION_ID =t16.SERVICE_VERSION_ID
    left outer join DIAMETER_PARAMETER t17 on t1.CUSTOM5 = t17.param_id
                and t1.SERVICE_VERSION_ID =t17.SERVICE_VERSION_ID
where
    t1.SERVICE_VERSION_ID = {1}
