select
	t1.TAX_ASSIGNMENT_ID as taxAssignmentId,
	t1.RESELLER_VERSION_ID as resellerVersionId,
	t3.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as accountCategoryKey,
	t1.ACCOUNT_CATEGORY as accountCategory,
	t1.BRAND_ID as brandId,
	t6.DISPLAY_VALUE as countryCodeRef,
	t1.COUNTRY_CODE as countryCode,
	t1.STATE_CODE as stateCode,
	t1.COUNTY_CODE as countyCode,
	t1.CITY_CODE as cityCode,
	t9.DISPLAY_VALUE as franchiseCodeKey,
	t1.FRANCHISE_CODE as franchiseCode,
	t11.DISPLAY_VALUE as chargeEltType,
	t1.CHARGE_ELT_ID as chargeEltId,
	t14.DISPLAY_VALUE as serviceProvidersKey,
	t1.PROVIDER_ID as providerId,
	t1.TAX_PKG_INST_ID as taxPkgInstId,
        t17.DISPLAY_VALUE as taxPkgInstIdKey
from
	TAX_ASSIGNMENTS t1
	left outer join ACCOUNT_CATEGORY_KEY t2 on t1.ACCOUNT_CATEGORY = t2.ACCOUNT_CATEGORY
	left outer join ACCOUNT_CATEGORY_REF t3 on t2.ACCOUNT_CATEGORY = t3.ACCOUNT_CATEGORY
		and t3.SERVICE_VERSION_ID = {1}
	left outer join ACCOUNT_CATEGORY_VALUES t4 on t2.ACCOUNT_CATEGORY = t4.ACCOUNT_CATEGORY
		and t4.SERVICE_VERSION_ID = {1}
		and t4.LANGUAGE_CODE = {2}
	left outer join COUNTRY_CODE_REF t5 on t1.COUNTRY_CODE = t5.COUNTRY_CODE
	left outer join COUNTRY_CODE_VALUES t6 on t5.COUNTRY_CODE = t6.COUNTRY_CODE
		and t6.LANGUAGE_CODE = {2}
	left outer join FRANCHISE_CODE_KEY t7 on t1.FRANCHISE_CODE = t7.FRANCHISE_CODE
	left outer join FRANCHISE_CODE_REF t8 on t7.FRANCHISE_CODE = t8.FRANCHISE_CODE
		and t8.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	left outer join FRANCHISE_CODE_VALUES t9 on t7.FRANCHISE_CODE = t9.FRANCHISE_CODE
		and t9.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t9.LANGUAGE_CODE = {2}
	left outer join GENERIC_ENUMERATION_REF t10 on t10.enumeration_key = lower(''CHARGE_ELT_TYPE'')
		and t10.value = t1.CHARGE_ELT_TYPE
	left outer join GENERIC_ENUMERATION_VALUES t11 on t11.enumeration_key = t10.enumeration_key
		and t11.value = t10.value
		and t11.LANGUAGE_CODE = {2}
	left outer join SERVICE_PROVIDERS_KEY t12 on t1.PROVIDER_ID = t12.PROVIDER_ID
	left outer join SERVICE_PROVIDERS_REF t13 on t12.PROVIDER_ID = t13.PROVIDER_ID
		and t13.SERVICE_VERSION_ID = {1}
	left outer join SERVICE_PROVIDERS_VALUES t14 on t12.PROVIDER_ID = t14.PROVIDER_ID
		and t14.SERVICE_VERSION_ID = {1}
		and t14.LANGUAGE_CODE = {2}
        inner join TAX_PKG_INST_ID_KEY t15 on t1.TAX_PKG_INST_ID = t15.TAX_PKG_INST_ID
        inner join TAX_PKG_INST_ID_REF t16 on t1.TAX_PKG_INST_ID = t16.TAX_PKG_INST_ID
                and t1.RESELLER_VERSION_ID = t16.RESELLER_VERSION_ID
        inner join TAX_PKG_INST_ID_VALUES t17 on t1.TAX_PKG_INST_ID = t17.TAX_PKG_INST_ID
                and t1.RESELLER_VERSION_ID = t17.RESELLER_VERSION_ID
                and t17.LANGUAGE_CODE = {2}
where
	t1.RESELLER_VERSION_ID = {0}
