select
	t1.ELIG_ATTR_ID as eligAttrId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t1.ELIG_ATTR_NAME as eligAttrName,
	t3.DISPLAY_VALUE as eligAttrDatatype,
	t1.BASE_TABLE as baseTable,
	t1.CAN_CONFIGURE_ELIGIBILITY as canConfigureEligibility,
	t1.CAN_CONFIGURE_SEGMENTATION_KEY as canConfigureSegmentationKey,
	t1.USE_FOR_ELIGIBILITY as useForEligibility,
	t1.USE_FOR_SEGMENTATION_KEY as useForSegmentationKey
from
	ELIGIBILITY_ATTRIBUTE_DEF t1
        left outer join GUI_INDICATOR_REF t2 on t2.table_name = ''ELIGIBILITY_ATTRIBUTE_DEF''
		and t2.field_name = lower(''ELIG_ATTR_DATATYPE'')
		and t2.integer_value = t1.ELIG_ATTR_DATATYPE
	left outer join GUI_INDICATOR_VALUES t3 on t3.table_name = t2.table_name
		and t3.field_name = t2.field_name
                and t3.integer_value = t2.integer_value
		and t3.LANGUAGE_CODE = {2}
where
	t1.SERVICE_VERSION_ID = {1}
