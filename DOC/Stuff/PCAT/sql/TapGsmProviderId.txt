select
	t1.TAP_GSM_PROV_ID as tapGsmProvId,
	t1.TAP_GSM_PROVIDER_ID as tapGsmProviderId,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue,
	t2.DESCRIPTION as description
from
	TAP_GSM_PROVIDER_ID_REF t1
	inner join TAP_GSM_PROVIDER_ID_VALUES t2 on t1.TAP_GSM_PROV_ID = t2.TAP_GSM_PROV_ID
where
	t2.LANGUAGE_CODE = {2}
