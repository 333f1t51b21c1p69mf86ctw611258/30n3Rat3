select
	t1.OWNING_COST_CTR as owningCostCtr,
	t1.G_L_CODE as GLCode,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	OWNING_COST_CTR_REF t1
	inner join OWNING_COST_CTR_VALUES t2 on t1.OWNING_COST_CTR = t2.OWNING_COST_CTR
where
	t2.LANGUAGE_CODE = {2}
