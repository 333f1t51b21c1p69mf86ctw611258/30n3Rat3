select
	t1.MCC as mcc,
	t1.COUNTRY_CODE as countryCode, 
        t3.display_value as countryCodeRef 
from
	MCC_COUNTRY_CODES t1 
        inner join COUNTRY_CODE_REF t2 on t1.COUNTRY_CODE = t2.COUNTRY_CODE 
        inner join COUNTRY_CODE_VALUES t3 on t2.COUNTRY_CODE = t3.COUNTRY_CODE 
          and t3.LANGUAGE_CODE = {2}
