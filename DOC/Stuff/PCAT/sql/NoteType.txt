select
	t1.NOTE_TYPE_ID as noteTypeId,
	t1.IS_GROUPED as isGrouped,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	NOTE_TYPE_REF t1
	inner join NOTE_TYPE_VALUES t2 on t1.NOTE_TYPE_ID = t2.NOTE_TYPE_ID
where
	t2.LANGUAGE_CODE = {2}
