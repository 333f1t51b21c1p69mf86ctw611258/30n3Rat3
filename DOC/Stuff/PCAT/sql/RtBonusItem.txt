select
	t1.BONUS_ITEM_ID as bonusItemId,
	t2.RESELLER_VERSION_ID as resellerVersionId,
	t6.DISPLAY_VALUE as accumulatorKey,
	t2.ACCUMULATOR_ID as accumulatorId,
	t8.SERVICE_VERSION_ID as serviceVersionId,
	t8.ISO_CODE as currencyIsoCode,
	t8.IMPLIED_DECIMAL as currencyImpliedDecimal,
	t9.DISPLAY_VALUE as rateCurrencyKey,
	t2.CURRENCY_CODE as currencyCode,
	t12.DISPLAY_VALUE as accumulatorKeyByAddonAccId1,
	t2.ADDON_ACC_ID1 as addonAccId1,
	t15.DISPLAY_VALUE as accumulatorKeyByAddonAccId2,
	t2.ADDON_ACC_ID2 as addonAccId2,
	t18.DISPLAY_VALUE as accumulatorKeyByAddonAccId3,
	t2.ADDON_ACC_ID3 as addonAccId3,
	t21.DISPLAY_VALUE as accumulatorKeyByAddonAccId4,
	t2.ADDON_ACC_ID4 as addonAccId4,
	t2.RESET_FLAG as resetFlag,
	t2.IS_INTERNAL as isInternal,
	t2.IS_DEFAULT as isDefault,
	t2.AWARD_TRIGGER as awardTrigger,
	t2.CALC_DATE_BASED_ON as calcDateBasedOn,
	t2.USE_ACTIVE_DAYS as useActiveDays,
	t2.USE_BONUS_CYCLES as useBonusCycles,
	t3.LANGUAGE_CODE as languageCode,
	t3.DISPLAY_VALUE as displayValue,
	t3.DESCRIPTION as description
from
	RT_BONUS_ITEM_KEY t1
	inner join RT_BONUS_ITEM_REF t2 on t1.BONUS_ITEM_ID = t2.BONUS_ITEM_ID
	inner join RT_BONUS_ITEM_VALUES t3 on t2.BONUS_ITEM_ID = t3.BONUS_ITEM_ID
		and t2.RESELLER_VERSION_ID = t3.RESELLER_VERSION_ID
	left outer join ACCUMULATOR_KEY t4 on t2.ACCUMULATOR_ID = t4.ACCUMULATOR_ID
	left outer join ACCUMULATOR_REF t5 on t4.ACCUMULATOR_ID = t5.ACCUMULATOR_ID
		and t5.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join ACCUMULATOR_VALUES t6 on t4.ACCUMULATOR_ID = t6.ACCUMULATOR_ID
		and t6.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t6.LANGUAGE_CODE = t3.LANGUAGE_CODE
	inner join RATE_CURRENCY_KEY t7 on t2.CURRENCY_CODE = t7.CURRENCY_CODE
	inner join RATE_CURRENCY_REF t8 on t7.CURRENCY_CODE = t8.CURRENCY_CODE
		and t8.SERVICE_VERSION_ID = {1}
	inner join RATE_CURRENCY_VALUES t9 on t7.CURRENCY_CODE = t9.CURRENCY_CODE
		and t9.SERVICE_VERSION_ID = t8.SERVICE_VERSION_ID
		and t9.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ACCUMULATOR_KEY t10 on t2.ADDON_ACC_ID1 = t10.ACCUMULATOR_ID
	left outer join ACCUMULATOR_REF t11 on t10.ACCUMULATOR_ID = t11.ACCUMULATOR_ID
		and t11.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join ACCUMULATOR_VALUES t12 on t10.ACCUMULATOR_ID = t12.ACCUMULATOR_ID
		and t12.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t12.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ACCUMULATOR_KEY t13 on t2.ADDON_ACC_ID2 = t13.ACCUMULATOR_ID
	left outer join ACCUMULATOR_REF t14 on t13.ACCUMULATOR_ID = t14.ACCUMULATOR_ID
		and t14.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join ACCUMULATOR_VALUES t15 on t13.ACCUMULATOR_ID = t15.ACCUMULATOR_ID
		and t15.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t15.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ACCUMULATOR_KEY t16 on t2.ADDON_ACC_ID3 = t16.ACCUMULATOR_ID
	left outer join ACCUMULATOR_REF t17 on t16.ACCUMULATOR_ID = t17.ACCUMULATOR_ID
		and t17.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join ACCUMULATOR_VALUES t18 on t16.ACCUMULATOR_ID = t18.ACCUMULATOR_ID
		and t18.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t18.LANGUAGE_CODE = t3.LANGUAGE_CODE
	left outer join ACCUMULATOR_KEY t19 on t2.ADDON_ACC_ID4 = t19.ACCUMULATOR_ID
	left outer join ACCUMULATOR_REF t20 on t19.ACCUMULATOR_ID = t20.ACCUMULATOR_ID
		and t20.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
	left outer join ACCUMULATOR_VALUES t21 on t19.ACCUMULATOR_ID = t21.ACCUMULATOR_ID
		and t21.RESELLER_VERSION_ID = t2.RESELLER_VERSION_ID
		and t21.LANGUAGE_CODE = t3.LANGUAGE_CODE
where
	t2.RESELLER_VERSION_ID = {0}
	and t3.LANGUAGE_CODE = {2}
