select
	t1.JOBSUBSTATE_ID as jobsubstateId,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	INVS_JOBSUBSTATE_REF t1
	inner join INVS_JOBSUBSTATE_VALUES t2 on t1.JOBSUBSTATE_ID = t2.JOBSUBSTATE_ID
where
	t2.LANGUAGE_CODE = {2}
