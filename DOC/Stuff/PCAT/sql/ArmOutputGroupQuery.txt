select
	t3.DISPLAY_VALUE as armOutputGroupRef,
	t1.OUTPUT_GROUP_ID as outputGroupId,
	t1.NUMBER_OF_FIELDS as numberOfFields,
	t1.ORDER_OF_FIELDS as orderOfFields,
	t1.TAG_NAMES as tagNames,
	t1.SQL_QUERY as sqlQuery
from
	ARM_OUTPUT_GROUP_QUERY t1
	inner join ARM_OUTPUT_GROUP_REF t2 on t1.OUTPUT_GROUP_ID = t2.OUTPUT_GROUP_ID
	inner join ARM_OUTPUT_GROUP_VALUES t3 on t2.OUTPUT_GROUP_ID = t3.OUTPUT_GROUP_ID
		and t3.LANGUAGE_CODE = {2}
