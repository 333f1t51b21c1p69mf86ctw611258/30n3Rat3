select
	t4.DISPLAY_VALUE as expressionTemplateIdKey,
	t1.EXPRESSION_TEMPLATE_ID as expressionTemplateId,
	t1.EXPRESSION_GROUP_ID as expressionGroupId,
	t1.RESELLER_VERSION_ID as resellerVersionId
from
	EXPRESSION_TEMPLATE t1
	inner join EXPRESSION_TEMPLATE_ID_KEY t2 on t1.EXPRESSION_TEMPLATE_ID = t2.EXPRESSION_TEMPLATE_ID
	inner join EXPRESSION_TEMPLATE_ID_REF t3 on t2.EXPRESSION_TEMPLATE_ID = t3.EXPRESSION_TEMPLATE_ID
		and t3.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
	inner join EXPRESSION_TEMPLATE_ID_VALUES t4 on t2.EXPRESSION_TEMPLATE_ID = t4.EXPRESSION_TEMPLATE_ID
		and t4.RESELLER_VERSION_ID = t1.RESELLER_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join EXPRESSION_GROUP_KEY t5 on t1.EXPRESSION_GROUP_ID = t5.EXPRESSION_GROUP_ID
where
	t1.RESELLER_VERSION_ID = {0}
