select
	t1.RESPONSE_CODE as responseCode,
	t1.RESPONSE_CATEGORY as responseCategory,
	t4.DISPLAY_VALUE as clearingHouseIdRef,
	t1.CLEARING_HOUSE_ID as clearingHouseId,
	t1.TREATMENT_IND as treatmentInd,
	t1.REVERSAL_TRANS_TYPE as reversalTransType,
	t1.IS_DEFAULT as isDefault,
	t1.IS_INTERNAL as isInternal,
	t1.EFT_EVENT_TYPE as eftEventType,
	t2.LANGUAGE_CODE as languageCode,
	t2.DISPLAY_VALUE as displayValue
from
	EFT_RESPONSE_CODE_REF t1
	inner join EFT_RESPONSE_CODE_VALUES t2 on t1.RESPONSE_CODE = t2.RESPONSE_CODE
		and t1.CLEARING_HOUSE_ID = t2.CLEARING_HOUSE_ID
	inner join CLEARING_HOUSE_ID_REF t3 on t1.CLEARING_HOUSE_ID = t3.CLEARING_HOUSE_ID
	inner join CLEARING_HOUSE_ID_VALUES t4 on t3.CLEARING_HOUSE_ID = t4.CLEARING_HOUSE_ID
		and t4.LANGUAGE_CODE = t2.LANGUAGE_CODE
where
	t2.LANGUAGE_CODE = {2}
