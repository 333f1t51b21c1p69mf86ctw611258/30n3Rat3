select
	t1.DIAM_FLEX_MAPPING_ID as diamFlexMappingId,
	t1.SERVICE_VERSION_ID as serviceVersionId,
	t4.DISPLAY_VALUE as applicationIdKey,
	t1.APPLICATION_ID as applicationId,
	t7.DISPLAY_VALUE as subTypeIdKey,
	t1.SUB_TYPE_ID as subTypeId,
	t1.SERVICE_CONTEXT_ID as serviceContextId,
	t10.DISPLAY_VALUE as diamFlexsleServiceKey,
	t1.SLE_SERVICE_ID as sleServiceId,
	t1.PARAMETER1 as parameter1,
	t1.PARAMETER2 as parameter2,
	t1.PARAMETER3 as parameter3,
	t1.PARAMETER4 as parameter4,
	t1.PARAMETER5 as parameter5,
	t1.PARAMETER6 as parameter6,
	t1.PARAMETER7 as parameter7,
	t1.PARAMETER8 as parameter8,
	t11.PARAM_VALUE as serviceContextIdDisplayValue,
	t12.PARAM_VALUE as parameter1DisplayValue,
	t13.PARAM_VALUE as parameter2DisplayValue,
	t14.PARAM_VALUE as parameter3DisplayValue,
	t15.PARAM_VALUE as parameter4DisplayValue,
	t16.PARAM_VALUE as parameter5DisplayValue,
	t17.PARAM_VALUE as parameter6DisplayValue,
	t18.PARAM_VALUE as parameter7DisplayValue,
	t19.PARAM_VALUE as parameter8DisplayValue
from
	DIAM_FLEXSLE_SERV_MAPPING t1
	inner join APPLICATION_ID_KEY t2 on t1.APPLICATION_ID = t2.APPLICATION_ID
	inner join APPLICATION_ID_REF t3 on t2.APPLICATION_ID = t3.APPLICATION_ID
		and t3.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join APPLICATION_ID_VALUES t4 on t2.APPLICATION_ID = t4.APPLICATION_ID
		and t4.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t4.LANGUAGE_CODE = {2}
	inner join SUB_TYPE_ID_KEY t5 on t1.SUB_TYPE_ID = t5.SUB_TYPE_ID
	inner join SUB_TYPE_ID_REF t6 on t5.SUB_TYPE_ID = t6.SUB_TYPE_ID
		and t6.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join SUB_TYPE_ID_VALUES t7 on t5.SUB_TYPE_ID = t7.SUB_TYPE_ID
		and t7.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t7.LANGUAGE_CODE = t4.LANGUAGE_CODE
	inner join DIAM_FLEXSLE_SERVICE_KEY t8 on t1.SLE_SERVICE_ID = t8.SLE_SERVICE_ID
	inner join DIAM_FLEXSLE_SERVICE_REF t9 on t8.SLE_SERVICE_ID = t9.SLE_SERVICE_ID
		and t9.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	inner join DIAM_FLEXSLE_SERVICE_VALUES t10 on t8.SLE_SERVICE_ID = t10.SLE_SERVICE_ID
		and t10.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
		and t10.LANGUAGE_CODE = t4.LANGUAGE_CODE
	left outer join DIAM_FLEXSLE_PARAMETER t11 on t1.SERVICE_CONTEXT_ID = t11.PARAM_ID
		and t11.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t12 on t1.PARAMETER1 = t12.PARAM_ID
		and t12.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t13 on t1.PARAMETER2 = t13.PARAM_ID
		and t13.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t14 on t1.PARAMETER3 = t14.PARAM_ID
		and t14.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t15 on t1.PARAMETER4 = t15.PARAM_ID
		and t15.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t16 on t1.PARAMETER5 = t16.PARAM_ID
		and t16.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t17 on t1.PARAMETER6 = t17.PARAM_ID
		and t17.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t18 on t1.PARAMETER7 = t18.PARAM_ID
		and t18.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
	left outer join DIAM_FLEXSLE_PARAMETER t19 on t1.PARAMETER8 = t19.PARAM_ID
		and t19.SERVICE_VERSION_ID = t1.SERVICE_VERSION_ID
where
	t1.SERVICE_VERSION_ID = {1}
